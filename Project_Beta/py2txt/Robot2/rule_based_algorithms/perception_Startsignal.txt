# perception_startsignal.py
# Detects red lamp pattern from a given RGB image (PIL) to determine race start.
#
# PURPOSE:
#   Monitors the camera feed to detect the F1-style starting light sequence.
#   Unity shows 3 red start lamps in a row. This module:
#     1. Detects when all 3 lamps turn ON (red pixels appear in 3 regions)
#     2. Sets an internal "ready_to_go" flag
#     3. Returns True exactly once when all 3 lamps turn OFF (race start signal)
#
# DETECTION ALGORITHM:
#   - Divide the upper portion of the image (top 30% height) into 3 horizontal regions
#   - In each region, count pixels where R is high AND G and B are low (red detection)
#   - If red pixel ratio > 0.03 (3%) in a region → that lamp is ON
#   - State machine:
#       All 3 ON  → set ready_to_go = True, return False
#       All 3 OFF after ready_to_go was True → set ready_to_go = False, return True (GO!)
#       Intermediate states (1 or 2 lamps ON) → return False
#
# LAMP REGION LAYOUT (horizontal, top 30% of image):
#   Lamp 1: x = 35% ~ 50% of width
#   Lamp 2: x = 55% ~ 70% of width
#   Lamp 3: x = 75% ~ 90% of width
#
# STATE NOTE:
#   detect_start_signal() uses function-level attribute storage (ready_to_go)
#   so it persists between calls without needing a class instance.
#   This means the state is reset only by module reload (e.g. importlib) or by
#   manually setting detect_start_signal.ready_to_go = False.
#
# IMPORTANT:
#   This function must be loaded via importlib when used simultaneously for
#   Robot1 and Robot2 (to avoid sharing the ready_to_go state between robots).
#   See ai_control_strategy.py and rule_based_input.py for the importlib pattern.

from PIL import Image

def is_red(pixel, red_thresh=140, green_thresh=130, blue_thresh=130):
    """
    Returns True if the given pixel is 'red'.

    Red pixel criteria:
      R > red_thresh   (strong red component)
      G < green_thresh (low green)
      B < blue_thresh  (low blue)

    Default thresholds (140/130/130) were calibrated for the Unity start lamp color.
    Adjust if detection is unreliable in different lighting conditions.
    """
    r, g, b = pixel
    return r > red_thresh and g < green_thresh and b < blue_thresh

def detect_start_signal(img):
    """
    Analyze a given PIL image to detect red start lamps.
    Returns True only once right after all lamps turn off (after being lit).

    Args:
        img: PIL Image (RGB)

    Returns:
        True: Race start signal detected (GO!) — returned exactly once per race
        False: Still waiting (lamps not yet lit, or currently lit, or already triggered)

    State machine (stored as function attribute):
      ready_to_go = False initially
      → All 3 lamps ON: set ready_to_go = True, return False
      → All 3 lamps OFF while ready_to_go: set ready_to_go = False, return True (fire once!)
      → Any other state: return False
    """
    # Initialize state on first call
    if not hasattr(detect_start_signal, 'ready_to_go'):
        detect_start_signal.ready_to_go = False

    try:
        width, height = img.size
        top = 0
        bottom = int(height * 0.3)   # Only check top 30% of image

        # Define 3 rectangular lamp regions (horizontal positions, same vertical band)
        lamp_positions = [
            (int(width * 0.35), int(width * 0.5)),   # Lamp 1
            (int(width * 0.55), int(width * 0.7)),   # Lamp 2
            (int(width * 0.75), int(width * 0.9))    # Lamp 3
        ]

        red_count = 0
        for left, right in lamp_positions:
            red_pixels = 0
            total_pixels = 0
            for y in range(top, bottom):
                for x in range(left, right):
                    pixel = img.getpixel((x, y))
                    if is_red(pixel):
                        red_pixels += 1
                    total_pixels += 1
            ratio = red_pixels / total_pixels
            # Count this lamp as ON if red pixel ratio exceeds threshold
            if ratio > 0.03:
                red_count += 1

        # Optional debug mode: save lamp ROI visualization for calibration
        DEBUG_MODE = False
        if DEBUG_MODE:
            from PIL import ImageDraw
            debug_img = img.copy()
            draw = ImageDraw.Draw(debug_img)
            for left, right in lamp_positions:
                draw.rectangle([left, top, right, bottom], outline="red", width=2)
            debug_img.save("debug_lamps.jpg")
            print("[StartSignal] Saved debug_lamps.jpg")

        # State machine transitions
        # All 3 red lights ON → mark as ready
        if red_count == 3:
            detect_start_signal.ready_to_go = True
            return False

        # All lights OFF after being ready → fire GO! signal
        if red_count == 0 and detect_start_signal.ready_to_go:
            print("[StartSignal] GO!!")
            detect_start_signal.ready_to_go = False
            return True   # This is the actual "start racing" trigger

        return False

    except Exception as e:
        print(f"[StartSignal] Error: {e}")
        return False
