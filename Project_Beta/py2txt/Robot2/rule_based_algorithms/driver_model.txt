# rule_based_algorithms/driver_model.py
# ----------------------------------------------------------------------
# STEER-TYPE CONTROL VERSION
# Higher layer decides lane_mode ("normal" / "hold" / "search") and lost_age.
# This driver computes: (drive_torque, steer_angle) for steer-type robots.
#
# Sign conventions:
#   lateral_px : right is + [px] (offset from image center)
#   theta_deg  : 0° is up, tilting to the right is + [deg]
#   Output:
#     drive_torque : forward torque (-1.0 to +1.0, + is forward)
#     steer_angle  : steering angle in radians (+ is right turn)
#
# ARCHITECTURE OVERVIEW:
# ========================
# DriverConfig  — dataclass of all tunable parameters (passed to constructor)
# DriverModel   — stateful driver object, one instance per robot
#   .update()   — main method, called each frame, returns (drive_torque, steer_angle)
#   .last_debug — dict of per-frame debug values (read by debug overlay saving code)
#
# DRIVING MODES (determined by caller rule_based_input.py):
#   "normal"  — Full lane detection available. Uses lateral_px and theta_deg to
#               compute steering. Speed reduced proportionally to steer angle (curve slowdown).
#               Pulse control engaged when |theta| or |lateral| exceeds threshold.
#   "hold"    — Lane temporarily lost. Decay the last known base speed and steer angle
#               by hold_decay_per_frame ** lost_age (exponential decay toward zero).
#               Prevents abrupt stop when white lane briefly disappears.
#   "search"  — Lane lost for extended time (lost_age > HOLD_FRAMES). Pivot in place
#               (drive=0, constant right steer) to find the lane again.
#
# PULSE CONTROL (mimics "tap-tap" throttle on keyboard):
#   When in a sharp curve (|theta| > pulse_theta_threshold or |lateral| > pulse_lateral_threshold),
#   alternate between throttle ON and OFF in short bursts instead of continuous drive.
#   This prevents accumulating too much speed entering curves.
#   ON  phase: pulse_drive_on  (lower speed, e.g. 0.35)
#   OFF phase: pulse_drive_off (coast, 0.0)
#   Pulse disabled when single_side=True (estimated lane is less accurate).
#
# SOC SCALING:
#   scale = soc_floor + (1.0 - soc_floor) * soc
#   When SOC drops toward 0, scale → soc_floor (e.g. 0.30), reducing speed by 70%.
#   Prevents the robot from running out of battery while still moving forward.
# ----------------------------------------------------------------------

from __future__ import annotations
from dataclasses import dataclass
from typing import Optional, Tuple, Dict
import math

def _clip(x: float, lo: float, hi: float) -> float:
    """Clamp x into [lo, hi]."""
    return lo if x < lo else hi if x > hi else x


@dataclass
class DriverConfig:
    """
    All tunable parameters for DriverModel.
    Edit these values to change driving behavior.
    Robot1 and Robot2 may use different DriverConfig instances.
    """
    # Geometry / normalization
    image_width: int = 224
    lateral_norm_halfwidth_px: Optional[float] = None  # None → image_width/2

    # Speed planning (auto slow-down in curves)
    v_min: float = 0.15             # Minimum drive torque (prevents stopping mid-curve)
    v_max: float = 0.55             # Maximum drive torque (reduced from 0.70 for stability)
    slow_w_theta: float = 0.70      # Weight of theta in speed reduction formula (legacy, not used)
    slow_w_lateral: float = 0.60    # Weight of lateral in speed reduction (legacy, not used)

    # Steering calculation: steer = k_theta * theta_rad + k_lateral * lateral_norm
    # Note: Reduced gains for torque-steer type (was differential drive gains)
    k_theta: float = 0.45          # Gain for theta (heading error). Was 0.90 for diff drive.
    k_lateral: float = 0.30        # Gain for lateral offset. Was 0.60 for diff drive.
    steer_limit: float = 0.524     # Max steer angle [rad] (~30 deg, matches Unity maxSteerAngleDeg)

    # Output shaping
    torque_limit: float = 1.00     # Hard limit on drive torque output
    alpha_smooth: float = 0.50     # IIR smoothing coefficient: output = (1-α)*new + α*prev
                                   # Higher α = more smoothing, but more lag

    # Battery (SOC) scaling
    use_soc_scaling: bool = True   # Enable/disable SOC-based speed reduction
    soc_floor: float = 0.30        # Minimum speed fraction when SOC → 0 (30% of full speed)

    # Conventions / safety
    forward_sign: int = +1         # +1 for forward=positive, -1 if robot is mounted backward
    theta_hard_limit_deg: float = 80.0  # Discard detection if |theta| > this (likely corrupt)
    invalid_brake: float = 0.0     # Drive output when NO-GO (0 = stop, could be small reverse)
    require_start_go: bool = True  # If True, output zero until start_go=True

    # ====== Lane-lost handling ======
    hold_decay_per_frame: float = 0.90   # Each frame in hold mode: output *= 0.90 (exponential decay)
    search_pivot: bool = True            # True: pivot in place (drive=0) during search
    search_speed: float = 0.00           # Drive speed during search (0 when pivoting)
    search_steer_const: float = 0.6      # Constant right steer during search [rad] (~34 deg)
    loop_period_s: float = 0.050         # Control loop period (50ms = 20Hz, used for logs only)

    # ====== Pulse control for curves ======
    pulse_enabled: bool = True           # Enable pulse throttle in curves
    pulse_theta_threshold: float = 25.0  # Activate pulse when |theta| > this [deg]
    pulse_lateral_threshold: float = 30.0  # Or when |lateral| > this [px]
    pulse_on_frames: int = 3             # Frames per ON phase (throttle applied)
    pulse_off_frames: int = 3            # Frames per OFF phase (coast, zero drive)
    pulse_drive_on: float = 0.35         # Drive torque during ON phase
    pulse_drive_off: float = 0.0         # Drive torque during OFF phase (coast)


class DriverModel:
    """Given lane_mode, produce (drive_torque, steer_angle) for steer-type control."""

    def __init__(self, cfg: DriverConfig):
        self.cfg = cfg
        self._prev_drive = 0.0
        self._prev_steer = 0.0
        # Compute half-width for lateral normalization
        self._half_w = max(
            1.0,
            (cfg.lateral_norm_halfwidth_px
             if cfg.lateral_norm_halfwidth_px is not None
             else cfg.image_width * 0.5)
        )
        # For hold mode, remember most recent base/steer
        self._last_base = 0.0
        self._last_steer = 0.0
        # Pulse control state
        self._pulse_counter = 0  # Frame counter within current pulse phase
        self._pulse_phase = True  # True = ON phase, False = OFF phase
        self.last_debug: Dict[str, float | int | str | bool | None] = {}

    def update(
        self,
        lateral_px: Optional[float],
        theta_deg: Optional[float],
        soc: Optional[float],
        image_width: Optional[int],
        start_go: bool,
        valid_lane: bool,
        lane_mode: str = "normal",   # "normal" / "hold" / "search"
        lost_age: int = 0,           # consecutive frames since last valid detection
        single_side: bool = False,   # True if only one lane detected (estimated other)
    ) -> Tuple[float, float]:
        """
        Compute (drive_torque, steer_angle) for the current frame.

        Called once per frame (~20Hz) by rule_based_input.update().
        Returns smoothed output after applying limits.
        """
        # Update image half-width for lateral normalization
        if image_width is not None and image_width > 0:
            self._half_w = max(1.0, image_width * 0.5)

        # === Start gate: hold zero until GO signal ===
        if self.cfg.require_start_go and not start_go:
            drive = steer = self.cfg.invalid_brake * self.cfg.forward_sign
            drive, steer = self._smooth(drive, steer)
            self._store_debug(False, False, lane_mode, lateral_px, theta_deg,
                              0.0, 0.0, 0.0, drive, steer, soc, None, None, lost_age)
            return drive, steer

        # === Theta sanity check: discard obviously corrupt detections ===
        if theta_deg is not None and abs(float(theta_deg)) > self.cfg.theta_hard_limit_deg:
            valid_lane = False

        # === SOC scaling: reduce speed proportionally when battery is low ===
        scale = 1.0
        if self.cfg.use_soc_scaling and (soc is not None):
            s = _clip(float(soc), 0.0, 1.0)
            # scale maps soc: 0 → soc_floor, 1 → 1.0
            scale = _clip(self.cfg.soc_floor + (1.0 - self.cfg.soc_floor) * s,
                          self.cfg.soc_floor, 1.0)

        mode = (lane_mode or "normal").lower()

        # ======== NORMAL MODE ========
        # Full lane detection available: compute steer from theta and lateral
        if mode == "normal":
            if not (valid_lane and (lateral_px is not None) and (theta_deg is not None)):
                # Missing required inputs → degrade to hold mode
                mode = "hold"
            else:
                # Normalize lateral offset to [-1, +1] range
                lateral_n = float(lateral_px) / self._half_w
                theta_rad = math.radians(float(theta_deg))

                # P-controller for steering: combines heading error and lateral offset
                steer = self.cfg.k_theta * theta_rad + self.cfg.k_lateral * lateral_n
                steer = _clip(steer, -self.cfg.steer_limit, self.cfg.steer_limit)

                # Speed planning: reduce speed proportionally to steer angle magnitude
                # steer_cost in [0,1]: 0 at straight, 1 at full steer
                steer_cost = abs(steer) / self.cfg.steer_limit
                base = self.cfg.v_max * max(0.0, 1.0 - steer_cost * 0.7)
                base = max(self.cfg.v_min, min(self.cfg.v_max, base))

                # Ensure minimum speed during cornering (prevent stall)
                corner_base_floor = 0.28
                theta_thr_deg = 15.0
                if abs(theta_deg) >= theta_thr_deg:
                    base = max(base, corner_base_floor)

                self._last_base = base
                self._last_steer = steer

                # ====== Pulse control for sharp curves ======
                # Disabled when single_side (less accurate), or when below thresholds
                use_pulse = False
                if self.cfg.pulse_enabled and not single_side:
                    if (abs(theta_deg) > self.cfg.pulse_theta_threshold or
                        abs(lateral_px) > self.cfg.pulse_lateral_threshold):
                        use_pulse = True

                if use_pulse:
                    # Alternate between ON and OFF phases at fixed frame counts
                    if self._pulse_phase:
                        # ON phase: apply reduced throttle
                        drive = self.cfg.pulse_drive_on * scale * self.cfg.forward_sign
                        self._pulse_counter += 1
                        if self._pulse_counter >= self.cfg.pulse_on_frames:
                            self._pulse_counter = 0
                            self._pulse_phase = False  # Switch to OFF
                    else:
                        # OFF phase: coast (zero drive)
                        drive = self.cfg.pulse_drive_off * self.cfg.forward_sign
                        self._pulse_counter += 1
                        if self._pulse_counter >= self.cfg.pulse_off_frames:
                            self._pulse_counter = 0
                            self._pulse_phase = True   # Switch to ON
                else:
                    # Normal continuous control
                    drive = base * scale * self.cfg.forward_sign
                    # Reset pulse state when leaving curve condition
                    self._pulse_counter = 0
                    self._pulse_phase = True

                drive, steer = self._post_process(drive, steer)
                self._store_debug(True, True, "normal", lateral_px, theta_deg,
                                  lateral_n, theta_rad, steer, drive, steer, soc, base, scale, lost_age,
                                  use_pulse=use_pulse, pulse_phase=self._pulse_phase)
                return drive, steer

        # ======== HOLD MODE ========
        # Lane temporarily lost: decay last known output to gradually slow down
        if mode == "hold":
            # Exponential decay: output × (decay_rate ^ lost_age)
            decay = self.cfg.hold_decay_per_frame ** max(0, int(lost_age))
            base = self._last_base * decay
            steer = self._last_steer * decay
            drive = base * scale * self.cfg.forward_sign
            drive, steer = self._post_process(drive, steer)
            self._store_debug(False, True, "hold", lateral_px, theta_deg,
                              0.0, 0.0, steer, drive, steer, soc, base, scale, lost_age)
            return drive, steer

        # ======== SEARCH MODE ========
        # Lane lost for extended time: pivot in place to scan for lane
        base = _clip(self.cfg.search_speed, self.cfg.v_min, self.cfg.v_max) if not self.cfg.search_pivot else 0.0
        steer = abs(self.cfg.search_steer_const)  # constant right turn (positive steer)
        drive = base * scale * self.cfg.forward_sign
        drive, steer = self._post_process(drive, steer)
        self._store_debug(False, True, "search", lateral_px, theta_deg,
                          0.0, 0.0, steer, drive, steer, soc, base, scale, lost_age)
        return drive, steer

    # ------------ helpers ------------
    def _post_process(self, drive: float, steer: float) -> Tuple[float, float]:
        """Apply hard limits then IIR smoothing."""
        drive = _clip(drive, -self.cfg.torque_limit, self.cfg.torque_limit)
        steer = _clip(steer, -self.cfg.steer_limit, self.cfg.steer_limit)
        return self._smooth(drive, steer)

    def _smooth(self, drive: float, steer: float) -> Tuple[float, float]:
        """
        IIR (exponential moving average) smoothing on output commands.
        Reduces high-frequency oscillation.
        output = (1 - alpha) * new_value + alpha * previous_value
        alpha=0.50 → 50% weight to previous output (moderate smoothing)
        alpha=0.00 → no smoothing (raw output)
        """
        a = _clip(self.cfg.alpha_smooth, 0.0, 1.0)
        drive_s = (1 - a) * drive + a * self._prev_drive
        steer_s = (1 - a) * steer + a * self._prev_steer
        self._prev_drive, self._prev_steer = drive_s, steer_s
        return drive_s, steer_s

    def _store_debug(
        self,
        valid_lane: bool,
        start_go: bool,
        lane_mode: str,
        lateral_px: Optional[float],
        theta_deg: Optional[float],
        lateral_n: float,
        theta_rad: float,
        steer_cmd: float,
        drive: float,
        steer: float,
        soc: Optional[float],
        base: Optional[float],
        scale: Optional[float],
        lost_age: int,
        use_pulse: bool = False,
        pulse_phase: bool = True,
    ) -> None:
        """
        Save per-frame debug values accessible as self.last_debug.
        Used by rule_based_input.py to annotate debug overlay images.
        """
        self.last_debug = {
            "start_go": bool(start_go),
            "valid_lane": bool(valid_lane),
            "lane_mode": str(lane_mode),
            "lost_age": int(lost_age),
            "lateral_px": None if lateral_px is None else float(lateral_px),
            "theta_deg": None if theta_deg is None else float(theta_deg),
            "lateral_norm": float(lateral_n),   # Normalized lateral in [-1, +1]
            "theta_rad": float(theta_rad),
            "steer_cmd": float(steer_cmd),       # Pre-smoothing steer command
            "base_speed": None if base is None else float(base),
            "soc": None if soc is None else float(soc),
            "soc_scale": None if scale is None else float(scale),
            "drive_torque": float(drive),         # Final (post-smoothing) drive output
            "steer_angle": float(steer),          # Final (post-smoothing) steer output
            "half_width_px": float(self._half_w),
            "forward_sign": int(self.cfg.forward_sign),
            "loop_period_s": float(self.cfg.loop_period_s),
            # Pulse control info (for HUD display)
            "use_pulse": bool(use_pulse),
            "pulse_phase": "ON" if pulse_phase else "OFF",
        }


# ------------- optional CLI test -------------
if __name__ == "__main__":
    import argparse
    ap = argparse.ArgumentParser()
    ap.add_argument("--image_width", type=int, default=224)
    ap.add_argument("--lateral_px", type=float, default=3.0)
    ap.add_argument("--theta_deg", type=float, default=2.0)
    ap.add_argument("--soc", type=float, default=1.0)
    ap.add_argument("--start_go", action="store_true")
    ap.add_argument("--valid_lane", action="store_true")
    ap.add_argument("--lane_mode", type=str, default="normal")
    ap.add_argument("--lost_age", type=int, default=0)
    ap.add_argument("--forward_sign", type=int, default=+1, choices=[-1, +1])
    args = ap.parse_args()

    cfg = DriverConfig(image_width=args.image_width, forward_sign=args.forward_sign)
    drv = DriverModel(cfg)
    drive, steer = drv.update(
        lateral_px=args.lateral_px,
        theta_deg=args.theta_deg,
        soc=args.soc,
        image_width=args.image_width,
        start_go=args.start_go,
        valid_lane=args.valid_lane,
        lane_mode=args.lane_mode,
        lost_age=args.lost_age,
    )
    print(f"drive={drive:+.3f}, steer={steer:+.3f} rad ({math.degrees(steer):+.1f} deg)")
    print("debug:", drv.last_debug)
