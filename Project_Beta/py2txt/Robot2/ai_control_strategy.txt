def should_wait_for_start(pil_img, race_started):
    """
    Determine if the car should wait (output zero torque).

    Args:
        pil_img     : Current camera image (PIL Image)
        race_started: Whether the race has already started (latched True after first GO)

    Returns:
        True if should wait (drive=0), False if should drive

    NOTE (Robot2 vs Robot1):
        Robot1 adds a START_DETECTION_TIMEOUT_FRAMES fallback:
        if start detection takes too long, Robot1 starts anyway.
        Robot2 does NOT have this fallback — it waits indefinitely
        for the start signal (simpler but potentially safer).
    """
    if STRATEGY == "pure_e2e":
        # Trust the model completely — never block on start signal
        return False

    # Hybrid mode: block until start signal is detected
    if not race_started and HYBRID_START_DETECTION:
        if detect_start_signal(pil_img):
            return False  # GO!
        else:
            return True   # Keep waiting

    return False
