# inference_input.py (Robot2 version)
# ==============================================================================
# AI Inference Engine - The Driver's Brain
# ==============================================================================
#
# This is the core inference engine that runs the AI model for Robot2.
# It reads camera images, runs neural network inference, and outputs controls.
#
# IMPORTANT: This file is the FRAMEWORK - don't modify it directly.
# To customize AI behavior, edit: ai_control_strategy.py
#
# DIFFERENCE FROM ROBOT1's inference_input.py:
#   - No warmup_cuda() function
#     Robot1 has warmup_cuda() to pre-initialize GPU kernels and eliminate
#     the 10+ second delay on first inference. Robot2 does not include this.
#   - adjust_output() call: Robot2's strategy function takes 4 args
#     (drive, steer, pil_img, soc) — no race_started parameter.
#   - MODULE_SOURCE = "Robot2", robot_id = "R2"
#   - No ValueNetwork import (Robot2 doesn't do RL training)
#
# MODULE LOADING (importlib pattern):
#   Both model.py and ai_control_strategy.py are loaded via importlib.util
#   rather than standard Python imports. This is critical when running Robot1
#   and Robot2 simultaneously: Python's module cache would otherwise mix up
#   the two robots' modules if they have the same filename.
#   importlib with a unique module name ("Robot2.model") keeps them separate.
#
# CONTROL FLOW:
#   1. _load_model()        : Load model.pth from Robot2/models/ (called once)
#   2. get SOC, RGB path    : Via data_manager (shared across robots)
#   3. should_wait_for_start(): Ask strategy whether to hold zero output
#   4. AI inference          : Run DrivingNetwork, saturate output
#   5. adjust_output()       : Apply strategy post-processing
#   6. get_latest_command()  : Return JSON-compatible control dict for WebSocket
# ==============================================================================

import os
import sys
from pathlib import Path

import torch
from torchvision import transforms
from PIL import Image

# Module identification
MODULE_SOURCE = "Robot2"
print(f"[inference_input] Loaded from {MODULE_SOURCE}/")

# Import data_manager from parent (Project_Beta/) directory
sys.path.insert(0, str(Path(__file__).parent.parent))
import data_manager

_this_dir = Path(__file__).parent

# Load model.py using importlib (prevents cache collision with Robot1's model.py)
import importlib.util
_model_spec = importlib.util.spec_from_file_location(
    f"{MODULE_SOURCE}.model",          # Unique name: "Robot2.model"
    _this_dir / "model.py"
)
_model_module = importlib.util.module_from_spec(_model_spec)
_model_spec.loader.exec_module(_model_module)
DrivingNetwork = _model_module.DrivingNetwork

# Load ai_control_strategy.py using importlib (prevents cache collision)
try:
    _strategy_spec = importlib.util.spec_from_file_location(
        f"{MODULE_SOURCE}.ai_control_strategy",   # Unique name: "Robot2.ai_control_strategy"
        _this_dir / "ai_control_strategy.py"
    )
    _strategy_module = importlib.util.module_from_spec(_strategy_spec)
    _strategy_spec.loader.exec_module(_strategy_module)
    should_wait_for_start = _strategy_module.should_wait_for_start
    adjust_output = _strategy_module.adjust_output
    on_race_start = _strategy_module.on_race_start
    STRATEGY = _strategy_module.STRATEGY
    print(f"[inference_input] Strategy loaded: {STRATEGY}")
except Exception as e:
    print(f"[inference_input] WARNING: ai_control_strategy.py not found, using defaults: {e}")
    # Fallback strategy: hybrid mode with inline start detection
    STRATEGY = "hybrid"
    def should_wait_for_start(pil_img, race_started):
        if not race_started:
            _percept_spec = importlib.util.spec_from_file_location(
                f"{MODULE_SOURCE}.perception_Startsignal",
                _this_dir / "rule_based_algorithms" / "perception_Startsignal.py"
            )
            _percept_module = importlib.util.module_from_spec(_percept_spec)
            _percept_spec.loader.exec_module(_percept_module)
            return not _percept_module.detect_start_signal(pil_img)
        return False
    def adjust_output(drive, steer, pil_img, soc):
        return drive, steer
    def on_race_start():
        pass

# Control output (polled by main.py via get_latest_command())
robot_id = "R2"
driveTorque = 0.0
steerAngle = 0.0

# Model state
_model = None
_transform = None
_model_loaded = False
_device = None

# Race state (latched True after first GO signal)
_race_started = False


def get_latest_command():
    """Return the latest control command in the standard dict format."""
    return {
        "type": "control",
        "robot_id": robot_id,
        "driveTorque": round(float(driveTorque), 3),
        "steerAngle": round(float(steerAngle), 3),
    }


def preload_model():
    """
    Preload the AI model before the control loop starts.
    Call this BEFORE the race starts to avoid model loading delays
    during the start signal sequence.

    NOTE: Robot2 does not have warmup_cuda() (unlike Robot1).
    First inference after preload may still have a brief delay.
    """
    print(f"[{robot_id} Inference] Preloading AI model...")
    _load_model()
    if _model is not None:
        print(f"[{robot_id} Inference] Model preloaded successfully!")
    else:
        print(f"[{robot_id} Inference] WARNING: Model preload failed (will use dummy output)")


def saturate(value, min_val=-1.0, max_val=1.0):
    """Clamp value within [min_val, max_val]."""
    return max(min_val, min(max_val, value))


def _load_model():
    """
    Load the trained model from Robot2/models/model.pth.
    Called automatically on first update() (or manually by preload_model()).
    Idempotent: subsequent calls do nothing.

    Model path: Robot2/models/model.pth
    If not found, driveTorque=0.0 and steerAngle=0.0 (safe default).

    Image transform (must match training normalization):
      Resize to 224×224, ToTensor, Normalize with ImageNet stats
    """
    global _model, _transform, _model_loaded, _device

    if _model_loaded:
        return

    _model_loaded = True

    _device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    print(f"[{robot_id} Inference] Using device: {_device}")

    model_path = Path(__file__).parent / "models" / "model.pth"

    model = DrivingNetwork()

    try:
        model.load_state_dict(torch.load(str(model_path), map_location=_device))
        model.to(_device)
        model.eval()
        print(f"[{robot_id} Inference] Model loaded from {model_path}")
        _model = model
    except FileNotFoundError:
        print(f"[{robot_id} Inference] WARNING: Model not found at {model_path}")
        print(f"[{robot_id} Inference] Using dummy predictions (drive=0.0, steer=0.0)")
        _model = None
    except Exception as e:
        print(f"[{robot_id} Inference] WARNING: Failed to load model: {e}")
        _model = None

    # Image preprocessing pipeline (must match training transforms)
    _transform = transforms.Compose([
        transforms.Resize((224, 224)),
        transforms.ToTensor(),
        transforms.Normalize(
            mean=[0.485, 0.456, 0.406],   # ImageNet channel means
            std=[0.229, 0.224, 0.225]     # ImageNet channel stds
        )
    ])


def update():
    """
    Single update cycle for AI inference control (~20Hz).
    Called repeatedly by main.py's asyncio control loop.

    Returns:
        True  : Continue (normal operation or recoverable error)
        False : Stop (should not occur in normal use)

    Control flow:
      1. Load model (first call only)
      2. Get SOC and latest camera image path from data_manager
      3. Check should_wait_for_start() from ai_control_strategy
      4. Run inference if model loaded, else output zero
      5. Apply adjust_output() from ai_control_strategy
      6. Log every 20 frames (~1 second)
    """
    global driveTorque, steerAngle, _race_started

    try:
        _load_model()

        # Get inputs from data_manager (shared image buffer system)
        soc = data_manager.get_latest_soc(robot_id)
        rgb_path = data_manager.get_latest_rgb_path(robot_id)

        if not rgb_path or not rgb_path.exists():
            return True  # No image yet, skip this frame

        try:
            pil_img = Image.open(rgb_path).convert("RGB")
        except Exception:
            return True  # Failed to open image, skip this frame

        if soc is None:
            soc = 1.0  # Default SOC if unavailable

        # === Strategy: Check if we should wait for start signal ===
        if should_wait_for_start(pil_img, _race_started):
            driveTorque = 0.0
            steerAngle = 0.0

            if not hasattr(update, '_frame_count'):
                update._frame_count = 0
            update._frame_count += 1
            if update._frame_count % 40 == 0:  # Log every ~2 seconds
                print(f"[{robot_id} Inference] Waiting for start signal... (Strategy: {STRATEGY})")
            return True

        # === Race Start Detection (latch once) ===
        if not _race_started:
            _race_started = True
            print(f"[{robot_id} Inference] RACE STARTED! (Strategy: {STRATEGY})")
            on_race_start()

        # === AI Inference ===
        if _model is not None:
            # Prepare input tensors
            image_tensor = _transform(pil_img).unsqueeze(0).to(_device)  # [1, 3, 224, 224]
            soc_tensor = torch.tensor([[soc]], dtype=torch.float32).to(_device)  # [1, 1]

            # Run inference (no_grad for efficiency)
            with torch.no_grad():
                output = _model(image_tensor, soc_tensor)
                raw_drive = output[0, 0].item()
                raw_steer = output[0, 1].item()

            # Clamp to safe ranges
            raw_drive = saturate(raw_drive, -1.0, 1.0)
            raw_steer = saturate(raw_steer, -0.524, 0.524)  # ±30 degrees max

            # Strategy post-processing (Robot2: 4-arg version, no race_started)
            driveTorque, steerAngle = adjust_output(raw_drive, raw_steer, pil_img, soc)
        else:
            # No model loaded: output zero (safe default)
            driveTorque = 0.0
            steerAngle = 0.0

        # === Logging every 20 frames (~1 second at 20Hz) ===
        if not hasattr(update, '_frame_count'):
            update._frame_count = 0
        update._frame_count += 1

        if update._frame_count % 20 == 0:
            import math
            steer_deg = math.degrees(steerAngle)
            soc_str = f"{float(soc):.2f}"
            print(
                f"[{robot_id} Inference] Drive={driveTorque:+.3f}, "
                f"Steer={steerAngle:+.3f}rad({steer_deg:+.1f}deg), "
                f"SOC={soc_str}"
            )

        return True

    except Exception as e:
        print(f"[{robot_id} Inference] Error: {e}")
        import traceback
        traceback.print_exc()
        return True  # Continue despite error


def reset():
    """Reset AI inference state (model is reloaded next update)."""
    global driveTorque, steerAngle, _model, _transform, _model_loaded, _device, _race_started
    driveTorque = 0.0
    steerAngle = 0.0
    _model = None
    _transform = None
    _model_loaded = False
    _device = None
    _race_started = False
    if hasattr(update, '_frame_count'):
        update._frame_count = 0
    # Reset start signal detector state
    try:
        from rule_based_algorithms.perception_Startsignal import detect_start_signal
        if hasattr(detect_start_signal, 'ready_to_go'):
            detect_start_signal.ready_to_go = False
    except ImportError:
        pass
    print(f"[{robot_id} Inference] State reset")
