# keyboard_input.py (Robot2 version)
# Allows manual control of drive torque and steering angle using keyboard keys
#
# PURPOSE:
#   Reads keyboard input in a background thread and maintains current
#   drive torque and steer angle values that main.py polls at ~20Hz.
#
# CONTROL SCHEME (steer-type, same as Robot1):
#   W / Z  : Increase / Decrease drive torque (hold for continuous)
#   J / L  : Steer left / right (resets to 0 on release)
#   I / M  : (reserved for steer reset - currently maps to steer=0 on release)
#
# DIFFERENCE FROM ROBOT1:
#   Identical except robot_id = "R2" and MODULE_SOURCE = "Robot2".
#   Both robots share the same KEY_MAP (W/Z/J/L/I/M).
#   Only one robot can use keyboard mode at a time (main.py enforces this).
#
# KEY BEHAVIOR:
#   - Drive torque: accumulates ±TORQUE_STEP per loop iteration while key held
#     Releases to 0.0 immediately when key released
#   - Steer angle: accumulates ±STEER_STEP per loop iteration while key held
#     Releases to 0.0 immediately when key released
#   Loop runs at ~20Hz (sleep 50ms)
#
# NOTE: The 'keyboard' library requires admin/root on some systems.
#   If the keyboard library fails, main.py's keyboard monitor falls back to msvcrt.

import threading
import keyboard
import time
import sys

# Module identification (printed on load for diagnostics)
MODULE_SOURCE = "Robot2"
print(f"[keyboard_input] Loaded from {MODULE_SOURCE}/")

# Windows only: flush any pending key events to avoid stale input on startup
def clear_input_buffer():
    if sys.platform == "win32":
        import msvcrt
        while msvcrt.kbhit():
            msvcrt.getch()

# === Control parameters ===
TORQUE_STEP = 0.25     # Drive torque increment per loop step (20Hz → 5 steps to max)
STEER_STEP  = 0.20     # Steer angle increment per loop step [rad] (about 11.5°/step)
MAX_TORQUE  = 1.0      # Maximum drive torque magnitude
MAX_STEER   = 0.524    # Maximum steer angle [rad] (~30 degrees, matches Unity maxSteerAngleDeg)

# === Key mapping ===
KEY_MAP = {"w","z","j","l","i","m"}

# === Current control values (read by main.py via get_latest_command()) ===
driveTorque: float = 0.0
steerAngle:  float = 0.0
robot_id:    str   = "R2"   # Robot identifier for Unity WebSocket messages

# Internal state
_key_states = {k: False for k in KEY_MAP}
_listener_thread: threading.Thread | None = None
_stop_event: threading.Event | None = None

def _update_key_state(event):
    """Callback for keyboard library: updates _key_states on press/release."""
    name = event.name
    if name in _key_states:
        _key_states[name] = (event.event_type == "down")

def _loop(stop_event: threading.Event):
    """
    Background thread main loop.
    Polls key states at ~20Hz and updates driveTorque/steerAngle.
    Drive torque accumulates while key held; releases to 0 on release.
    Steer angle accumulates while key held; releases to 0 on release.
    """
    global driveTorque, steerAngle
    keyboard.hook(_update_key_state)

    try:
        while not stop_event.is_set():
            # --- Drive torque (W=forward, Z=reverse, release=stop) ---
            if _key_states["w"]:
                driveTorque += TORQUE_STEP
            elif _key_states["z"]:
                driveTorque -= TORQUE_STEP
            else:
                driveTorque = 0.0  # Return to zero on release

            # --- Steering angle (J=left, L=right, release=center) ---
            if _key_states["j"]:
                steerAngle -= STEER_STEP   # Negative = left turn
            elif _key_states["l"]:
                steerAngle += STEER_STEP   # Positive = right turn
            else:
                steerAngle = 0.0  # Return to center on release

            # Apply limits
            if driveTorque >  MAX_TORQUE: driveTorque =  MAX_TORQUE
            if driveTorque < -MAX_TORQUE: driveTorque = -MAX_TORQUE
            if steerAngle  >  MAX_STEER:  steerAngle  =  MAX_STEER
            if steerAngle  < -MAX_STEER:  steerAngle  = -MAX_STEER

            time.sleep(0.05)  # 20Hz polling rate
    finally:
        clear_input_buffer()
        print("[Keyboard] Listener stopped.")

# === Public API (called by main.py) ===
def start_listener():
    """Start background keyboard listener thread (idempotent - safe to call multiple times)."""
    global _listener_thread, _stop_event
    if _listener_thread and _listener_thread.is_alive():
        return
    _stop_event = threading.Event()
    _listener_thread = threading.Thread(target=_loop, args=(_stop_event,), daemon=True)
    _listener_thread.start()
    print("[Keyboard] Listener started.")

def stop_listener():
    """Stop background keyboard listener thread."""
    global _listener_thread, _stop_event
    if _stop_event:
        _stop_event.set()
    _stop_event = None
    _listener_thread = None

def get_latest_command():
    """
    Return latest control command dict for WebSocket sender.
    Called by main.py at ~20Hz in the control loop.
    """
    return {
        "type": "control",
        "robot_id": robot_id,
        "driveTorque": round(float(driveTorque), 3),
        "steerAngle": round(float(steerAngle), 3),  # radians
    }

def debug_print_state():
    """Print current drive and steer values (for manual debugging)."""
    print(f"[DEBUG] drive={driveTorque:.2f}, steer={steerAngle:.2f}")
