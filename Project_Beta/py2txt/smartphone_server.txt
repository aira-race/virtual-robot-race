"""
smartphone_server.py
============================================================
FILE: smartphone_server.py
ROLE: HTTP + WebSocket server that lets a smartphone browser act as a
      game controller for Virtual Robot Race Beta 1.6.

OVERVIEW:
  SmartphoneServer provides two services over a single aiohttp application:
    1) HTTP — serves the controller web page (smartphone_controller.html)
       and generates QR code images for easy URL sharing.
    2) WebSocket — one endpoint per robot (/ws/R1 … /ws/R5) that receives
       joystick control input from the phone and forwards it to Unity via
       the existing RobotWebSocketClient.

ARCHITECTURE:
  SmartphoneServer
    └─ RobotController (one per registered robot)
         ├─ handle_smartphone_connection() — aiohttp WebSocket session
         ├─ _forward_control()             — phone → Unity relay
         └─ stream_task                    — background data push to phone
              • STREAM_MODE="soc"    → _soc_stream_loop()    (default)
              • STREAM_MODE="camera" → _camera_stream_loop() (optional)

CONNECTION CONFIRMATION (L+R button press):
  Before control input is forwarded to Unity, the smartphone must send a
  'connect_confirm' message triggered by pressing BOTH L and R buttons
  simultaneously. This prevents accidental control during connection setup.
  main.py waits for all robots to confirm before sending Unity ready signals.

DOUBLE-BUFFER IMAGE SYSTEM (for camera streaming):
  Camera frames are saved by the Python WebSocket client as:
    Robot{N}/data_interactive/latest_RGB_a.jpg
    Robot{N}/data_interactive/latest_RGB_b.jpg
  A pointer file (latest_RGB_now.txt) contains "a" or "b" indicating which
  buffer holds the most recent complete frame. This prevents reading a
  partially written JPEG.

QR CODE DISPLAY:
  On start(), QR code PNGs are generated for each robot and displayed in
  per-robot tkinter popup windows. Each window closes automatically when
  the smartphone for that robot connects. tkinter runs in a daemon thread
  because it requires its own thread and must not block the asyncio loop.

STREAM_MODE VALUES:
  "soc"    — Sends battery SOC (0.0–1.0) from latest_SOC.txt at 2 Hz.
             Low overhead; default mode.
  "camera" — Streams JPEG frames from the double-buffer system at CAMERA_FPS.
             Higher bandwidth; optional.
"""

# Streaming settings
STREAM_MODE = "soc"          # "camera" for video, "soc" for SOC data only
SOC_UPDATE_INTERVAL = 0.5    # SOC update interval in seconds (2 updates/second)

# Camera streaming settings (only used if STREAM_MODE = "camera")
CAMERA_FPS = 15              # Target FPS for camera streaming (5-20 recommended)
CAMERA_OPTIMIZE = False      # True: re-compress JPEG (adds CPU overhead)
CAMERA_QUALITY = 50          # JPEG quality if CAMERA_OPTIMIZE=True (1-100)
CAMERA_MAX_WIDTH = None      # Resize width if CAMERA_OPTIMIZE=True (None = no resize)

# QR code fill colors per robot (used in both popup windows and PNG files)
QR_COLORS = {
    'R1': '#CC0000',  # Red for Robot1
    'R2': '#00AA00',  # Green for Robot2
    'R3': '#0000CC',  # Blue for Robot3
    'R4': '#CC6600',  # Orange for Robot4
    'R5': '#6600CC',  # Purple for Robot5
}

logger = logging.getLogger(__name__)


class RobotController:
    """
    Manages the smartphone WebSocket session for a single robot.

    RESPONSIBILITIES:
      - Accept / replace incoming smartphone connections.
      - Gate control forwarding behind L+R dual-button confirmation.
      - Stream SOC or camera data to the smartphone asynchronously.
      - Signal readiness to SmartphoneServer.wait_for_all_ready().

    STATE MACHINE (is_ready):
      UNREADY (is_ready=False)
        → smartphone connects
        → send 'reset' message (phone shows L+R confirmation screen)
        → phone sends 'connect_confirm'
      READY (is_ready=True)
        → control input is forwarded to Unity
        → ready_event is set (main() can proceed)
      On reconnection: state resets to UNREADY.
    """

    def __init__(self, robot_id: str, robot_websocket_client=None, server=None):
        self.robot_id = robot_id          # e.g., "R1"
        self.robot_num = int(robot_id[1]) # Extract integer from "R1" → 1
        self.ws_client = robot_websocket_client  # Unity WebSocket client reference
        self.smartphone_ws = None         # Active aiohttp WebSocket to the smartphone
        self.stream_task = None           # Background task for camera or SOC streaming
        self.is_streaming = True          # Set False on cleanup to stop streaming loops
        self.server = server              # Parent SmartphoneServer (for QR popup control)

        # Readiness flag and event for dual-button confirmation
        self.is_ready = False             # True once L+R confirmed
        self.ready_event = asyncio.Event()# Set when is_ready becomes True

        # Paths for the double-buffer image system
        self.data_dir = Path(f"Robot{self.robot_num}/data_interactive")
        self.buffer_pointer_file = self.data_dir / "latest_RGB_now.txt"

        logger.info(f"[{self.robot_id}] Controller initialized")

    def set_websocket_client(self, ws_client):
        """Bind (or rebind) the Unity WebSocket client after creation."""
        self.ws_client = ws_client
        logger.info(f"[{self.robot_id}] WebSocket client connected")

    async def handle_smartphone_connection(self, websocket):
        """
        Manage the full lifetime of one smartphone WebSocket session.

        Called by SmartphoneServer.handle_websocket() when a phone connects.
        Flow:
          1) Close any existing smartphone connection (phone reconnect case).
          2) Store new websocket reference.
          3) Reset readiness state (require L+R confirmation again).
          4) Close this robot's QR code popup window.
          5) Send 'reset' message → phone shows confirmation screen.
          6) Start streaming task (camera or SOC).
          7) Process incoming messages until disconnect or error.
          8) Cleanup (cancel stream task, clear ws reference).
        """
        # Close existing connection if a new phone scans the same QR code
        if self.smartphone_ws and not self.smartphone_ws.closed:
            logger.info(f"[{self.robot_id}] Closing previous smartphone connection")
            try:
                await self.smartphone_ws.close()
            except Exception as e:
                logger.warning(f"[{self.robot_id}] Error closing old connection: {e}")

        self.smartphone_ws = websocket
        logger.info(f"[{self.robot_id}] Smartphone connected")

        # Reset readiness for the new connection
        self.is_ready = False
        self.ready_event.clear()

        # Close QR popup for this robot now that a phone has connected
        if self.server:
            self.server.close_qr_popup(self.robot_id)

        # Tell the phone to display the L+R confirmation screen
        try:
            await websocket.send_str(json.dumps({
                'type': 'reset',
                'message': 'Please press L+R buttons to confirm connection.'
            }))
            logger.info(f"[{self.robot_id}] Reset signal sent to smartphone (reconnection)")
        except Exception as e:
            logger.warning(f"[{self.robot_id}] Failed to send reset on connect: {e}")

        # Launch the background streaming task
        self.is_streaming = True
        if STREAM_MODE == "camera":
            self.stream_task = asyncio.create_task(self._camera_stream_loop())
        else:
            self.stream_task = asyncio.create_task(self._soc_stream_loop())

        try:
            # Process messages until the phone disconnects
            async for msg in websocket:
                if msg.type == WSMsgType.TEXT:
                    await self._handle_message(msg.data)
                elif msg.type == WSMsgType.ERROR:
                    logger.error(f"[{self.robot_id}] WebSocket error: {websocket.exception()}")

        except Exception as e:
            logger.error(f"[{self.robot_id}] Connection error: {e}")

        finally:
            await self._cleanup()

    async def _handle_message(self, message_text: str):
        """
        Dispatch incoming JSON messages from the smartphone.

        MESSAGE TYPES:
          'control'        — joystick input; forwarded to Unity if READY.
          'ping'           — connection keepalive; responds with 'pong'.
          'connect_confirm'— dual L+R button press; sets is_ready=True and
                             fires ready_event for wait_for_all_ready().
          'camera_control' — toggle camera streaming on/off.
        """
        try:
            data = json.loads(message_text)
            msg_type = data.get('type')

            if msg_type == 'control':
                # Forward joystick input to Unity (blocked until is_ready)
                await self._forward_control(data)

            elif msg_type == 'ping':
                # Keepalive: smartphone sends pings to detect disconnection
                await self._send_to_smartphone({'type': 'pong'})

            elif msg_type == 'connect_confirm':
                # L+R dual-button press confirmation
                if not self.is_ready:
                    self.is_ready = True
                    self.ready_event.set()
                    logger.info(f"[{self.robot_id}] ✓ Connection confirmed (dual button press)")

                    # Acknowledge to phone so it switches to driving UI
                    await self._send_to_smartphone({
                        'type': 'ready_confirmed',
                        'message': 'Connection confirmed! Race starting soon.'
                    })

            elif msg_type == 'camera_control':
                # Allow phone to pause/resume camera stream (saves bandwidth)
                self.is_streaming = data.get('enabled', True)
                logger.info(f"[{self.robot_id}] Camera streaming: {self.is_streaming}")

        except json.JSONDecodeError as e:
            logger.error(f"[{self.robot_id}] Invalid JSON: {e}")
        except Exception as e:
            logger.error(f"[{self.robot_id}] Message handling error: {e}")

    async def _forward_control(self, control_data: dict):
        """
        Relay joystick control from smartphone to Unity.

        SAFETY GATE:
          Control is silently ignored until is_ready=True (L+R confirmed).
          This prevents accidental robot movement during the connection phase.

        MESSAGE FORMAT sent to Unity:
          {
            "type": "control",
            "robot_id": "R1",
            "steerAngle": float (-1.0 to 1.0),
            "driveTorque": float (-1.0 to 1.0)
          }
        """
        if not self.ws_client:
            logger.warning(f"[{self.robot_id}] No Unity WebSocket client available")
            return

        # Block control before L+R confirmation to prevent unintended movement
        if not self.is_ready:
            return  # Silently ignore

        steer_angle = control_data.get('steerAngle', 0.0)
        drive_torque = control_data.get('driveTorque', 0.0)

        control_msg = {
            "type": "control",
            "robot_id": self.robot_id,
            "steerAngle": steer_angle,
            "driveTorque": drive_torque
        }

        try:
            await self.ws_client.send_json(control_msg)
            logger.debug(f"[{self.robot_id}] Control: steer={steer_angle:.2f}, torque={drive_torque:.2f}")

        except Exception as e:
            logger.error(f"[{self.robot_id}] Failed to forward control: {e}")

    async def _camera_stream_loop(self):
        """
        Continuously push camera frames to the smartphone as Base64 JSON.

        Reads from the double-buffer system at CAMERA_FPS.
        If CAMERA_OPTIMIZE=True, the JPEG is resized/recompressed before sending.
        The loop exits on CancelledError (cleanup) or when is_streaming becomes False.

        MESSAGE FORMAT sent to phone:
          { "type": "camera", "image": "<base64>", "robot_id": "R1" }
        """
        frame_interval = 1.0 / CAMERA_FPS  # e.g., 1/15 ≈ 0.067s

        while self.is_streaming and self.smartphone_ws:
            try:
                if not self.smartphone_ws.closed:
                    jpeg_data = self._read_latest_camera_frame()

                    if jpeg_data:
                        # Optionally resize/recompress for lower bandwidth
                        if CAMERA_OPTIMIZE:
                            jpeg_data = self._optimize_image(jpeg_data)

                        # Base64-encode for JSON transport
                        b64_image = base64.b64encode(jpeg_data).decode('utf-8')
                        message = {
                            'type': 'camera',
                            'image': b64_image,
                            'robot_id': self.robot_id
                        }
                        await self._send_to_smartphone(message)

                await asyncio.sleep(frame_interval)

            except asyncio.CancelledError:
                logger.info(f"[{self.robot_id}] Camera stream cancelled")
                break
            except Exception as e:
                logger.error(f"[{self.robot_id}] Camera stream error: {e}")
                await asyncio.sleep(frame_interval)

    async def _soc_stream_loop(self):
        """
        Periodically push battery SOC (State of Charge) data to the smartphone.

        Reads Robot{N}/data_interactive/latest_SOC.txt at SOC_UPDATE_INTERVAL.
        SOC is a float 0.0 (empty) to 1.0 (full) written by the Unity client.
        This is the default STREAM_MODE as it has very low overhead.

        MESSAGE FORMAT sent to phone:
          { "type": "soc", "soc": 0.85, "robot_id": "R1" }
        """
        while self.is_streaming and self.smartphone_ws:
            try:
                if not self.smartphone_ws.closed:
                    soc = self._read_latest_soc()

                    message = {
                        'type': 'soc',
                        'soc': soc,
                        'robot_id': self.robot_id
                    }
                    await self._send_to_smartphone(message)

                await asyncio.sleep(SOC_UPDATE_INTERVAL)

            except asyncio.CancelledError:
                logger.info(f"[{self.robot_id}] SOC stream cancelled")
                break
            except Exception as e:
                logger.error(f"[{self.robot_id}] SOC stream error: {e}")
                await asyncio.sleep(SOC_UPDATE_INTERVAL)

    def _read_latest_soc(self) -> Optional[float]:
        """
        Read the current SOC value from the data_interactive directory.
        Returns None if the file is missing or unreadable.
        """
        try:
            soc_file = self.data_dir / "latest_SOC.txt"
            if soc_file.exists():
                soc_str = soc_file.read_text(encoding="utf-8").strip()
                return float(soc_str)
            return None
        except Exception:
            return None

    def _optimize_image(self, jpeg_data: bytes) -> Optional[bytes]:
        """
        Optionally resize and recompress a JPEG for lower-bandwidth streaming.

        Only called when CAMERA_OPTIMIZE=True.
        Resizes to CAMERA_MAX_WIDTH (maintaining aspect ratio) if specified,
        then re-encodes at CAMERA_QUALITY (1-100).
        Falls back to the original bytes if PIL is unavailable or fails.
        """
        try:
            from PIL import Image
            from io import BytesIO

            img = Image.open(BytesIO(jpeg_data))

            # Resize if width exceeds the configured maximum
            if CAMERA_MAX_WIDTH and img.width > CAMERA_MAX_WIDTH:
                ratio = CAMERA_MAX_WIDTH / img.width
                new_height = int(img.height * ratio)
                img = img.resize((CAMERA_MAX_WIDTH, new_height), Image.LANCZOS)

            # Re-encode at lower quality
            output = BytesIO()
            img.save(output, format='JPEG', quality=CAMERA_QUALITY, optimize=True)
            return output.getvalue()

        except Exception as e:
            logger.debug(f"[{self.robot_id}] Image optimization error: {e}")
            return jpeg_data  # Return original bytes if optimization fails

    def _read_latest_camera_frame(self) -> Optional[bytes]:
        """
        Read the most recently written camera frame using the double-buffer system.

        DOUBLE-BUFFER PROTOCOL:
          The Python WebSocket client writes frames alternately to:
            latest_RGB_a.jpg
            latest_RGB_b.jpg
          and updates latest_RGB_now.txt to contain "a" or "b" only after
          the write is complete. This ensures we never read a partial JPEG.

        INTEGRITY CHECK:
          A valid JPEG ends with bytes 0xFF 0xD9 (FFD9 end marker).
          Frames missing this marker are discarded to avoid sending corrupt data.

        Returns:
            JPEG bytes if a valid frame is available, None otherwise.
        """
        try:
            if not self.data_dir.exists():
                return None

            if not self.buffer_pointer_file.exists():
                return None

            # Read which buffer (a or b) contains the latest complete frame
            with open(self.buffer_pointer_file, 'r') as f:
                buffer_name = f.read().strip()  # "a" or "b"

            jpeg_path = self.data_dir / f"latest_RGB_{buffer_name}.jpg"

            if not jpeg_path.exists():
                return None

            with open(jpeg_path, 'rb') as f:
                jpeg_data = f.read()

            # Validate JPEG integrity: file must end with FFD9
            if len(jpeg_data) > 2 and jpeg_data[-2:] == b'\xff\xd9':
                return jpeg_data
            else:
                logger.warning(f"[{self.robot_id}] Incomplete JPEG (missing FFD9)")
                return None

        except Exception as e:
            logger.debug(f"[{self.robot_id}] Camera read error: {e}")
            return None

    async def _send_to_smartphone(self, message: dict):
        """Send a JSON-serialized message to the smartphone WebSocket."""
        if self.smartphone_ws and not self.smartphone_ws.closed:
            try:
                await self.smartphone_ws.send_str(json.dumps(message))
            except Exception as e:
                logger.error(f"[{self.robot_id}] Failed to send to smartphone: {e}")

    async def _cleanup(self):
        """
        Release resources when the smartphone disconnects.
        Cancels the streaming task and clears the WebSocket reference.
        """
        logger.info(f"[{self.robot_id}] Smartphone disconnected")

        # Cancel the background streaming task (camera or SOC)
        if self.stream_task and not self.stream_task.done():
            self.stream_task.cancel()
            try:
                await self.stream_task
            except asyncio.CancelledError:
                pass

        self.smartphone_ws = None
        self.is_streaming = False


class SmartphoneServer:
    """
    Unified HTTP + WebSocket server for smartphone game controllers.

    ROUTES:
      GET /                    — Index page with robot selection links
      GET /controller?robot=R1 — Serve smartphone_controller.html for robot R1
      GET /qr/{robot_id}       — Generate + serve QR code PNG for robot_id
      GET /ws/R1 … /ws/R5      — WebSocket endpoints for smartphone control

    LIFECYCLE (managed by main.py):
      1) SmartphoneServer(port=8080) — create instance
      2) server.register_robot(robot_id, ws_client) — bind Unity clients
      3) await server.start() — start aiohttp, generate QR codes, open popups
      4) await server.wait_for_all_ready(...) — block until all L+R confirmed
      5) [race runs] ...
      6) await server.shutdown() — cancel streams, close phone connections
    """

    def __init__(self, port: int = 8080):
        self.port = port
        self.app = web.Application()
        self.controllers: Dict[str, RobotController] = {}

        # tkinter QR popup state (per-robot windows)
        self._qr_window = None
        self._qr_close_requested = False

        self._setup_routes()

        logger.info(f"Smartphone server initialized on port {self.port}")

    def _setup_routes(self):
        """Register all HTTP and WebSocket routes with aiohttp."""
        self.app.router.add_get('/', self.handle_index)
        self.app.router.add_get('/controller', self.handle_controller_page)
        self.app.router.add_get('/qr/{robot_id}', self.handle_qr_code)

        # Register WebSocket endpoint for each possible robot (R1-R5)
        for i in range(1, 6):
            robot_id = f"R{i}"
            self.app.router.add_get(f'/ws/{robot_id}', self.handle_websocket)

    def register_robot(self, robot_id: str, websocket_client=None):
        """
        Register a robot controller.
        If the robot is already registered, update its Unity WS client reference.
        """
        if robot_id not in self.controllers:
            self.controllers[robot_id] = RobotController(robot_id, websocket_client, server=self)
            logger.info(f"Registered robot: {robot_id}")
        else:
            self.controllers[robot_id].set_websocket_client(websocket_client)

    async def wait_for_all_ready(self, timeout: float = 300.0, stop_event=None):
        """
        Block until all registered robots have confirmed connection via L+R.

        First sends a 'reset' message to all currently connected smartphones
        (handles the case where a phone is already connected from a previous race).

        Polls at 0.5s intervals; logs progress every 5 seconds.

        Args:
            timeout: Maximum wait time in seconds (default 300s = 5 minutes).
            stop_event: threading.Event; if set (user pressed 'q'), returns False.

        Returns:
            True if all robots confirmed; False on timeout or cancellation.
        """
        # Reset all controllers: clear is_ready and notify phones to reset UI
        for robot_id, ctrl in self.controllers.items():
            ctrl.is_ready = False
            ctrl.ready_event.clear()
            # Re-send reset to already-connected smartphones (e.g., reconnect scenario)
            if ctrl.smartphone_ws and not ctrl.smartphone_ws.closed:
                try:
                    await ctrl.smartphone_ws.send_str(json.dumps({
                        'type': 'reset',
                        'message': 'New race starting. Press L+R buttons again.'
                    }))
                    logger.info(f"[{robot_id}] Reset signal sent to smartphone")
                except Exception as e:
                    logger.warning(f"[{robot_id}] Failed to send reset: {e}")

        logger.info(f"Waiting for {len(self.controllers)} robot(s) to confirm connection...")
        logger.info("Please scan QR code and press BOTH L+R buttons:")

        start_time = asyncio.get_event_loop().time()
        last_progress = -1  # Tracks which 5-second interval was last logged

        while True:
            # Allow cancellation via 'q' key
            if stop_event and stop_event.is_set():
                logger.info("Cancelled by user")
                return False

            # Check if all robots are confirmed
            all_ready = all(ctrl.is_ready for ctrl in self.controllers.values())
            if all_ready:
                logger.info("✓ All robots are ready!")
                return True

            # Check timeout
            elapsed = asyncio.get_event_loop().time() - start_time
            if elapsed > timeout:
                logger.warning(f"Timeout waiting for readiness ({timeout}s)")
                ready_count = sum(1 for ctrl in self.controllers.values() if ctrl.is_ready)
                logger.warning(f"Only {ready_count}/{len(self.controllers)} robots are ready")
                return False

            # Log progress once per 5-second interval
            current_5s = int(elapsed) // 5
            if current_5s > last_progress:
                last_progress = current_5s
                ready_count = sum(1 for ctrl in self.controllers.values() if ctrl.is_ready)
                logger.info(f"Progress: {ready_count}/{len(self.controllers)} ready ({int(elapsed)}s elapsed)")

            await asyncio.sleep(0.5)

    async def shutdown(self):
        """
        Gracefully stop all streaming tasks and close all smartphone connections.
        Called by main.py after the race ends (Phase 7 cleanup).
        """
        logger.info("Shutting down smartphone server...")

        for robot_id, controller in self.controllers.items():
            # Stop streaming flag first to prevent new sends
            controller.is_streaming = False

            # Cancel the background streaming task (camera or SOC)
            if controller.stream_task and not controller.stream_task.done():
                controller.stream_task.cancel()
                try:
                    await controller.stream_task
                except asyncio.CancelledError:
                    pass
                logger.info(f"[{robot_id}] Streaming task cancelled")

            # Close the smartphone WebSocket connection
            if controller.smartphone_ws and not controller.smartphone_ws.closed:
                try:
                    await controller.smartphone_ws.close()
                except Exception as e:
                    logger.warning(f"[{robot_id}] Error closing smartphone connection: {e}")

        logger.info("Smartphone server shutdown complete")

        """
        return web.Response(text=html, content_type='text/html')

    async def handle_controller_page(self, request):
        """
        Serve the smartphone controller HTML page for a specific robot.

        Loads smartphone_controller.html from the project directory and substitutes
        the '{{ROBOT_ID}}' placeholder with the requested robot ID.
        Falls back to a minimal inline HTML if the file is not found.
        """
        robot_id = request.query.get('robot', 'R1')

        html_path = Path(__file__).parent / 'smartphone_controller.html'

        if html_path.exists():
            with open(html_path, 'r', encoding='utf-8') as f:
                html = f.read()

            # Template substitution: replace robot ID placeholder
            html = html.replace('{{ROBOT_ID}}', robot_id)

            return web.Response(text=html, content_type='text/html')
        else:
            # Fallback minimal page (controller HTML not created yet)
            html = f"""
            <!DOCTYPE html>
            <html>
            <head><title>Robot {robot_id} Controller</title></head>
            <body>
                <h1>Robot {robot_id} Controller</h1>
                <p>Controller page not found. Create smartphone_controller.html</p>
            </body>
            </html>
            """
            return web.Response(text=html, content_type='text/html')

    async def handle_qr_code(self, request):
        """
        Dynamically generate and return a QR code PNG for the given robot_id.
        The QR code encodes the full controller URL including the local IP.
        Robot-specific fill color is used for easy visual identification.
        Returns HTTP 503 if the qrcode library is not installed.
        """
        robot_id = request.match_info.get('robot_id', 'R1')

        if not QRCODE_AVAILABLE:
            return web.Response(text='QR code generation not available. Install qrcode library.', status=503)

        # Build the full controller URL using local IP (not localhost, for LAN access)
        local_ip = self._get_local_ip()
        controller_url = f"http://{local_ip}:{self.port}/controller?robot={robot_id}"

        # Generate QR code with robot-specific fill color
        qr = qrcode.QRCode(
            version=1,
            error_correction=qrcode.constants.ERROR_CORRECT_L,
            box_size=10,
            border=4,
        )
        qr.add_data(controller_url)
        qr.make(fit=True)

        qr_color = QR_COLORS.get(robot_id, 'black')
        img = qr.make_image(fill_color=qr_color, back_color="white")

        # Encode PNG to bytes and return as HTTP response
        from io import BytesIO
        buffer = BytesIO()
        img.save(buffer, format='PNG')
        buffer.seek(0)

        return web.Response(body=buffer.read(), content_type='image/png')

    async def handle_websocket(self, request):
        """
        Handle incoming WebSocket connection from a smartphone.

        Extracts robot_id from the URL path (/ws/R1 → "R1") and delegates
        to the appropriate RobotController.handle_smartphone_connection().
        Returns 404 if the robot was not registered.
        """
        # Extract robot_id from path: /ws/R1 → "R1"
        path = request.path
        robot_id = path.split('/')[-1]

        if robot_id not in self.controllers:
            logger.warning(f"Unknown robot: {robot_id}")
            return web.Response(text='Robot not found', status=404)

        # Upgrade HTTP request to WebSocket
        ws = web.WebSocketResponse()
        await ws.prepare(request)

        controller = self.controllers[robot_id]
        await controller.handle_smartphone_connection(ws)

        return ws

    async def start(self):
        """
        Start the aiohttp HTTP server and display QR codes.

        Binds to 0.0.0.0 (all interfaces) so smartphones on the same LAN
        can reach it using the machine's local IP address.
        Generates QR code PNG files and opens tkinter popup windows.
        """
        runner = web.AppRunner(self.app)
        await runner.setup()

        # Bind to all network interfaces (0.0.0.0) for LAN access
        site = web.TCPSite(runner, '0.0.0.0', self.port)
        await site.start()

        local_ip = self._get_local_ip()
        logger.info(f"=" * 60)
        logger.info(f"📱 Smartphone Server Running")
        logger.info(f"=" * 60)
        logger.info(f"Local URL:  http://localhost:{self.port}")
        logger.info(f"Network URL: http://{local_ip}:{self.port}")
        logger.info(f"=" * 60)
        logger.info(f"QR Codes available at:")

        for robot_id in self.controllers.keys():
            logger.info(f"  {robot_id}:")
            logger.info(f"    Controller: http://{local_ip}:{self.port}/controller?robot={robot_id}")
            logger.info(f"    QR Code:    http://{local_ip}:{self.port}/qr/{robot_id}")

        logger.info(f"=" * 60)

        # Generate QR code PNGs and show popup windows
        if QRCODE_AVAILABLE:
            self._generate_qr_code_files(local_ip)
        else:
            logger.warning("QR code generation skipped (qrcode library not installed)")
            logger.warning("Install with: pip install qrcode[pil]")

    def _generate_qr_code_files(self, local_ip: str):
        """
        Save QR code PNG files to qr_codes/ directory (one per registered robot).
        After saving, opens tkinter popup windows to display them.
        """
        qr_dir = Path("qr_codes")
        qr_dir.mkdir(exist_ok=True)

        for robot_id in self.controllers.keys():
            controller_url = f"http://{local_ip}:{self.port}/controller?robot={robot_id}"

            qr = qrcode.QRCode(
                version=1,
                error_correction=qrcode.constants.ERROR_CORRECT_L,
                box_size=10,
                border=4,
            )
            qr.add_data(controller_url)
            qr.make(fit=True)

            qr_color = QR_COLORS.get(robot_id, 'black')
            img = qr.make_image(fill_color=qr_color, back_color="white")

            qr_file = qr_dir / f"{robot_id}_controller.png"
            img.save(qr_file)

            logger.info(f"  QR code saved: {qr_file}")

        logger.info(f"All QR codes saved to {qr_dir}/")

        # Open a popup window for each robot showing the QR code
        self._show_qr_popups(qr_dir, local_ip)

    def _show_qr_popups(self, qr_dir: Path, local_ip: str):
        """
        Display tkinter popup windows showing QR codes (one per robot).

        WHY A SEPARATE THREAD:
          tkinter requires its own thread and a blocking mainloop() call.
          Running it on the asyncio event loop thread would block all
          async operations. A daemon thread is used so it exits automatically
          when the main program exits.

        WINDOW BEHAVIOR:
          - Each robot gets its own Toplevel window positioned side-by-side.
          - Windows close automatically when the smartphone connects
            (triggered by close_qr_popup() → sets 'close_requested' flag).
          - A periodic after(100ms) callback polls the flag to destroy windows
            (cross-thread safe: tkinter is not thread-safe, so direct destroy()
            from the asyncio thread is avoided).
          - If all windows are closed, root.quit() stops the tkinter mainloop.

        Falls back gracefully if PIL/tkinter is not available.
        """
        try:
            from PIL import Image, ImageDraw, ImageFont, ImageTk
            import tkinter as tk
            import threading

            # Per-robot window state
            self._qr_windows = {}  # {robot_id: {'window': Toplevel, 'close_requested': bool}}
            self._tk_root = None
            self._tk_images = {}   # Keep PhotoImage references to prevent GC

            def run_tk_loop():
                """tkinter main loop running in a daemon thread."""
                try:
                    root = tk.Tk()
                    root.withdraw()  # Hide the invisible root window
                    self._tk_root = root

                    # Create one Toplevel window per registered robot
                    for idx, robot_id in enumerate(self.controllers.keys()):
                        qr_file = qr_dir / f"{robot_id}_controller.png"
                        if not qr_file.exists():
                            logger.warning(f"QR code file not found for {robot_id}")
                            continue

                        qr_img = Image.open(qr_file)
                        qr_width, qr_height = qr_img.size

                        # Layout: QR image + title + URL + instruction
                        padding = 30
                        text_height = 70
                        total_width = qr_width + padding * 2
                        total_height = qr_height + text_height + padding * 2 + 30

                        canvas = Image.new('RGB', (total_width, total_height), 'white')
                        draw = ImageDraw.Draw(canvas)

                        # Use Arial if available, fall back to default font
                        try:
                            title_font = ImageFont.truetype("arial.ttf", 28)
                            url_font = ImageFont.truetype("arial.ttf", 12)
                        except:
                            title_font = ImageFont.load_default()
                            url_font = ImageFont.load_default()

                        # Robot-specific title color for easy identification
                        color_map = {'R1': 'red', 'R2': 'green', 'R3': 'blue', 'R4': 'orange', 'R5': 'purple'}
                        title_color = color_map.get(robot_id, 'black')

                        draw.text((padding, padding), f"Robot {robot_id}", fill=title_color, font=title_font)
                        url = f"http://{local_ip}:{self.port}/controller?robot={robot_id}"
                        draw.text((padding, padding + 35), url, fill='blue', font=url_font)
                        canvas.paste(qr_img, (padding, padding + text_height))
                        draw.text(
                            (padding, total_height - 25),
                            "Scan QR -> Press L+R buttons",
                            fill='gray',
                            font=url_font
                        )

                        # Toplevel window (child of hidden root)
                        window = tk.Toplevel(root)
                        window.title(f"QR Code - {robot_id}")
                        window.resizable(False, False)

                        # Position windows side-by-side on screen
                        x_offset = 100 + idx * (total_width + 50)
                        window.geometry(f"+{x_offset}+100")

                        # Convert PIL image to tkinter-compatible PhotoImage
                        tk_image = ImageTk.PhotoImage(canvas)
                        self._tk_images[robot_id] = tk_image  # Prevent GC

                        label = tk.Label(window, image=tk_image)
                        label.pack()

                        self._qr_windows[robot_id] = {
                            'window': window,
                            'close_requested': False
                        }

                        # Closing the window via title bar removes it from tracking
                        def make_on_close(rid):
                            def on_close():
                                if rid in self._qr_windows:
                                    self._qr_windows[rid]['window'].destroy()
                                    del self._qr_windows[rid]
                                if not self._qr_windows:
                                    root.quit()
                            return on_close

                        window.protocol("WM_DELETE_WINDOW", make_on_close(robot_id))

                        logger.info(f"QR code popup window opened for {robot_id}")

                    def check_close_requests():
                        """
                        Poll for close requests set by close_qr_popup().
                        Called every 100ms from the tkinter event loop.
                        Thread-safe: only tkinter thread calls destroy().
                        """
                        for rid in list(self._qr_windows.keys()):
                            if self._qr_windows[rid]['close_requested']:
                                self._qr_windows[rid]['window'].destroy()
                                del self._qr_windows[rid]
                                logger.info(f"QR code popup closed for {rid}")
                        if not self._qr_windows:
                            root.quit()  # Stop mainloop when all windows are gone
                        else:
                            root.after(100, check_close_requests)

                    root.after(100, check_close_requests)
                    root.mainloop()

                except Exception as e:
                    logger.error(f"Failed to show QR popups: {e}")

            # Run tkinter in a daemon thread so it doesn't prevent program exit
            popup_thread = threading.Thread(target=run_tk_loop, daemon=True)
            popup_thread.start()

        except ImportError as e:
            logger.warning(f"Required library not available for QR popup display: {e}")
        except Exception as e:
            logger.error(f"Error creating QR popups: {e}")

    def close_qr_popup(self, robot_id: str = None):
        """
        Request closure of QR code popup window(s).

        THREAD SAFETY:
          This method may be called from the asyncio thread (via
          handle_smartphone_connection), while the tkinter windows live on a
          different thread. Direct window.destroy() would be unsafe, so this
          method only sets a 'close_requested' flag. The tkinter thread's
          periodic check_close_requests() callback reads the flag and destroys
          the window safely from within the tkinter thread.

        Args:
            robot_id: Close only this robot's window. None → close all windows.
        """
        if robot_id:
            if hasattr(self, '_qr_windows') and robot_id in self._qr_windows:
                try:
                    self._qr_windows[robot_id]['close_requested'] = True
                    logger.info(f"QR code popup close requested for {robot_id}")
                except Exception as e:
                    logger.warning(f"Error requesting QR popup close for {robot_id}: {e}")
        else:
            # Close all windows (used for full server shutdown)
            if hasattr(self, '_qr_windows'):
                for rid in list(self._qr_windows.keys()):
                    try:
                        self._qr_windows[rid]['close_requested'] = True
                    except:
                        pass
                logger.info("QR code popup close requested for all")

    @staticmethod
    def _get_local_ip() -> str:
        """
        Determine the local network IP address.

        Uses a UDP connect to 8.8.8.8:80 (no data is sent) to ask the OS
        which network interface would be used to reach the internet.
        The resulting socket address is the local IP for LAN access.
        Falls back to 127.0.0.1 if the trick fails.
        """
        try:
            s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
            s.connect(("8.8.8.8", 80))  # No actual data sent; just gets local IP
            ip = s.getsockname()[0]
            s.close()
            return ip
        except Exception:
            return "127.0.0.1"
