# main.py
# ============================================================
# FILE: main.py
# ROLE: Top-level entry point for Virtual Robot Race Beta 1.6
#
# OVERVIEW:
#   Orchestrates the entire race session in 8 phases:
#     1) Load all active robot configs (Robot1/robot_config.txt, etc.)
#     2) Launch Unity executable (WebSocket server on port 12346)
#     3) Wait for Unity WebSocket server to be ready (exponential backoff)
#     4) Connect all RobotWebSocketClients (one per active robot)
#     4.5) Preload AI models + CUDA warmup (before race start, to avoid delay)
#     4.7) Start smartphone server if MODE=smartphone (wait for L+R confirmation)
#     4.8) Send ready signals to Unity (triggers race countdown)
#     5) Start all control modules + receive loops concurrently
#     6) Wait for stop_event (race end or user 'q' key)
#     7) Graceful shutdown + metadata wait (up to 1.5s after force-end)
#     8) Build output_video.mp4 for each robot (if AUTO_MAKE_VIDEO=1)
#
# CONTROL MODES (MODE_NUM in robot_config.txt):
#   1=keyboard, 2=table, 3=rule_based, 4=ai, 5=smartphone, 6=rl_training
#
# KEY DESIGN DECISIONS:
#   - Uses importlib.util.spec_from_file_location() to load Robot{N}/ modules.
#     This avoids Python module cache collisions when Robot1 and Robot2 use
#     modules with the same filename (e.g., both have keyboard_input.py).
#     Each robot gets its own isolated module object.
#   - LogCapture intercepts sys.stdout/sys.stderr at module load time so that
#     the terminal log can be saved to a file after the session ends.
#   - auto_rename_images() uses a 2-phase rename (→.tmp.jpg → final) to safely
#     handle cases where the new filename is already used by another file.
# ============================================================

# Terminal Log Capture

class LogCapture:
    """
    Intercepts sys.stdout or sys.stderr to capture all terminal output.
    Captured text is also passed through to the original stream, so the
    terminal still shows all messages in real time.

    Usage:
        sys.stdout = LogCapture(sys.stdout)
        # Now all print() calls are captured in .logs as well as displayed.
    """
    def __init__(self, original_stream):
        self.logs = []               # Accumulated log strings
        self.original = original_stream  # The real stdout/stderr

    def write(self, text):
        # Only capture non-empty strings to avoid blank list entries
        if text:
            self.logs.append(text)
            self.original.write(text)

    def flush(self):
        self.original.flush()

    def get_log_text(self) -> str:
        """Return all captured log text as a single string."""
        return "".join(self.logs)


# Install log capture at module load time.
# Installed before any other imports so that ALL print() and error output
# is captured from the very beginning of the session.
_stdout_capture = LogCapture(sys.stdout)
_stderr_capture = LogCapture(sys.stderr)
sys.stdout = _stdout_capture
sys.stderr = _stderr_capture


def get_terminal_log() -> str:
    """
    Retrieve all captured terminal output (stdout + stderr).
    Called by data_manager at the end of a session to save the log to disk.
    """
    # Combine stdout and stderr logs, with a clear section divider
    stdout_log = _stdout_capture.get_log_text()
    stderr_log = _stderr_capture.get_log_text()
    if stderr_log:
        return stdout_log + "\n=== STDERR ===\n" + stderr_log
    return stdout_log


# Register the terminal log getter with data_manager so it can save
# the session log automatically when saving other race data.
dm.register_terminal_log_getter(get_terminal_log)

def launch_unity_exe() -> Optional[subprocess.Popen]:
    """
    Launch the built Unity .exe if found in Windows/ subdirectory.
    Returns the Popen handle so we can terminate Unity when Python exits.
    Returns None if the .exe is not found.
    """
    base_dir = os.path.dirname(os.path.abspath(__file__))
    exe_path = os.path.join(base_dir, "Windows", "VirtualRobotRace_Beta.exe")

    if os.path.exists(exe_path):
        print(f"[Main] Launching Unity server: {exe_path}")
        proc = subprocess.Popen([exe_path], shell=False)
        print(f"[Main] Unity server launched (PID: {proc.pid})")
        return proc
    else:
        # Expected when running from Unity Editor (no .exe built yet)
        print(f"[Main] Unity .exe not found at: {exe_path}")
        return None


async def wait_for_unity_server(server_url: str, timeout: float = 30.0) -> bool:
    """
    Poll the Unity WebSocket server until it accepts a connection.
    Uses exponential backoff starting at 0.5s, capped at 2.0s.

    Args:
        server_url: e.g. "ws://127.0.0.1:12346/robot"
        timeout: Maximum wait time in seconds before giving up.

    Returns:
        True if connected successfully, False if timeout expired.
    """
    import websockets

    print(f"[Main] Waiting for Unity server at {server_url}...")
    start_time = time.time()
    delay = 0.5  # Initial retry delay

    while time.time() - start_time < timeout:
        try:
            # Attempt a brief connect-then-close to confirm the server is up
            async with websockets.connect(server_url) as ws:
                await ws.close()
                print(f"[Main] Unity server is ready!")
                return True
        except Exception:
            await asyncio.sleep(delay)
            delay = min(delay * 1.5, 2.0)  # Exponential backoff, max 2s

    print(f"[Main] Unity server did not start within {timeout}s")
    return False


def auto_rename_images(run_dir: Path) -> bool:
    """
    Rename image files in images/ to match the filenames recorded in metadata.csv.

    BACKGROUND:
      Unity saves images with tick-based names (e.g., frame_0012345678.jpg)
      while metadata.csv records sequential names (e.g., frame_000001.jpg).
      This mismatch must be corrected before building the output video.

    ALGORITHM:
      1) Load metadata.csv to get the expected list of filenames.
      2) Sort the actual images/ files alphabetically.
      3) Build a mapping: actual_file[i] → expected_name[i].
      4) Phase 1: Rename all mismatched files to .tmp.jpg (avoids name conflicts
         if a target filename is already used by another file in the set).
      5) Phase 2: Rename .tmp.jpg files to their final expected names.
      6) Verify all expected filenames now exist.
      7) On success, remove the backup; on failure, preserve it for recovery.

    A backup directory (images_backup/) is created before any renames.
    If the backup already exists, the rename is skipped (safety check).

    Args:
        run_dir: Path to the run_YYYYMMDD_HHMMSS directory.

    Returns:
        True if rename succeeded or was not needed; False on error.
    """
    csv_path = run_dir / "metadata.csv"
    images_dir = run_dir / "images"

    # Basic validation — both must exist to proceed
    if not csv_path.exists() or not images_dir.exists():
        print("[Main] Auto-rename skipped: metadata.csv or images/ not found")
        return False

    # Load metadata CSV (expected filename list)
    try:
        df = pd.read_csv(csv_path)
    except Exception as e:
        print(f"[Main] Auto-rename failed: Cannot load metadata.csv - {e}")
        return False

    # Get sorted list of actual JPG files in images/
    image_files = sorted([f for f in images_dir.iterdir() if f.suffix == '.jpg'])

    metadata_count = len(df)
    image_count = len(image_files)

    # File count must match for a 1-to-1 mapping to work
    if metadata_count != image_count:
        print(f"[Main] Auto-rename failed: Count mismatch (metadata={metadata_count}, images={image_count})")
        return False

    # Build rename mapping: skip files that already have the correct name
    rename_map = []
    for idx, row in df.iterrows():
        expected_name = row['filename']
        actual_name = image_files[idx].name
        if actual_name != expected_name:
            rename_map.append((image_files[idx], images_dir / expected_name))

    # If all filenames are already correct, nothing to do
    if len(rename_map) == 0:
        print("[Main] Auto-rename: All filenames already correct")
        return True

    print(f"[Main] Auto-rename: Renaming {len(rename_map)} files to match metadata...")

    # Check for existing backup (indicates a previous failed attempt)
    backup_dir = run_dir / "images_backup"
    if backup_dir.exists():
        print(f"[Main] Auto-rename skipped: Backup already exists (images_backup/)")
        return False

    try:
        # Create backup before modifying anything
        shutil.copytree(images_dir, backup_dir)
        print(f"[Main] Auto-rename: Backup created → images_backup/")

        # Phase 1: Rename each file to a temporary .tmp.jpg name.
        # This prevents conflicts when file A's new name equals file B's current name.
        temp_map = []
        for old_path, new_path in rename_map:
            temp_path = old_path.with_suffix('.tmp.jpg')
            old_path.rename(temp_path)
            temp_map.append((temp_path, new_path))

        # Phase 2: Rename .tmp.jpg files to their final expected names
        for temp_path, new_path in temp_map:
            temp_path.rename(new_path)

        # Verify all expected filenames now exist
        missing = []
        for _, row in df.iterrows():
            expected_path = images_dir / row['filename']
            if not expected_path.exists():
                missing.append(row['filename'])

        if missing:
            print(f"[Main] Auto-rename failed: {len(missing)} files missing after rename")
            return False

        print(f"[Main] Auto-rename: Successfully renamed {len(rename_map)} files")

        # Remove the backup now that rename succeeded
        try:
            shutil.rmtree(backup_dir)
            print(f"[Main] Auto-rename: Backup removed (rename successful)")
        except Exception as cleanup_error:
            print(f"[Main] Auto-rename: Warning - Could not remove backup: {cleanup_error}")

        return True

    except Exception as e:
        print(f"[Main] Auto-rename failed: {e}")
        print(f"[Main] Auto-rename: Backup preserved at images_backup/ for recovery")
        return False


async def build_video_and_open_explorer(robot_config: dict) -> None:
    """
    Post-race video pipeline:
      1) Auto-rename images to match metadata.csv.
      2) Encode images/ → output_video.mp4 using ffmpeg.
      3) Open Windows Explorer (or macOS/Linux file manager) with the video selected.

    Skipped entirely if AUTO_MAKE_VIDEO=0 or DATA_SAVE=0.

    The ffmpeg encoding runs in a thread pool executor so it does not block
    the asyncio event loop.
    """
    if not robot_config.get("AUTO_MAKE_VIDEO", 1):
        print("[Main] AUTO_MAKE_VIDEO=0 → Skip video pipeline.")
        return

    # Without saved images, there's nothing to encode
    if not robot_config.get("DATA_SAVE", 1):
        print("[Main] WARNING: AUTO_MAKE_VIDEO=1 but DATA_SAVE=0. Cannot create video without saved images. Skipping video pipeline.")
        return

    robot_id = robot_config.get("ROBOT_ID", "R1")
    run_dir = read_last_run_dir(robot_id)  # Path recorded by data_manager during the race
    if not run_dir:
        print("[Main] Post-race video pipeline skipped: last_run_dir not found.")
        return

    images_dir = run_dir / "images"
    if not images_dir.exists():
        print(f"[Main] Post-race video pipeline skipped: images dir not found → {images_dir}")
        return

    # Fix tick→sequential name mismatch BEFORE encoding
    auto_rename_images(run_dir)

    out_path = run_dir / "output_video.mp4"
    fps = robot_config.get("VIDEO_FPS", 20)
    infer = robot_config.get("INFER_FPS", 1)

    print(f"[Main] Building MP4 → {out_path} (fps={fps}, infer_fps={infer})")

    loop = asyncio.get_running_loop()

    def _encode():
        # Synchronous ffmpeg call; runs in a thread pool to avoid blocking
        make_video.images_to_video_ffmpeg(str(images_dir), str(out_path), fps=fps, infer_fps=infer)

    await loop.run_in_executor(None, _encode)

    # Open the file manager with the video pre-selected (Windows) or run dir (others)
    try:
        if sys.platform.startswith("win") and out_path.exists():
            subprocess.Popen(["explorer", f"/select,{str(out_path)}"])
            print("[Main] Explorer opened with the MP4 selected.")
        else:
            opener = "open" if sys.platform == "darwin" else "xdg-open"
            subprocess.Popen([opener, str(run_dir)])
            print("[Main] Opened run directory in file manager.")
    except Exception as e:
        print(f"[Main] Failed to open file manager: {e}")

    # If DATA_SAVE=0, clean up JPGs after exporting the video
    if robot_config.get("DATA_SAVE", 1) == 0:
        try:
            if images_dir.exists():
                count = 0
                for p in images_dir.glob("*.jpg"):
                    try:
                        p.unlink(missing_ok=True)
                        count += 1
                    except Exception as e:
                        print(f"[Main] Failed to delete {p.name}: {e}")
                print(f"[Main] DATA_SAVE=0 → Deleted {count} JPG files after video export.")
        except Exception as e:
            print(f"[Main] Cleanup after video failed: {e}")


async def run_control_module(client: RobotWebSocketClient, mode: str, robot_num: int, robot_config: dict):
    """
    Load and run the appropriate control module for one robot.

    IMPORTLIB PATTERN:
      Modules are loaded with importlib.util.spec_from_file_location() using
      unique module names like "Robot1.keyboard_input" and "Robot2.keyboard_input".
      This prevents Python's module cache (sys.modules) from returning Robot1's
      cached module when Robot2 tries to load its own copy of the same filename.
      Without this, concurrent Robot1 and Robot2 would share the same module
      state (e.g., same keyboard listener, same AI model instance).

    MODE DISPATCH:
      "keyboard"    → keyboard_input.py: key events → driveTorque/steerAngle
      "ai"          → inference_input.py: CNN inference → driveTorque/steerAngle
      "rule_based"  → rule_based_input.py: sliding windows + DriverModel
      "table"       → table_input.py: CSV playback
      "smartphone"  → no-op loop; SmartphoneServer forwards control independently
      "rl_training" → rl_training_input.py: PPO policy + value network updates

    All modes send control commands at ~20 Hz (asyncio.sleep(0.05)).

    Args:
        client: The connected RobotWebSocketClient for this robot.
        mode: Control mode string ("keyboard", "ai", etc.).
        robot_num: Integer robot number (1, 2, ...).
        robot_config: Config dict from robot_config.txt.
    """
    print(f"[Main] Starting control module: {mode} (Robot{robot_num})")

    # Special case: if two robots are set to keyboard mode, the second one
    # is disabled (KEYBOARD_DISABLED=True) to avoid control conflicts.
    if mode == "keyboard" and robot_config.get('KEYBOARD_DISABLED', False):
        print(f"[Main] Robot{robot_num} keyboard control is DISABLED (another robot has priority)")
        print(f"[Main] Robot{robot_num} will send zero control commands")
        # Send zero commands indefinitely to keep the robot stationary
        while not stop_event.is_set():
            control_msg = {
                "type": "control",
                "robot_id": client.robot_id,
                "driveTorque": 0.0,
                "steerAngle": 0.0
            }
            await client.send_json(control_msg)
            await asyncio.sleep(0.05)  # 20Hz
        return

    # importlib is used here (not a top-level import) to load modules lazily
    # with robot-specific names that prevent module cache collisions.
    import importlib.util
    robot_dir = Path(f"Robot{robot_num}")

    try:
        if mode == "keyboard":
            # Load Robot{N}/keyboard_input.py with a unique module name
            module_file = robot_dir / "keyboard_input.py"
            spec = importlib.util.spec_from_file_location(
                f"Robot{robot_num}.keyboard_input",
                module_file
            )
            keyboard_input = importlib.util.module_from_spec(spec)
            spec.loader.exec_module(keyboard_input)

            keyboard_input.start_listener()  # Start key event thread

            # Poll key state at 20 Hz and forward to Unity
            while not stop_event.is_set():
                try:
                    cmd = keyboard_input.get_latest_command()
                    drive = cmd.get("driveTorque", 0.0)
                    steer = cmd.get("steerAngle", 0.0)

                    control_msg = {
                        "type": "control",
                        "robot_id": client.robot_id,
                        "driveTorque": drive,
                        "steerAngle": steer
                    }
                    await client.send_json(control_msg)

                except Exception as e:
                    print(f"[Main] Keyboard control error: {e}")

                await asyncio.sleep(0.05)  # 20Hz

            keyboard_input.stop_listener()

        elif mode == "ai":
            # Use the preloaded module stored in robot_config by Phase 1.5.
            # Preloading happens before the race start to eliminate the 10+ second
            # GPU initialization delay that would otherwise occur on first inference.
            inference_input = robot_config.get('_preloaded_inference_module')
            if inference_input is None:
                # Fallback: load fresh if preloading was skipped or failed
                module_file = robot_dir / "inference_input.py"
                try:
                    spec = importlib.util.spec_from_file_location(
                        f"Robot{robot_num}.inference_input",
                        module_file
                    )
                    inference_input = importlib.util.module_from_spec(spec)
                    spec.loader.exec_module(inference_input)
                    inference_input.preload_model()
                except Exception as e:
                    print(f"[Main] Failed to load inference_input: {e}")
                    import traceback
                    traceback.print_exc()
                    return
            else:
                print(f"[Main] Using preloaded inference module for Robot{robot_num}")

            print("[Main] AI mode: Autonomous driving with neural network")

            # Poll AI inference at 20 Hz and forward results to Unity
            while not stop_event.is_set():
                try:
                    # update() reads the latest camera frame, runs CNN inference,
                    # and updates the internal command buffer.
                    if not inference_input.update():
                        # update() returns False to signal a requested stop
                        print("[Main] AI inference requested stop")
                        break

                    cmd = inference_input.get_latest_command()
                    drive = cmd.get("driveTorque", 0.0)
                    steer = cmd.get("steerAngle", 0.0)

                    control_msg = {
                        "type": "control",
                        "robot_id": client.robot_id,
                        "driveTorque": drive,
                        "steerAngle": steer
                    }
                    await client.send_json(control_msg)

                except Exception as e:
                    print(f"[Main] AI control error: {e}")

                await asyncio.sleep(0.05)  # 20Hz

        elif mode == "rule_based":
            # Load Robot{N}/rule_based_input.py with robot-specific module name
            module_file = robot_dir / "rule_based_input.py"
            spec = importlib.util.spec_from_file_location(
                f"Robot{robot_num}.rule_based_input",
                module_file
            )
            rule_based_input = importlib.util.module_from_spec(spec)
            spec.loader.exec_module(rule_based_input)

            print("[Main] Rule-based mode: Autonomous lane following")

            # Poll sliding-windows + DriverModel at 20 Hz
            while not stop_event.is_set():
                try:
                    # update() processes the latest camera frame through the full
                    # perception pipeline: HSV threshold → sliding windows → DriverModel
                    if not rule_based_input.update():
                        print("[Main] Rule-based control requested stop")
                        break

                    cmd = rule_based_input.get_latest_command()
                    drive = cmd.get("driveTorque", 0.0)
                    steer = cmd.get("steerAngle", 0.0)

                    control_msg = {
                        "type": "control",
                        "robot_id": client.robot_id,
                        "driveTorque": drive,
                        "steerAngle": steer
                    }
                    await client.send_json(control_msg)

                except Exception as e:
                    print(f"[Main] Rule-based control error: {e}")

                await asyncio.sleep(0.05)  # 20Hz

        elif mode == "table":
            # Load Robot{N}/table_input.py with robot-specific module name
            module_file = robot_dir / "table_input.py"
            spec = importlib.util.spec_from_file_location(
                f"Robot{robot_num}.table_input",
                module_file
            )
            table_input = importlib.util.module_from_spec(spec)
            spec.loader.exec_module(table_input)

            print("[Main] Table mode: Sending commands from CSV")

            # Poll CSV row index at 20 Hz
            while not stop_event.is_set():
                try:
                    cmd = table_input.get_latest_command()
                    drive = cmd.get("driveTorque", 0.0)
                    steer = cmd.get("steerAngle", 0.0)

                    control_msg = {
                        "type": "control",
                        "robot_id": client.robot_id,
                        "driveTorque": drive,
                        "steerAngle": steer
                    }
                    await client.send_json(control_msg)

                    # Log playback progress every 100 commands
                    progress = table_input.get_progress()
                    if progress[0] % 100 == 0:
                        print(f"[Main] Table mode progress: {progress[0]}/{progress[1]}")

                    # Detect end-of-CSV: both commands zero AND index at end
                    if drive == 0.0 and steer == 0.0:
                        if progress[0] >= progress[1]:
                            print(f"[Main] Table mode: End of CSV reached")
                            break

                except Exception as e:
                    print(f"[Main] Table control error: {e}")

                await asyncio.sleep(0.05)  # 20Hz

        elif mode == "smartphone":
            # In smartphone mode, SmartphoneServer handles control forwarding directly.
            # This coroutine just holds the WebSocket connection open while the
            # smartphone server is running.
            print("[Main] Smartphone mode: Control via smartphone (waiting for smartphone_server)")

            # Keep-alive loop; actual control forwarding is done by SmartphoneServer
            while not stop_event.is_set():
                await asyncio.sleep(0.1)

        elif mode == "rl_training":
            # Load Robot{N}/rl_training_input.py (currently only Robot1 supports this)
            module_file = robot_dir / "rl_training_input.py"
            try:
                spec = importlib.util.spec_from_file_location(
                    f"Robot{robot_num}.rl_training_input",
                    module_file
                )
                rl_training_input = importlib.util.module_from_spec(spec)
                spec.loader.exec_module(rl_training_input)
                rl_training_input.preload_model()   # Load DrivingNetwork + ValueNetwork
                rl_training_input.warmup_cuda()     # Initialize GPU kernels
            except Exception as e:
                print(f"[Main] Failed to load rl_training_input: {e}")
                import traceback
                traceback.print_exc()
                return

            print("[Main] RL Training mode: Reinforcement learning with policy updates")

            # Poll PPO policy at 20 Hz; rewards and gradient updates handled inside update()
            while not stop_event.is_set():
                try:
                    # update() runs policy inference, collects (s, a, r, s') transitions,
                    # and triggers PPO gradient updates when the buffer is full.
                    if not rl_training_input.update():
                        print("[Main] RL training requested stop")
                        break

                    cmd = rl_training_input.get_latest_command()
                    drive = cmd.get("driveTorque", 0.0)
                    steer = cmd.get("steerAngle", 0.0)

                    control_msg = {
                        "type": "control",
                        "robot_id": client.robot_id,
                        "driveTorque": drive,
                        "steerAngle": steer
                    }
                    await client.send_json(control_msg)

                except Exception as e:
                    print(f"[Main] RL training control error: {e}")

                await asyncio.sleep(0.05)  # 20Hz

            # Save trained model weights at end of episode
            try:
                rl_training_input.save_model()
            except Exception as e:
                print(f"[Main] Failed to save RL model: {e}")

        else:
            print(f"[Main] Unknown MODE: {mode}")

    finally:
        # importlib modules are garbage-collected naturally; no manual cleanup needed.
        pass


async def keyboard_monitor() -> None:
    """
    Background task that monitors for the 'q' key press to force-stop the session.

    Two backends are tried in order:
      1) keyboard library (cross-platform, non-blocking is_pressed()).
         May require admin privileges on Windows in some environments.
      2) msvcrt.kbhit() (Windows-only fallback, reads from console buffer).

    When 'q' is detected, stop_event is set, which causes the main() loop
    and all control module loops to terminate.

    Heartbeat messages are printed every ~5 seconds (100 × 50ms checks)
    to confirm the monitor is still running.
    """
    import sys

    print("[Main] ========================================")
    print("[Main] Keyboard Monitor Starting")
    print("[Main] Press 'q' at any time to force stop")
    print("[Main] ========================================")

    # --- Try keyboard library first ---
    keyboard_lib_works = False
    try:
        import keyboard
        print("[Main] [KeyMon] Testing keyboard library...")
        try:
            keyboard.is_pressed("q")
            keyboard_lib_works = True
            print("[Main] [KeyMon] ✓ keyboard library working")
        except Exception as e:
            print(f"[Main] [KeyMon] ✗ keyboard library test failed: {e}")
            print("[Main] [KeyMon] Will use msvcrt instead")
    except ImportError:
        print("[Main] [KeyMon] keyboard library not installed")

    if keyboard_lib_works:
        print("[Main] [KeyMon] Using keyboard library for monitoring")
        check_count = 0
        while not stop_event.is_set():
            try:
                if keyboard.is_pressed("q"):
                    print("[Main] [KeyMon] ✓✓✓ 'q' KEY DETECTED ✓✓✓")
                    print("[Main] [KeyMon] Setting stop_event...")
                    stop_event.set()
                    break
                # Heartbeat every 100 checks (~5 seconds)
                check_count += 1
                if check_count % 100 == 0:
                    print(f"[Main] [KeyMon] Heartbeat: {check_count} checks, still monitoring...")
            except Exception as e:
                print(f"[Main] [KeyMon] ERROR during monitoring: {e}")
                break
            await asyncio.sleep(0.05)  # Check every 50ms

    # --- Fallback: msvcrt on Windows ---
    if not stop_event.is_set() and not keyboard_lib_works:
        try:
            if sys.platform == 'win32':
                import msvcrt
                print("[Main] [KeyMon] Using msvcrt for monitoring (Windows fallback)")
                check_count = 0
                while not stop_event.is_set():
                    if msvcrt.kbhit():
                        key = msvcrt.getch()
                        print(f"[Main] [KeyMon] Key pressed: {key}")
                        if key == b'q' or key == b'Q':
                            print("[Main] [KeyMon] ✓✓✓ 'q' KEY DETECTED ✓✓✓")
                            print("[Main] [KeyMon] Setting stop_event...")
                            stop_event.set()
                            break
                    check_count += 1
                    if check_count % 100 == 0:
                        print(f"[Main] [KeyMon] Heartbeat: {check_count} checks, still monitoring...")
                    await asyncio.sleep(0.05)
            else:
                print("[Main] [KeyMon] Not on Windows - no fallback available")
        except Exception as e:
            print(f"[Main] [KeyMon] ERROR: {e}")

    print("[Main] [KeyMon] Monitor stopped")


async def main() -> None:
    """
    Top-level orchestration coroutine (multi-robot version).

    PHASE SUMMARY:
      Phase 1   — Load robot configs, detect keyboard mode conflicts
      Phase 2   — Launch Unity .exe (skipped in DEBUG_MODE=1)
      Phase 3   — Wait for Unity WebSocket server to be ready
      Phase 4   — Connect all RobotWebSocketClients
      Phase 1.5 — Preload AI models (avoids first-inference delay during race)
      Phase 1.6 — CUDA warmup (eliminates 10+ second GPU init on first inference)
      Phase 1.7 — Start SmartphoneServer + wait for L+R confirmation (if needed)
      Phase 1.8 — Send ready signals to Unity (triggers race countdown)
      Phase 2   — Start all control modules + receive loops concurrently
      Phase 6   — Wait for stop_event or all clients to disconnect
      Phase 7   — Graceful shutdown:
                   - If force-ended via 'q': wait up to 1.5s for Unity metadata
                   - Cancel all asyncio tasks
                   - Shutdown SmartphoneServer
                   - Close all WebSocket clients
      Phase 8   — Build output_video.mp4 for each robot

    KEYBOARD CONFLICT HANDLING:
      Only one robot can use the keyboard at a time. If multiple robots are
      configured with MODE_NUM=1 (keyboard), the first one gets keyboard
      control and the rest are marked KEYBOARD_DISABLED=True, causing them
      to send zero commands indefinitely.
    """
    global robot_clients

    print("[Main] Starting new architecture (Unity=Server, Python=Client)...")
    print(f"[Main] Active robots: {config_loader.ACTIVE_ROBOTS}")

    # ---- Phase 1: Load all active robot configs ----
    robot_configs = {}
    keyboard_robot = None  # Track which robot gets keyboard control

    for robot_num in config_loader.ACTIVE_ROBOTS:
        robot_configs[robot_num] = config_loader.get_robot_config(robot_num)
        rc = robot_configs[robot_num]
        mode_num = rc.get('MODE_NUM', 1)
        mode_str = config_loader.get_mode_string(mode_num)

        print(f"[Main] Robot{robot_num} config loaded:")
        print(f"  - ROBOT_ID: {rc.get('ROBOT_ID')}")
        print(f"  - MODE: {mode_str} (MODE_NUM={mode_num})")
        print(f"  - NAME: {rc.get('NAME', 'Player0000')}")
        print(f"  - RACE_FLAG: {rc.get('RACE_FLAG', 1)}")

        # Detect keyboard mode conflict
        if mode_str == "keyboard":
            if keyboard_robot is None:
                keyboard_robot = robot_num
                print(f"[Main] ✓ Robot{robot_num} will use keyboard control")
            else:
                print(f"[Main] ⚠ WARNING: Robot{robot_num} is set to keyboard mode, but Robot{keyboard_robot} already has keyboard control!")
                print(f"[Main] ⚠ Robot{robot_num} will be DISABLED (no control input)")
                rc['KEYBOARD_DISABLED'] = True  # Disable the conflicting robot

    unity_proc = None

    try:
        # ---- Phase 2: Launch Unity ----
        if config_loader.DEBUG_MODE == 0:
            # Production mode: auto-launch the built .exe
            unity_proc = launch_unity_exe()
            if not unity_proc:
                print("[Main] Failed to launch Unity. Exiting.")
                return
        else:
            # Debug mode: user launches Unity manually (e.g., from Unity Editor)
            print("[Main] DEBUG_MODE = 1 → Please launch Unity manually.")

        # ---- Phase 3: Wait for Unity server to be ready ----
        server_url = f"ws://{config_loader.HOST}:{config_loader.PORT}/robot"
        if not await wait_for_unity_server(server_url, timeout=30.0):
            print("[Main] Unity server did not start. Exiting.")
            return

        # ---- Phase 4: Connect all clients ----
        all_tasks = []
        robot_modes = {}  # robot_id → (mode_str, robot_num, rc)

        for i, robot_num in enumerate(config_loader.ACTIVE_ROBOTS):
            rc = robot_configs[robot_num]
            robot_id = rc.get("ROBOT_ID", f"R{robot_num}")
            mode_num = rc.get("MODE_NUM", 1)
            mode = config_loader.get_mode_string(mode_num)

            # Only the first robot sends the active_robots list to Unity
            # (so Unity knows how many robots to expect)
            client = RobotWebSocketClient(
                robot_id=robot_id,
                server_url=server_url,
                robot_config=rc,
                active_robots=config_loader.ACTIVE_ROBOTS if i == 0 else None
            )
            robot_clients[robot_id] = client
            robot_modes[robot_id] = (mode, robot_num, rc)

            await client.connect()
            print(f"[Main] Robot{robot_num} ({robot_id}) connected")

        print(f"[Main] All {len(robot_clients)} robots connected.")

        # ---- Phase 1.5: Preload AI models ----
        # Done BEFORE ready signals to Unity so that GPU initialization
        # finishes before the start lights sequence begins.
        ai_robots = [(robot_id, robot_num, rc) for robot_id, (mode, robot_num, rc) in robot_modes.items() if mode == "ai"]
        if ai_robots:
            import importlib.util  # Import here for Phase 1.5
            print(f"[Main] Preloading AI models for {len(ai_robots)} AI-mode robot(s)...")
            for robot_id, robot_num, rc in ai_robots:
                robot_dir = Path(f"Robot{robot_num}")
                module_file = robot_dir / "inference_input.py"
                try:
                    # Use a distinct module name ("_preload" suffix) to avoid
                    # conflicts with the later "inference_input" module load in
                    # run_control_module().
                    spec = importlib.util.spec_from_file_location(
                        f"Robot{robot_num}.inference_input_preload",
                        module_file
                    )
                    inference_module = importlib.util.module_from_spec(spec)
                    spec.loader.exec_module(inference_module)
                    inference_module.preload_model()
                    # Store the preloaded module so run_control_module() can reuse it
                    rc['_preloaded_inference_module'] = inference_module
                except Exception as e:
                    print(f"[Main] WARNING: Failed to preload model for Robot{robot_num}: {e}")
                    import traceback
                    traceback.print_exc()
            print("[Main] AI model preloading complete.")

            # ---- Phase 1.6: CUDA warmup ----
            # Even after model loading, the first inference call is slow because
            # PyTorch compiles CUDA kernels on first use. warmup_cuda() runs a
            # dummy inference to trigger compilation before the race starts.
            print("[Main] Warming up CUDA for AI-mode robots...")
            for robot_id, robot_num, rc in ai_robots:
                inference_module = rc.get('_preloaded_inference_module')
                if inference_module:
                    try:
                        inference_module.warmup_cuda()
                    except Exception as e:
                        print(f"[Main] WARNING: CUDA warmup failed for Robot{robot_num}: {e}")
            print("[Main] CUDA warmup complete.")

        # Start keyboard monitor early so 'q' can cancel the smartphone wait
        print("[Main] Starting keyboard monitor...")
        keyboard_task = asyncio.create_task(keyboard_monitor())
        all_tasks.append(keyboard_task)

        # ---- Phase 1.7: Start smartphone server ----
        # Must start BEFORE sending ready signals to Unity so the smartphone
        # connection can be confirmed before the race countdown begins.
        smartphone_server = None
        smartphone_modes = {robot_id: mode for robot_id, (mode, _, _) in robot_modes.items() if mode == "smartphone"}

        if smartphone_modes:
            print(f"[Main] Starting smartphone server for {len(smartphone_modes)} robot(s)...")
            from smartphone_server import SmartphoneServer

            smartphone_server = SmartphoneServer(port=8080)

            # Register each smartphone-mode robot (binds the Unity WS client)
            for robot_id in smartphone_modes.keys():
                client = robot_clients[robot_id]
                smartphone_server.register_robot(robot_id, client)

            await smartphone_server.start()
            print("[Main] Smartphone server started and ready for connections")

            # Wait until all smartphones confirm connection via L+R button press
            print("[Main] " + "=" * 50)
            print("[Main] Waiting for smartphone connection confirmation...")
            print("[Main] Instructions:")
            print("[Main]   1. Scan QR code with your smartphone")
            print("[Main]   2. Press BOTH L+R buttons at the same time")
            print("[Main]   3. Race will start automatically when confirmed")
            print("[Main]   (Press 'q' to cancel)")
            print("[Main] " + "=" * 50)

            # Timeout: 300s (5 minutes); cancelled by stop_event ('q' key)
            all_ready = await smartphone_server.wait_for_all_ready(timeout=300.0, stop_event=stop_event)

            if stop_event.is_set():
                print("[Main] Cancelled by user ('q' key pressed)")
                raise KeyboardInterrupt("User cancelled with 'q' key")

            if not all_ready:
                print("[Main] ERROR: Not all robots confirmed connection within timeout")
                print("[Main] Proceeding anyway, but connection may be unstable")
            else:
                print("[Main] ✓ All smartphones confirmed! Starting race sequence...")
                print("[Main] Waiting 3 seconds before starting...")
                await asyncio.sleep(3.0)

        # ---- Phase 1.8: Send ready signals to Unity ----
        # Unity waits for all registered robots to send a "ready" message before
        # starting the race lights sequence. This must happen after all setup
        # (models loaded, smartphones confirmed) is complete.
        print("[Main] Sending ready signals to Unity...")
        for robot_id, client in robot_clients.items():
            try:
                await client.send_ready_signal()
            except Exception as e:
                print(f"[Main] WARNING: Failed to send ready signal for {robot_id}: {e}")
        print("[Main] All ready signals sent. Unity will start race now...")

        print("[Main] Starting control modules simultaneously...")

        # ---- Phase 2: Start all control modules + receive loops ----
        # Each robot runs two concurrent tasks:
        #   control_task  — sends driveTorque/steerAngle to Unity at 20 Hz
        #   receive_task  — receives camera frames + race state from Unity
        for robot_id, client in robot_clients.items():
            mode, robot_num, rc = robot_modes[robot_id]

            control_task = asyncio.create_task(
                run_control_module(client, mode, robot_num, rc)
            )
            receive_task = asyncio.create_task(
                client.receive_loop()
            )

            all_tasks.extend([control_task, receive_task])

        print(f"[Main] All control modules started at the same time")
        print(f"[Main] Keyboard monitor active - Press 'q' to force stop")

        # ---- Phase 6: Wait for stop_event or race end ----
        force_ended = False  # True if stopped via 'q' key
        while not stop_event.is_set():
            # Check if any Unity client is still connected
            any_running = any(client.running for client in robot_clients.values())
            if not any_running:
                print("[Main] All clients stopped (server disconnected or race ended)")
                break

            await asyncio.sleep(0.1)

        # ---- Phase 7: Graceful shutdown ----
        if stop_event.is_set():
            print("[Main] Stop event detected → Sending force-end signal to Unity...")
            force_ended = True

            # Notify Unity that Python is stopping so it sends the metadata packet
            for robot_id, client in robot_clients.items():
                try:
                    force_end_msg = {
                        "type": "force_end",
                        "robot_id": robot_id,
                        "message": "Python client force-ended with 'q' key"
                    }
                    await client.send_json(force_end_msg)
                except Exception as e:
                    print(f"[Main] Failed to send force_end to {robot_id}: {e}")

        print("[Main] Shutting down...")

        if force_ended:
            # After force-end, wait up to 1.5s for Unity to send metadata.csv
            # (which records lap times, frame filenames, etc.)
            print("[Main] Force-end detected. Waiting for Unity metadata (up to 1.5s)...")

            wait_start = asyncio.get_event_loop().time()
            metadata_timeout = 1.5

            while asyncio.get_event_loop().time() - wait_start < metadata_timeout:
                all_received = True
                for robot_id, client in robot_clients.items():
                    if client.data_manager is not None:
                        meta_csv = client.data_manager.current_run_dir / "metadata.csv"
                        if meta_csv.exists():
                            try:
                                with open(meta_csv, 'r', encoding='utf-8') as f:
                                    lines = f.readlines()
                                    # Valid metadata: header + more than 1 data row
                                    # (force-end fallback has only 1 row with zeros)
                                    if len(lines) > 2:
                                        print(f"[Main] {robot_id} metadata received ({len(lines)-1} data rows)")
                                    else:
                                        all_received = False
                            except Exception as e:
                                all_received = False
                        else:
                            all_received = False

                if all_received:
                    print("[Main] All metadata received from Unity!")
                    break

                await asyncio.sleep(0.1)  # Keep receive_loop active during wait

            stop_event.set()  # Now actually stop everything

            # For robots that still didn't receive metadata, save a fallback
            for robot_id, client in robot_clients.items():
                if client.data_manager is not None:
                    meta_csv = client.data_manager.current_run_dir / "metadata.csv"
                    needs_fallback = True

                    if meta_csv.exists():
                        try:
                            with open(meta_csv, 'r', encoding='utf-8') as f:
                                lines = f.readlines()
                                if len(lines) > 2:
                                    needs_fallback = False
                        except:
                            pass

                    if needs_fallback:
                        print(f"[Main] {robot_id} did not receive metadata from Unity. Saving fallback...")
                        try:
                            client.data_manager.save_force_end_metadata()
                        except Exception as e:
                            print(f"[Main] Failed to save force-end logs for {robot_id}: {e}")
        else:
            # Normal race-end shutdown: stop immediately
            stop_event.set()

        # Cancel all asyncio tasks (control loops, receive loops, keyboard monitor)
        for task in all_tasks:
            task.cancel()

        try:
            await asyncio.gather(*all_tasks, return_exceptions=True)
        except Exception:
            pass

        # Shutdown smartphone server first (cancels streaming tasks)
        if smartphone_server is not None:
            await smartphone_server.shutdown()

        # Close all Unity WebSocket clients
        for robot_id, client in robot_clients.items():
            await client.close()
            print(f"[Main] {robot_id} closed")

        # ---- Phase 8: Build videos ----
        for robot_num in config_loader.ACTIVE_ROBOTS:
            rc = robot_configs[robot_num]
            try:
                await build_video_and_open_explorer(rc)
            except Exception as e:
                print(f"[Main] Robot{robot_num} video pipeline failed: {e}")

        print("[Main] System fully stopped.")

    except KeyboardInterrupt:
        print("[Main] KeyboardInterrupt received. Exiting...")
        stop_event.set()

    finally:
        # Always terminate Unity if we launched it, even on exception
        if unity_proc and unity_proc.poll() is None:
            print("[Main] Terminating Unity process...")
            unity_proc.terminate()
            try:
                unity_proc.wait(timeout=5)
            except subprocess.TimeoutExpired:
                unity_proc.kill()


async def _drain_all_tasks() -> None:
    """
    Cancel and await all remaining asyncio tasks.
    Called in the finally block of the __main__ entry point to ensure clean
    event loop shutdown even if main() exits with an exception.
    """
    pending = [t for t in asyncio.all_tasks() if t is not asyncio.current_task()]
    for t in pending:
        t.cancel()
    if pending:
        await asyncio.gather(*pending, return_exceptions=True)


if __name__ == "__main__":
    # Create a fresh event loop (avoids issues with loop reuse on Windows)
    loop = asyncio.new_event_loop()
    asyncio.set_event_loop(loop)
    try:
        loop.run_until_complete(main())
    finally:
        # Drain any lingering tasks, allow pending I/O to flush, then close loop
        loop.run_until_complete(_drain_all_tasks())
        loop.run_until_complete(asyncio.sleep(0.05))
        loop.close()
        print("[Main] Program exited.")
