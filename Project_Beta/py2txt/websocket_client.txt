# websocket_client.py
# ============================================================
# WebSocket Client - Python side of Virtual Robot Race Beta 1.6
# ============================================================
# PURPOSE: Manages the WebSocket connection from Python to the Unity server.
#          Each robot gets one RobotWebSocketClient instance.
#
# ARCHITECTURE:
#   Unity runs a WebSocket SERVER on port 12346.
#   Python runs this CLIENT that connects to Unity.
#   Data flow:
#     Python -> Unity: control commands (driveTorque, steerAngle), ready signals, force_end
#     Unity -> Python: camera images (binary), SOC values (JSON), metadata CSV at race end
#
# KEY RESPONSIBILITIES:
#   1. connect()          - Establish WebSocket connection and send handshake
#   2. receive_loop()     - Continuously receive messages from Unity (images, JSON)
#   3. send_json()        - Send JSON control/telemetry messages to Unity
#   4. handle_binary_message() - Save received camera frames to disk (double-buffer)
#   5. save_metadata()    - Write race result CSV and Unity log at race end
#
# DOUBLE-BUFFER IMAGE SYSTEM:
#   To avoid race conditions between image write and read, two buffer files are used:
#   latest_RGB_a.jpg and latest_RGB_b.jpg. A pointer file (latest_RGB_now.txt)
#   indicates which buffer has the most recent image. The rule-based/AI module
#   always reads the "now" buffer, while new images overwrite the other buffer.
# ============================================================

import asyncio
import json
import websockets
from pathlib import Path
import data_manager

class RobotWebSocketClient:
    """
    WebSocket client for a single robot connecting to the Unity server.

    Attributes:
        robot_id: Robot identifier string (e.g., "R1", "R2")
        server_url: WebSocket URL to connect to (e.g., "ws://localhost:12346/robot")
        robot_config: Dict of robot configuration values from robot_config.txt
        active_robots: List of active robot numbers (only first robot sends this)
        data_manager: DataManager instance for saving images (if DATA_SAVE=1)
    """

    def __init__(self, robot_id: str = "R1", server_url: str = "ws://localhost:12346/robot", robot_config: dict = None, active_robots: list = None):
        self.robot_id = robot_id
        self.server_url = server_url
        self.websocket = None
        self.running = False
        self.robot_config = robot_config or {}
        self.active_robots = active_robots  # List of active robot numbers [1, 2, ...]

        # Latest control command values (set by Unity, read by control modules)
        self.drive_torque = 0.0
        self.steer_angle = 0.0

        # DataManager for saving images to disk (if DATA_SAVE=1 in robot_config.txt)
        self.data_manager = None
        if self.robot_config.get('DATA_SAVE', 0) == 1:
            from pathlib import Path
            base_dir = Path(__file__).parent
            self.data_manager = data_manager.DataManager(base_dir, robot_id=robot_id)
            self.data_manager.start_new_run()
            print(f"[{self.robot_id}] DataManager initialized for image saving")

    async def connect(self):
        """
        Establish WebSocket connection to Unity server.
        After connecting, sends the initial handshake message.
        Raises an exception if connection fails (handled by main.py).
        """
        try:
            self.websocket = await websockets.connect(self.server_url)
            self.running = True
            print(f"[{self.robot_id}] Connected to Unity server at {self.server_url}")

            # Send handshake so Unity knows this robot's ID, name, mode, and race participation flag
            await self.send_handshake()

        except Exception as e:
            print(f"[{self.robot_id}] Connection failed: {e}")
            raise

    async def send_handshake(self):
        """
        Send initial handshake message to Unity server.
        Includes robot identity (player_name, mode, race_flag) and
        the list of active robots (only sent by the first robot connecting).
        """
        handshake = {
            "type": "connection",
            "robot_id": self.robot_id,
            "message": "Hello from Python client",
            # Robot identity (used by Unity for race result posting)
            "player_name": self.robot_config.get("NAME", "Player0000"),
            "mode": self._get_mode_string(),
            "race_flag": self.robot_config.get("RACE_FLAG", 0)
        }

        # Only the first robot sends the active_robots list (avoids duplication)
        if self.active_robots is not None:
            handshake["active_robots"] = self.active_robots
            print(f"[{self.robot_id}] Sending active_robots: {self.active_robots}")

        await self.send_json(handshake)
        print(f"[{self.robot_id}] Handshake sent")

    async def send_ready_signal(self):
        """
        Send ready signal to Unity after AI model initialization is complete.
        Unity waits for all robots to send this before starting the race countdown.
        This prevents the race from starting while CUDA warmup is still in progress
        (which can take 10+ seconds on first run).
        """
        ready_msg = {
            "type": "ready",
            "robot_id": self.robot_id,
            "message": "AI model loaded and CUDA warmed up"
        }
        await self.send_json(ready_msg)
        print(f"[{self.robot_id}] Ready signal sent to Unity")

    def _get_mode_string(self) -> str:
        """Convert MODE_NUM integer to mode string for handshake."""
        mode_num = self.robot_config.get("MODE_NUM", 1)
        mode_map = {1: "keyboard", 2: "table", 3: "rule_based", 4: "ai"}
        return mode_map.get(mode_num, "keyboard")

    async def send_json(self, data: dict):
        """Send a JSON-serialized dict as a text WebSocket message to Unity."""
        if self.websocket:
            try:
                await self.websocket.send(json.dumps(data))
            except Exception as e:
                print(f"[{self.robot_id}] Send error: {e}")

    async def send_binary(self, data: bytes):
        """Send raw binary data (e.g., image bytes) to Unity server."""
        if self.websocket:
            try:
                await self.websocket.send(data)
            except Exception as e:
                print(f"[{self.robot_id}] Binary send error: {e}")

    async def send_telemetry(self, tick: int, utc_ms: float, soc: float, status: str = "active"):
        """
        Send per-tick telemetry data to Unity server.
        Used to inform Unity of the Python client's current state.
        """
        telemetry = {
            "type": "data",
            "robot_id": self.robot_id,
            "tick": tick,
            "utc_ms": utc_ms,
            "soc": soc,
            "status": status,
            "filename": f"frame_{tick:06d}.jpg"
        }
        await self.send_json(telemetry)

    async def receive_loop(self):
        """
        Main receive loop: continuously processes incoming messages from Unity.
        - Text messages (JSON): dispatched to handle_json_message()
        - Binary messages:      dispatched to handle_binary_message() (camera frames)
        Sets self.running = False when connection closes (signals main.py to stop).
        """
        try:
            async for message in self.websocket:
                if isinstance(message, str):
                    await self.handle_json_message(message)
                elif isinstance(message, bytes):
                    await self.handle_binary_message(message)
        except websockets.exceptions.ConnectionClosed as e:
            print(f"[{self.robot_id}] Connection closed by server (code: {e.code}, reason: {e.reason})")
            print(f"[{self.robot_id}] Server shutdown detected - stopping client...")
        except Exception as e:
            print(f"[{self.robot_id}] Receive error: {e}")
        finally:
            self.running = False
            print(f"[{self.robot_id}] Client stopped (running=False)")

    async def handle_json_message(self, message: str):
        """
        Dispatch incoming JSON messages by type:
          "control"  - Unity-driven control commands (driveTorque, steerAngle)
          "connection"- Connection confirmation from Unity
          "metadata" - Race result data (CSV + Unity log) sent at race end
          "soc"      - Battery State of Charge value for inference modules to read
        """
        try:
            data = json.loads(message)
            msg_type = data.get("type", "")

            if msg_type == "control":
                # Store the latest control values (used by get_latest_control())
                self.drive_torque = data.get("driveTorque", 0.0)
                self.steer_angle = data.get("steerAngle", 0.0)
                print(f"[{self.robot_id}] Control received: drive={self.drive_torque:.3f}, steer={self.steer_angle:.3f}")

            elif msg_type == "connection":
                # Confirmation that Unity accepted our handshake
                status = data.get("status", "")
                message = data.get("message", "")
                print(f"[{self.robot_id}] Server response: {status} - {message}")

            elif msg_type == "metadata":
                # Race end: Unity sends the complete tick-by-tick metadata CSV
                print(f"[{self.robot_id}] Metadata received from Unity")
                await self.save_metadata(data)

            elif msg_type == "soc":
                # Real-time SOC update: write to file so inference/rule-based modules can read it
                soc = data.get("soc", 1.0)
                soc_file = data_manager.get_soc_file(self.robot_id)
                soc_file.write_text(str(soc), encoding="utf-8")

            else:
                print(f"[{self.robot_id}] Unknown message type: {msg_type}")

        except json.JSONDecodeError as e:
            print(f"[{self.robot_id}] JSON decode error: {e}")

    async def handle_binary_message(self, message: bytes):
        """
        Handle incoming binary (camera frame) message from Unity.

        DOUBLE-BUFFER SYSTEM:
          Two JPEG files (latest_RGB_a.jpg, latest_RGB_b.jpg) store alternating frames.
          A pointer file (latest_RGB_now.txt) = "a" or "b" tells which is newest.
          The inference/rule-based module always reads the "now" pointer buffer.
          This prevents reading a partially-written file.

        If DATA_SAVE=1, also saves each frame to training_data/run_xxx/images/.
        """
        try:
            import datetime
            receive_time = datetime.datetime.now().strftime("%H:%M:%S.%f")[:-3]

            # Get paths for the double-buffer system
            rgb_now_file = data_manager.get_rgb_now_file(self.robot_id)
            rgb_file_a = data_manager.get_rgb_file_a(self.robot_id)
            rgb_file_b = data_manager.get_rgb_file_b(self.robot_id)
            frame_name_file = data_manager.get_latest_frame_name_file(self.robot_id)

            # Toggle between buffer A and buffer B
            try:
                current = rgb_now_file.read_text(encoding="utf-8").strip()
                use_a = (current != "a")  # If currently "a", switch to "b", and vice versa
            except Exception:
                use_a = True  # Default to buffer A on first frame

            # Write new frame to the non-current buffer
            target = rgb_file_a if use_a else rgb_file_b
            target.write_bytes(message)

            # Update pointer to indicate which buffer is now current
            rgb_now_file.write_text("a" if use_a else "b", encoding="utf-8")

            # Track sequential frame count for naming
            if not hasattr(self, '_image_count'):
                self._image_count = 0
            self._image_count += 1
            frame_name = f"frame_{self._image_count:06d}.jpg"
            frame_name_file.write_text(frame_name, encoding="utf-8")

            # Save to training_data directory if DATA_SAVE=1
            if self.data_manager is not None:
                image_path = self.data_manager.images_dir / frame_name
                self.data_manager.save_image_bytes(image_path, message)

            # Periodic logging (every 100 images to reduce console spam)
            if self._image_count % 100 == 1:
                print(f"[{self.robot_id}] Image received: {len(message)} bytes (count={self._image_count})")

        except Exception as e:
            print(f"[{self.robot_id}] Error saving image: {e}")

    async def save_metadata(self, data: dict):
        """
        Save race result data sent by Unity at race end.
        Writes:
          - metadata.csv: tick-by-tick position/SOC/control data
          - unity_log.txt: Unity's internal log messages
          - terminal_log.txt: Python stdout/stderr captured during the run
        """
        try:
            if self.data_manager is None or self.data_manager.current_run_dir is None:
                print(f"[{self.robot_id}] No DataManager - metadata not saved")
                return

            # Save the tick-by-tick CSV data from Unity's DataLogger
            csv_data = data.get('csv_data', '')
            if csv_data:
                # Unescape the CSV data (reverse the JSON string escaping applied by Unity)
                csv_data = csv_data.replace('\\n', '\n').replace('\\r', '\r').replace('\\"', '"').replace('\\\\', '\\')

                metadata_file = self.data_manager.current_run_dir / "metadata.csv"
                metadata_file.write_text(csv_data, encoding='utf-8')

                line_count = len(csv_data.split('\n')) - 1  # Subtract header row
                print(f"[{self.robot_id}] Metadata CSV saved to {metadata_file} ({line_count} data rows)")
            else:
                print(f"[{self.robot_id}] Warning: No CSV data in metadata")

            # Save Unity's internal log file
            unity_log = data.get('unity_log', '')
            if unity_log:
                unity_log = unity_log.replace('\\n', '\n').replace('\\r', '\r').replace('\\"', '"').replace('\\\\', '\\')

                log_file = self.data_manager.current_run_dir / "unity_log.txt"
                log_file.write_text(unity_log, encoding='utf-8')
                print(f"[{self.robot_id}] Unity log saved to {log_file}")
            else:
                print(f"[{self.robot_id}] Warning: No Unity log in metadata")

            # Save Python terminal output (stdout + stderr captured by LogCapture in main.py)
            self.data_manager.save_terminal_log_from_main()

        except Exception as e:
            print(f"[{self.robot_id}] Error saving metadata: {e}")
            import traceback
            traceback.print_exc()

    def get_latest_control(self) -> dict:
        """Return the most recently received control command from Unity."""
        return {
            "type": "control",
            "robot_id": self.robot_id,
            "driveTorque": self.drive_torque,
            "steerAngle": self.steer_angle
        }

    async def close(self):
        """Close the WebSocket connection gracefully."""
        self.running = False
        if self.websocket:
            try:
                await self.websocket.close()
                print(f"[{self.robot_id}] Connection closed")
            except Exception:
                pass


# ===== Test main function =====
async def main():
    """Standalone test: connect to Unity and send 5 telemetry messages."""
    client = RobotWebSocketClient(robot_id="R1", server_url="ws://localhost:12346/robot")

    try:
        await client.connect()

        # Start receiving messages in the background
        receive_task = asyncio.create_task(client.receive_loop())

        # Send test telemetry every second for 5 seconds
        for i in range(5):
            await asyncio.sleep(1)
            await client.send_telemetry(
                tick=i,
                utc_ms=asyncio.get_event_loop().time() * 1000,
                soc=100.0 - i * 2,
                status="testing"
            )
            print(f"[R1] Telemetry sent: tick={i}")

        await client.close()
        await receive_task

    except Exception as e:
        print(f"[Main] Error: {e}")


if __name__ == "__main__":
    asyncio.run(main())
