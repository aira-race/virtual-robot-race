# keyboard_input.py (Robot1 version)
# ============================================================
# Keyboard Control Input Module - Robot 1
# ============================================================
# PURPOSE: Allows a human operator to manually drive Robot 1 using keyboard keys.
#          Runs a background thread that polls key states and updates drive/steer values.
#          main.py reads get_latest_command() at ~20Hz and sends the values to Unity.
#
# KEY MAPPING:
#   W       - Drive forward (increase driveTorque)
#   Z       - Drive backward (decrease driveTorque)
#   J       - Steer left (decrease steerAngle)
#   L       - Steer right (increase steerAngle)
#   (release) - Drive/steer returns to 0 (like releasing a smartphone joystick)
#
# CONTROL VALUES:
#   driveTorque: -1.0 to +1.0  (+1 = full forward)
#   steerAngle:  -0.524 to +0.524 rad  (~±30 degrees, matches Unity's max steer)
#
# NOTE: This module is loaded via importlib by main.py (not directly imported).
#       robot_id = "R1" identifies this as the Robot 1 controller.
# ============================================================

import threading
import keyboard  # Third-party library for global keyboard event hooks
import time
import sys

# Module identification (used in log messages)
MODULE_SOURCE = "Robot1"
print(f"[keyboard_input] Loaded from {MODULE_SOURCE}/")


def clear_input_buffer():
    """
    Clear the Windows keyboard input buffer to avoid 'stuck key' issues.
    Called when the listener stops. Windows-specific (uses msvcrt).
    """
    if sys.platform == "win32":
        import msvcrt
        while msvcrt.kbhit():
            msvcrt.getch()


# === Control parameters ===
TORQUE_STEP = 0.25     # Amount to increment/decrement driveTorque per polling cycle (50ms)
STEER_STEP  = 0.20     # Amount to increment/decrement steerAngle per polling cycle [rad]
MAX_TORQUE  = 1.0      # Maximum absolute drive torque
MAX_STEER   = 0.524    # Maximum absolute steer angle in radians (~30 degrees, matches Unity)

# === Key mapping ===
# Only these keys are monitored. Others are ignored.
KEY_MAP = {"w", "z", "j", "l", "i", "m"}

# === Current control values (updated by background thread, read by main.py) ===
driveTorque: float = 0.0   # Current drive torque (-1.0 to 1.0)
steerAngle:  float = 0.0   # Current steer angle in radians (-0.524 to 0.524)
robot_id:    str   = "R1"  # Robot identifier for WebSocket messages

# Internal thread state
_key_states = {k: False for k in KEY_MAP}  # True = key is currently held down
_listener_thread: threading.Thread | None = None
_stop_event: threading.Event | None = None


def _update_key_state(event):
    """
    Callback for keyboard library events. Updates _key_states dict.
    Called from the keyboard library's hook on each key press/release.
    """
    name = event.name
    if name in _key_states:
        _key_states[name] = (event.event_type == "down")


def _loop(stop_event: threading.Event):
    """
    Background thread loop: polls key states every 50ms and updates drive/steer values.
    Uses a hold-and-release model:
      - While W is held: drive increases each tick (up to MAX_TORQUE)
      - When W is released: drive immediately returns to 0
    Same for steering (J/L).
    """
    global driveTorque, steerAngle
    keyboard.hook(_update_key_state)  # Register event callback

    try:
        while not stop_event.is_set():
            # --- Drive torque control ---
            if _key_states["w"]:
                driveTorque += TORQUE_STEP   # Accelerate forward
            elif _key_states["z"]:
                driveTorque -= TORQUE_STEP   # Reverse
            else:
                driveTorque = 0.0            # No key held: stop

            # --- Steering angle control (release = return to center) ---
            if _key_states["j"]:
                steerAngle -= STEER_STEP     # Steer left (negative = left)
            elif _key_states["l"]:
                steerAngle += STEER_STEP     # Steer right (positive = right)
            else:
                steerAngle = 0.0             # No key held: center

            # Clamp values to allowed ranges
            if driveTorque >  MAX_TORQUE: driveTorque =  MAX_TORQUE
            if driveTorque < -MAX_TORQUE: driveTorque = -MAX_TORQUE
            if steerAngle  >  MAX_STEER:  steerAngle  =  MAX_STEER
            if steerAngle  < -MAX_STEER:  steerAngle  = -MAX_STEER

            time.sleep(0.05)  # Poll at 20Hz
    finally:
        clear_input_buffer()
        print("[Keyboard] Listener stopped.")


# === Public API (called by main.py) ===

def start_listener():
    """
    Start the background keyboard listener thread. Idempotent: safe to call multiple times.
    The thread runs until stop_listener() is called or the program exits.
    """
    global _listener_thread, _stop_event
    if _listener_thread and _listener_thread.is_alive():
        return  # Already running, do nothing
    _stop_event = threading.Event()
    _listener_thread = threading.Thread(target=_loop, args=(_stop_event,), daemon=True)
    _listener_thread.start()
    print("[Keyboard] Listener started.")


def stop_listener():
    """Stop the background keyboard listener thread."""
    global _listener_thread, _stop_event
    if _stop_event:
        _stop_event.set()
    _stop_event = None
    _listener_thread = None


def get_latest_command():
    """
    Return the latest control command as a dict for WebSocket sending.
    Called by main.py at ~20Hz to get the current drive/steer values.
    Values are rounded to 3 decimal places to keep messages compact.
    """
    return {
        "type": "control",
        "robot_id": robot_id,
        "driveTorque": round(float(driveTorque), 3),
        "steerAngle": round(float(steerAngle), 3),  # radians
    }


def debug_print_state():
    """Print current drive/steer values for debugging."""
    print(f"[DEBUG] drive={driveTorque:.2f}, steer={steerAngle:.2f}")
