# rl_training_input.py (Robot1 version)
# ==============================================================================
# Reinforcement Learning Training Engine  (Robot 1 only - no Robot2 equivalent)
# ==============================================================================
#
# PURPOSE:
#   Runs real-time RL training during a race session (MODE_NUM=6).
#   Unlike inference_input.py (mode=4) which uses a frozen model, this module:
#     - Runs the policy in TRAIN mode (not eval mode)
#     - Adds Gaussian exploration noise to actions
#     - Accumulates experiences in a rolling buffer
#     - Performs policy gradient updates every BUFFER_SIZE (32) steps
#     - Saves the updated model at episode end
#
# KEY HYPERPARAMETERS:
#   LEARNING_RATE     = 1e-5   (small LR for fine-tuning a pretrained model)
#   EXPLORATION_NOISE = 0.1    (std dev of noise added to drive)
#                      0.05   (half noise for steering, more conservative)
#   BUFFER_SIZE       = 32    (update every 32 experiences = ~1.6 sec at 20Hz)
#   GAMMA             = 0.99  (discount factor for returns)
#
# REWARD SIGNAL:
#   Per-step rewards calculated by rl_reward.py:
#     - Survival bonus (every step)
#     - SOC efficiency (maintain battery)
#     - Action smoothness (penalize jitter)
#     - Extreme steering penalty
#   Terminal rewards:
#     - Finish: +500, Lap1: +100, Fallen: -200
#
# LIMITATION (Beta 1.4):
#   Unity does not send real-time position/status telemetry.
#   Only SOC and step count are available during runtime.
#   Full telemetry (pos_x, pos_z, status) is only in metadata.csv at run end.
#   The reward function uses SOC + step count as a proxy for performance.
#
# ==============================================================================

def get_latest_command():
    """Return the latest control command in the expected format."""
    return {
        "type": "control",
        "robot_id": robot_id,
        "driveTorque": round(float(driveTorque), 3),
        "steerAngle": round(float(steerAngle), 3),
    }


def preload_model():
    """
    Preload the pretrained model as the RL starting point.
    RL training requires an existing model.pth (starts from imitation learning checkpoint).
    """
    print(f"[{robot_id} RL] Preloading model for reinforcement learning...")
    _load_model()
    if _model is not None:
        print(f"[{robot_id} RL] Model preloaded successfully!")
    else:
        print(f"[{robot_id} RL] WARNING: Model preload failed")


def warmup_cuda():
    """Warm up CUDA context with dummy inference (same as inference_input.py)."""
    global _model, _transform, _device

    if _model is None:
        print(f"[{robot_id} RL] CUDA warmup skipped: Model not loaded")
        return

    if _device is None or _device.type != 'cuda':
        print(f"[{robot_id} RL] CUDA warmup skipped: Using CPU")
        return

    print(f"[{robot_id} RL] Warming up CUDA context...")

    try:
        dummy_img = Image.new('RGB', (640, 480), color='black')
        dummy_tensor = _transform(dummy_img).unsqueeze(0).to(_device)
        dummy_soc = torch.tensor([[1.0]], dtype=torch.float32).to(_device)

        with torch.no_grad():
            _ = _model(dummy_tensor, dummy_soc)

        print(f"[{robot_id} RL] CUDA warmup complete!")

    except Exception as e:
        print(f"[{robot_id} RL] CUDA warmup failed: {e}")


def saturate(value, min_val=-1.0, max_val=1.0):
    """Clamp the input value within the specified range."""
    return max(min_val, min(max_val, value))


def _load_model():
    """
    Load pretrained model from Robot1/models/model.pth and set up Adam optimizer.
    Model is set to TRAIN mode (not eval mode) so weights are updated.
    """
    global _model, _optimizer, _transform, _model_loaded, _device

    if _model_loaded:
        return

    _model_loaded = True

    _device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    print(f"[{robot_id} RL] Using device: {_device}")

    model_path = Path(__file__).parent / "models" / "model.pth"

    model = DrivingNetwork()

    try:
        model.load_state_dict(torch.load(str(model_path), map_location=_device))
        model.to(_device)
        model.train()  # IMPORTANT: train() mode enables gradients and dropout
        print(f"[{robot_id} RL] Pretrained model loaded from {model_path}")
        _model = model

        # Adam optimizer for parameter updates
        _optimizer = optim.Adam(_model.parameters(), lr=LEARNING_RATE)
        print(f"[{robot_id} RL] Optimizer initialized (LR={LEARNING_RATE})")

    except FileNotFoundError:
        print(f"[{robot_id} RL] ERROR: Model not found at {model_path}")
        print(f"[{robot_id} RL] RL training requires a pretrained model!")
        _model = None
    except Exception as e:
        print(f"[{robot_id} RL] ERROR: Failed to load model: {e}")
        _model = None

    # Image preprocessing (must match training transforms)
    _transform = transforms.Compose([
        transforms.Resize((224, 224)),
        transforms.ToTensor(),
        transforms.Normalize(
            mean=[0.485, 0.456, 0.406],
            std=[0.229, 0.224, 0.225]
        )
    ])


def _add_exploration_noise(action_drive, action_steer):
    """
    Add Gaussian exploration noise to actions.
    Encourages the policy to explore different behaviors beyond its current mean.
    Steering gets half the noise of drive (steering errors are more dangerous).
    """
    noise_drive = torch.randn(1).item() * EXPLORATION_NOISE
    noise_steer = torch.randn(1).item() * EXPLORATION_NOISE * 0.5  # Less noise for steering

    noisy_drive = saturate(action_drive + noise_drive, -1.0, 1.0)
    noisy_steer = saturate(action_steer + noise_steer, -0.524, 0.524)

    return noisy_drive, noisy_steer


def _get_current_state():
    """
    Get current state from data_manager.

    NOTE: Currently limited to SOC + step count because Unity does not send
    real-time position telemetry during the race. Full pos_x/pos_z are only
    in metadata.csv at the end of the run (future Beta 1.5+ enhancement).
    """
    state = {
        'soc': data_manager.get_latest_soc(robot_id) or 1.0,
        'step': _episode_steps,
    }
    return state


def _update_policy(experiences):
    """
    Policy gradient update using collected experiences.

    Algorithm:
      1. Calculate Monte Carlo returns (discounted sum of rewards)
      2. Normalize returns (zero mean, unit std) for stability
      3. For each experience, compute forward pass and policy gradient loss
         loss = -G_t * (1 - (predicted - actual)^2)   [encourages actions with high returns]
      4. Backward pass + gradient clipping (max_norm=1.0) + optimizer step
    """
    global _model, _optimizer

    if _model is None or _optimizer is None:
        return

    if len(experiences) == 0:
        return

    # Step 1: Calculate discounted returns (Monte Carlo, no bootstrapping)
    returns = []
    G = 0
    for exp in reversed(experiences):
        G = exp['reward'] + GAMMA * G  # G_t = r_t + gamma * G_{t+1}
        returns.insert(0, G)

    returns = torch.tensor(returns, dtype=torch.float32).to(_device)

    # Step 2: Normalize returns for training stability
    if len(returns) > 1:
        returns = (returns - returns.mean()) / (returns.std() + 1e-8)

    # Step 3: Policy gradient loss and backward pass
    _optimizer.zero_grad()

    total_loss = 0
    for i, exp in enumerate(experiences):
        # Re-run forward pass with stored tensors (gradients flow through this)
        image_tensor = exp['image_tensor']
        soc_tensor = exp['soc_tensor']

        output = _model(image_tensor, soc_tensor)
        predicted_drive = output[0, 0]
        predicted_steer = output[0, 1]

        # The actions that were actually taken (with exploration noise)
        action_drive = exp['action_drive']
        action_steer = exp['action_steer']

        # Policy gradient loss: encourage actions that led to positive returns
        # Loss = -G_t * similarity(predicted, actual)
        drive_loss = -returns[i] * (1 - (predicted_drive - action_drive).pow(2))
        steer_loss = -returns[i] * (1 - (predicted_steer - action_steer).pow(2))

        total_loss += drive_loss + steer_loss

    total_loss = total_loss / len(experiences)
    total_loss.backward()

    # Gradient clipping prevents exploding gradients
    torch.nn.utils.clip_grad_norm_(_model.parameters(), max_norm=1.0)

    _optimizer.step()

    avg_return = returns.mean().item()
    print(f"[{robot_id} RL] Policy updated: {len(experiences)} experiences, avg_return={avg_return:.3f}")


def update():
    """
    Single update cycle for RL training. Called ~20 Hz by main.py.

    Flow:
      1. Wait for start signal (same as inference mode)
      2. Get current state (SOC + step)
      3. Calculate reward for previous action (using rl_reward.calculate_reward)
      4. Run policy to get action (with exploration noise)
      5. Store experience in buffer
      6. Update policy when buffer is full (BUFFER_SIZE steps)

    Returns True to continue.
    """
    global driveTorque, steerAngle, _race_started, _start_detected, _wait_frames
    global _prev_state, _experience_buffer, _episode_reward, _episode_steps

    try:
        _load_model()

        if _model is None:
            driveTorque = 0.0
            steerAngle = 0.0
            return True

        # === Get inputs ===
        soc = data_manager.get_latest_soc(robot_id) or 1.0
        rgb_path = data_manager.get_latest_rgb_path(robot_id)

        if not rgb_path or not rgb_path.exists():
            return True

        try:
            pil_img = Image.open(rgb_path).convert("RGB")
        except Exception:
            return True

        # === Start signal detection (same logic as inference_input.py) ===
        if not _race_started:
            if not _start_detected:
                if detect_start_signal(pil_img):
                    _start_detected = True
                    _wait_frames = 0
                    print(f"[{robot_id} RL] Start signal detected!")
                else:
                    _wait_frames += 1
                    if _wait_frames >= START_DETECTION_TIMEOUT_FRAMES:
                        _start_detected = True
                        print(f"[{robot_id} RL] Timeout: Assuming race started")
                    else:
                        driveTorque = 0.0
                        steerAngle = 0.0
                        if _wait_frames % 40 == 0:
                            print(f"[{robot_id} RL] Waiting for start signal...")
                        return True

            if _start_detected and not _race_started:
                _race_started = True
                _episode_reward = 0.0
                _episode_steps = 0
                print(f"[{robot_id} RL] RACE STARTED! Beginning RL training episode.")

        # === Get current state (SOC + step count) ===
        curr_state = _get_current_state()

        # === Calculate reward for the action taken in the previous step ===
        if _prev_state is not None:
            action = {'drive': driveTorque, 'steer': steerAngle}
            reward = calculate_reward(_prev_state, curr_state, action)
            _episode_reward += reward

            # Fill in the reward for the experience stored in the previous step
            if len(_experience_buffer) > 0:
                _experience_buffer[-1]['reward'] = reward
                _experience_buffer[-1]['next_state'] = curr_state

        # === Run policy to get action (no_grad for inference, grad for update) ===
        image_tensor = _transform(pil_img).unsqueeze(0).to(_device)
        soc_tensor = torch.tensor([[soc]], dtype=torch.float32).to(_device)

        with torch.no_grad():
            output = _model(image_tensor, soc_tensor)
            raw_drive = output[0, 0].item()
            raw_steer = output[0, 1].item()

        raw_drive = saturate(raw_drive, -1.0, 1.0)
        raw_steer = saturate(raw_steer, -0.524, 0.524)

        # Add exploration noise to encourage diverse behavior
        noisy_drive, noisy_steer = _add_exploration_noise(raw_drive, raw_steer)

        driveTorque = noisy_drive
        steerAngle = noisy_steer

        # === Store experience (reward will be filled next step) ===
        experience = {
            'image_tensor': image_tensor,   # Stored for gradient computation during update
            'soc_tensor': soc_tensor,
            'action_drive': noisy_drive,    # The actual action taken (with noise)
            'action_steer': noisy_steer,
            'reward': 0.0,                  # Will be filled at the start of the next step
            'state': curr_state,
            'next_state': None,
        }
        _experience_buffer.append(experience)

        # === Update policy when buffer is full ===
        if len(_experience_buffer) >= BUFFER_SIZE:
            _update_policy(_experience_buffer)
            _experience_buffer = []  # Clear buffer after update

        # === Update state for next step ===
        _prev_state = curr_state
        _episode_steps += 1

        # === Periodic logging ===
        if _episode_steps % 20 == 0:
            steer_deg = math.degrees(steerAngle)
            print(
                f"[{robot_id} RL] Step={_episode_steps}, Drive={driveTorque:+.3f}, "
                f"Steer={steerAngle:+.3f}rad({steer_deg:+.1f}deg), "
                f"SOC={soc:.2f}, Reward={_episode_reward:.2f}"
            )

        return True

    except Exception as e:
        print(f"[{robot_id} RL] Error: {e}")
        import traceback
        traceback.print_exc()
        return True


def reset():
    """Reset all RL training state (clears model, buffer, episode stats)."""
    global driveTorque, steerAngle, _model, _optimizer, _transform
    global _model_loaded, _device, _race_started, _start_detected, _wait_frames
    global _prev_state, _experience_buffer, _episode_reward, _episode_steps

    driveTorque = 0.0
    steerAngle = 0.0
    _model = None
    _optimizer = None
    _transform = None
    _model_loaded = False
    _device = None
    _race_started = False
    _start_detected = False
    _wait_frames = 0
    _prev_state = None
    _experience_buffer = []
    _episode_reward = 0.0
    _episode_steps = 0

    reset_reward_state()  # Reset action smoothness tracker in rl_reward.py

    print(f"[{robot_id} RL] State reset")


def end_episode(final_status, final_soc=None):
    """
    Called by main.py when the episode ends (race finish, fall, or 'q' key).
    Applies the terminal reward, does a final policy update, and returns episode stats.

    Args:
        final_status: str - 'Finish', 'Lap1', 'Fallen', 'Force end', etc.
        final_soc: float - remaining SOC at end (optional, for efficiency bonus)

    Returns:
        dict with episode statistics (status, total_reward, steps, final_soc)
    """
    global _episode_reward, _experience_buffer

    # Add terminal reward (large bonuses/penalties for episode outcome)
    terminal_reward = get_terminal_reward(final_status, final_soc)
    _episode_reward += terminal_reward

    # Final policy update with any remaining experiences in buffer
    if len(_experience_buffer) > 0:
        if _experience_buffer:
            _experience_buffer[-1]['reward'] += terminal_reward  # Credit last action
        _update_policy(_experience_buffer)
        _experience_buffer = []

    print(f"[{robot_id} RL] Episode ended: {final_status}")
    print(f"[{robot_id} RL] Total episode reward: {_episode_reward:.2f}, Steps: {_episode_steps}")

    return {
        'status': final_status,
        'total_reward': _episode_reward,
        'steps': _episode_steps,
        'final_soc': final_soc,
    }


def save_model(path=None):
    """Save the RL-updated model weights to disk (default: Robot1/models/model_rl.pth)."""
    global _model

    if _model is None:
        print(f"[{robot_id} RL] No model to save")
        return

    if path is None:
        path = Path(__file__).parent / "models" / "model_rl.pth"

    torch.save(_model.state_dict(), str(path))
    print(f"[{robot_id} RL] Model saved to {path}")
