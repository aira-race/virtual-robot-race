# rl_reward.py (Robot1 version)
# ==============================================================================
# Reinforcement Learning Reward Function  (Robot 1 only)
# ==============================================================================
#
# PURPOSE:
#   Defines the reward signal for RL training. Edit this file to shape the
#   agent's learning objectives. The reward function determines what behaviors
#   the agent will optimize for.
#
# REWARD COMPONENTS (per step):
#   survival_bonus      +0.10  - Reward just for staying in the race each step
#   soc_efficiency      +0.50  - Scale with current SOC (rewards energy conservation)
#   action_smoothness   +0.05  - Reward smooth, non-jerky actions
#   soc_low_penalty     -1.00  - Penalty when SOC < 0.2 (low battery warning)
#   soc_empty_penalty  -100.0  - Large penalty for dead battery (SOC <= 0)
#   extreme_action_penalty -0.10 - Discourage extreme steering (instability risk)
#
# TERMINAL REWARDS (applied at episode end):
#   finish_bonus        +500   - Completed 2 laps
#   lap_complete        +100   - Completed 1 lap
#   fallen_penalty      -200   - Robot fell off track
#
# REWARD DESIGN NOTE:
#   Terminal rewards (500, 200) are intentionally much larger than per-step rewards
#   (0.1 - 1.0) so the agent prioritizes finishing over short-term gains.
#   Keep this ratio when modifying weights.
#
# LIMITATION:
#   Currently only SOC and step count are available at runtime.
#   Position-based rewards (progress along track) would require Unity real-time
#   telemetry (planned for Beta 1.5+).
#
# ==============================================================================

# Reward weights - modify these to change agent behavior priorities
REWARD_WEIGHTS = {
    # Per-step rewards (applied every frame at ~20Hz)
    'survival_bonus': 0.1,        # Small constant reward for surviving each step
    'soc_efficiency': 0.5,        # Proportional to current SOC level (0.0 to 0.5 max)
    'action_smoothness': 0.05,    # Reward for not making sudden action changes

    # Per-step penalties
    'soc_low_penalty': -1.0,      # Triggered when SOC < SOC_LOW_THRESHOLD (0.2)
    'soc_empty_penalty': -100.0,  # Large penalty for completely dead battery
    'extreme_action_penalty': -0.1,  # Triggered when |steer| > STEERING_EXTREME_THRESHOLD

    # Terminal rewards (applied once at episode end)
    'finish_bonus': 500.0,        # Bonus for completing 2 laps (race finished)
    'lap_complete': 100.0,        # Bonus for completing lap 1 (halfway)
    'fallen_penalty': -200.0,     # Penalty for falling off track
}

# SOC threshold below which low-battery penalty is applied
SOC_LOW_THRESHOLD = 0.2  # 20% battery remaining

# Steering threshold for extreme action penalty (~26 degrees, close to the 30-degree max)
STEERING_EXTREME_THRESHOLD = 0.45  # radians

# State for action smoothness calculation (tracks previous action between calls)
_prev_action = {'drive': 0.0, 'steer': 0.0}


def calculate_reward(prev_state, curr_state, action):
    """
    Calculate per-step reward based on state transition and action taken.

    Called every frame by rl_training_input.py immediately before storing
    the experience in the buffer.

    Args:
        prev_state: dict with previous state info:
            - 'soc':  float, battery level (0.0-1.0)
            - 'step': int, current step count
        curr_state: dict with current state info (same format as prev_state)
        action: dict with the action taken:
            - 'drive': float, drive torque (-1.0 to 1.0)
            - 'steer': float, steering angle (radians)

    Returns:
        float: reward value (can be negative)

    NOTE:
        Terminal rewards (finish, lap, fallen) are NOT included here.
        They are applied separately by get_terminal_reward() when the
        episode ends.
    """
    global _prev_action
    reward = 0.0

    # === (1) Survival Bonus ===
    # Small positive reward every step encourages staying alive and in the race
    reward += REWARD_WEIGHTS['survival_bonus']

    # === (2) SOC-based Rewards ===
    curr_soc = curr_state.get('soc', 1.0)
    prev_soc = prev_state.get('soc', 1.0)

    # Reward proportional to current battery level - rewards energy efficiency
    # Higher SOC at this moment = higher reward (encourages not wasting energy)
    reward += curr_soc * REWARD_WEIGHTS['soc_efficiency']

    # Penalty for running low on battery
    if curr_soc < SOC_LOW_THRESHOLD:
        reward += REWARD_WEIGHTS['soc_low_penalty']

    # Large penalty for completely empty battery (race effectively over)
    if curr_soc <= 0:
        reward += REWARD_WEIGHTS['soc_empty_penalty']
        print(f"[Reward] SOC empty! Penalty={REWARD_WEIGHTS['soc_empty_penalty']}")

    # === (3) Action Smoothness Reward ===
    # Compares current action to previous action - rewards minimal sudden changes
    # Smoothness in range [0, 1]: 1.0 = identical actions, 0.0 = max possible change
    drive_diff = abs(action.get('drive', 0) - _prev_action.get('drive', 0))
    steer_diff = abs(action.get('steer', 0) - _prev_action.get('steer', 0))
    smoothness = 1.0 - (drive_diff + steer_diff) / 2.0  # Normalize to [0, 1] range
    reward += smoothness * REWARD_WEIGHTS['action_smoothness']

    # === (4) Extreme Action Penalty ===
    # Discourages extreme steering which often leads to instability and falls
    if abs(action.get('steer', 0)) > STEERING_EXTREME_THRESHOLD:
        reward += REWARD_WEIGHTS['extreme_action_penalty']

    # Update previous action for next step's smoothness calculation
    _prev_action = action.copy()

    return reward


def get_terminal_reward(final_status, final_soc=None, race_time_ms=None):
    """
    Get terminal reward based on how the episode ended.
    Called once by rl_training_input.end_episode() when the race finishes.

    Args:
        final_status: str - Unity-provided status string:
            'Finish'     - Completed 2 laps
            'Lap1'       - Completed 1 lap
            'Fallen'     - Robot fell off track
            'Force end'  - User pressed 'q' (neutral, no bonus/penalty)
        final_soc: float - remaining battery (0.0-1.0) at episode end
        race_time_ms: int - total race time in milliseconds (for time-based bonus)

    Returns:
        float: terminal reward value
    """
    reward = 0.0

    if final_status == 'Finish':
        # Large bonus for completing the full race (2 laps)
        reward += REWARD_WEIGHTS['finish_bonus']
        print(f"[Reward] Finish! +{REWARD_WEIGHTS['finish_bonus']}")

        # Additional efficiency bonus for finishing with battery remaining
        if final_soc is not None and final_soc > 0:
            soc_bonus = final_soc * 50.0  # Up to +50 extra for full battery finish
            reward += soc_bonus
            print(f"[Reward] SOC efficiency bonus: +{soc_bonus:.1f}")

    elif final_status == 'Lap1':
        # Partial bonus for completing lap 1 (halfway point)
        reward += REWARD_WEIGHTS['lap_complete']
        print(f"[Reward] Lap 1 completed! +{REWARD_WEIGHTS['lap_complete']}")

    elif final_status == 'Fallen':
        # Penalty for falling off the track (severe failure)
        reward += REWARD_WEIGHTS['fallen_penalty']
        print(f"[Reward] Fallen! {REWARD_WEIGHTS['fallen_penalty']}")

    # 'Force end' (user pressed 'q'): neutral - no bonus or penalty

    return reward


def reset_state():
    """
    Reset internal state for a new episode.
    Must be called at the start of each episode to reset action smoothness tracking.
    """
    global _prev_action
    _prev_action = {'drive': 0.0, 'steer': 0.0}


def get_reward_info():
    """Return current reward weight configuration (for inspection/logging)."""
    return REWARD_WEIGHTS.copy()


def set_reward_weight(key, value):
    """
    Update a single reward weight at runtime.
    Useful for experiments without restarting the training session.
    """
    if key in REWARD_WEIGHTS:
        REWARD_WEIGHTS[key] = value
        print(f"[Reward] Updated {key} = {value}")
    else:
        print(f"[Reward] Unknown weight key: {key}")


# ==============================================================================
# REWARD DESIGN TIPS
# ==============================================================================
#
# Starting principles:
#   - Start simple (survival + finish), add complexity gradually
#   - Terminal rewards (finish, fallen) should be 100-1000x larger than step rewards
#   - Normalize step rewards to similar scales before summing
#
# Common issues and fixes:
#   - Agent drives in circles: Add penalty for low speed or not making forward progress
#   - Agent too conservative: Reduce penalties or add a forward speed bonus
#   - Agent crashes often: Increase fallen_penalty, add steering smoothness reward
#   - Agent ignores battery: Increase soc_low_penalty or soc_efficiency weight
#
# Future improvements (requires Unity real-time telemetry):
#   - Track progress reward: position along centerline (requires pos_x, pos_z)
#   - Speed reward: proportional to forward velocity
#   - Lane following reward: based on lateral offset from centerline
#
# ==============================================================================
