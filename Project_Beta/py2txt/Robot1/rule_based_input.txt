# rule_based_input.py (Robot1 version - New Architecture)
# ==============================================================================
# Rule-Based Autonomous Driving Controller
# ==============================================================================
#
# PURPOSE:
#   Drives Robot 1 autonomously using a classical computer-vision + control pipeline:
#     1. Start signal detection   - Waits for F1-style red-lamp GO signal
#     2. Lane perception          - Sliding windows on white lane markings
#     3. Driver model decision    - Converts (lateral, theta) to (drive, steer)
#     4. Debug overlay saving     - Annotated images for analysis
#
# CONTROL PIPELINE (per update cycle at ~20 Hz):
#   PIL image -> sliding_windows_white()
#              -> SWResult(lateral_px, theta_deg)
#              -> DriverModel.update()
#              -> (driveTorque, steerAngle)
#              -> WebSocket -> Unity
#
# LANE-LOSS HANDLING:
#   hold   - Decays last known drive/steer for up to HOLD_SEC (2.0s = 40 frames)
#   search - Pivots in place with constant steer after hold expires
#
# DRIVER CONFIG (key parameters):
#   v_min=0.15, v_max=0.75  - Speed range
#   k_theta=0.90            - Gain for heading angle error
#   k_lateral=0.60          - Gain for lateral offset error
#   steer_limit=0.524 rad   - Max steer (~30 deg, matches Unity)
#   alpha_smooth=0.30       - IIR filter coefficient (lower = more smoothing)
#
# Compatible with Unity Server + Python Client architecture
# ==============================================================================

import time
import os
import sys
import importlib.util
from pathlib import Path
from PIL import Image

# Module identification (used in log messages)
MODULE_SOURCE = "Robot1"
print(f"[rule_based_input] Loaded from {MODULE_SOURCE}/")

# Add Project_Beta (parent) to sys.path for data_manager import
# Add Robot1 directory to sys.path for rule_based_algorithms package import
_this_dir = Path(__file__).parent
sys.path.insert(0, str(_this_dir.parent))  # Add Project_Beta to path
sys.path.insert(0, str(_this_dir))  # Add Robot1 to path (for rule_based_algorithms)
import data_manager

# Import rule-based algorithm modules from rule_based_algorithms/ package
from rule_based_algorithms import status_Robot              # Global robot state machine
from rule_based_algorithms.sliding_windows import sliding_windows_white  # Lane detector
from rule_based_algorithms.driver_model import DriverModel, DriverConfig  # Control logic

# ---- Load perception_Startsignal via importlib ----
# importlib ensures a unique module instance per robot (Robot1 vs Robot2 can run
# simultaneously without sharing the ready_to_go state variable)
_startsignal_path = _this_dir / "rule_based_algorithms" / "perception_Startsignal.py"
_startsignal_spec = importlib.util.spec_from_file_location(
    f"perception_Startsignal_{MODULE_SOURCE}",  # Unique module name: avoids Robot2 clash
    _startsignal_path
)
_startsignal_module = importlib.util.module_from_spec(_startsignal_spec)
_startsignal_spec.loader.exec_module(_startsignal_module)
detect_start_signal = _startsignal_module.detect_start_signal  # Bound function ref

# Debug visualization utilities
from rule_based_algorithms.debug_utils import annotate_and_save_canvas
from rule_based_algorithms.debug_utils import overlay_and_save as overlay_and_save_fallback

# ---- Global control output ----
# Written by update(), read by get_latest_command() and main.py
robot_id = "R1"
driveTorque: float = 0.0
steerAngle: float = 0.0

# ---- Internal state ----
_started_latch = False  # Once GO is detected, this stays True for the session
SAVE_DEBUG_OVERLAYS = True  # Save annotated debug images every frame (to Robot1/debug/)

# Lane-loss timing: hold for 2 seconds (40 frames at 20Hz), then switch to search
_lost_age = 0
HOLD_SEC = 2.0
PERIOD = 0.050          # 20Hz = 50ms per frame
HOLD_FRAMES = max(1, int(HOLD_SEC / PERIOD))  # = 40 frames

# Human-readable mode labels for logging
MODE_LABELS = {
    "normal": "Normal",
    "hold": "Hold",
    "search": "Search",
}

# ---- DriverModel instantiation ----
# DriverModel converts lane perception outputs to motor commands
# These gains are tuned for Robot1's physical characteristics
_driver = DriverModel(DriverConfig(
    image_width=224,
    forward_sign=+1,        # +1 = forward is positive torque
    v_min=0.15,             # Minimum speed even in tight corners
    v_max=0.75,             # Maximum speed on straights
    k_theta=0.90,           # Heading angle correction gain
    k_lateral=0.60,         # Lateral offset correction gain
    steer_limit=0.524,      # Max steering angle in radians (~30 deg, matches Unity)
    alpha_smooth=0.30,      # IIR smoothing: output = 0.7*new + 0.3*prev
    torque_limit=1.00,
    theta_hard_limit_deg=80.0,   # Reject lane detections with |theta| > 80 deg
    use_soc_scaling=True,        # Reduce speed proportionally when battery is low
    soc_floor=0.30,              # Minimum speed scale factor (at SOC=0)
))


def saturate(value, min_val=-1.0, max_val=1.0):
    """Clamp value within [min_val, max_val]."""
    return max(min_val, min(max_val, value))


def get_latest_command():
    """Return latest control command for WebSocket sender."""
    return {
        "type": "control",
        "robot_id": robot_id,
        "driveTorque": round(float(driveTorque), 3),
        "steerAngle": round(float(steerAngle), 3),
    }


def update():
    """
    Single update cycle for rule-based control. Called ~20 Hz by main.py.

    Pipeline:
      1. Read image + SOC from data_manager (double-buffer file system)
      2. Start signal detection (latched after first GO)
      3. Sliding window lane detection (only after GO)
      4. Determine lane_mode: normal / hold / search
      5. DriverModel produces (drive, steer)
      6. Saturate outputs and store in globals
      7. Optionally save debug overlay image
      8. Log periodically

    Returns True to continue, True on error (keeps running).
    """
    global driveTorque, steerAngle, _started_latch, _lost_age

    try:
        # === Step 1: Inputs ===
        soc = data_manager.get_latest_soc(robot_id)  # float 0.0..1.0 (can be None)
        rgb_path = data_manager.get_latest_rgb_path(robot_id)  # Path to latest JPEG

        if not rgb_path or not rgb_path.exists():
            # Unity hasn't sent the first frame yet; wait silently
            return True

        try:
            pil_img = Image.open(rgb_path).convert("RGB")
        except Exception:
            return True  # Frame may be partially written; skip

        # === Step 2: Start signal detection (latched) ===
        # detect_start_signal() returns True exactly ONCE when all red lamps turn OFF
        # After that, _started_latch stays True for the rest of the session
        if not _started_latch:
            raw_go = detect_start_signal(pil_img)
            if raw_go:
                _started_latch = True
                status_Robot.set_state(status_Robot.RUN_STRAIGHT)
                print(f"[{robot_id} RuleBased] START latched")

        start_go = _started_latch  # DriverModel uses this as its start gate

        # === Step 3: Lane perception (only after GO) ===
        if start_go:
            try:
                # return_canvas=True: returns the annotated BGR debug image in sw.canvas_bgr
                sw = sliding_windows_white(
                    pil_img, save_debug=False, src_path=str(rgb_path), return_canvas=True
                )
            except TypeError:
                # Backward compatibility: older version without return_canvas parameter
                sw = sliding_windows_white(pil_img, save_debug=False, src_path=str(rgb_path))

            lane_ok = bool(getattr(sw, "ok", False))        # Both lanes detected and valid
            lateral = getattr(sw, "lateral_px", None) if lane_ok else None  # px from center
            theta = getattr(sw, "theta_deg", None) if lane_ok else None     # heading deg
            img_w = getattr(sw, "img_width", None) or pil_img.size[0]
            single_side = bool(getattr(sw, "single_side", False))  # One-lane fallback active
        else:
            # Before start: no lane detection needed
            lane_ok = False
            lateral, theta = None, None
            img_w = pil_img.size[0]
            sw = None
            single_side = False

        # === Step 4: Determine lane_mode from detection result ===
        # normal  -> lane is currently visible (both or single-side)
        # hold    -> lane just lost, maintain last heading for HOLD_FRAMES
        # search  -> hold expired, pivot in place to find lane
        if lane_ok:
            _lost_age = 0
            lane_mode = "normal"
        else:
            _lost_age += 1
            lane_mode = "hold" if _lost_age <= HOLD_FRAMES else "search"

        # === Step 5: Driver model produces (drive, steer) ===
        drive, steer = _driver.update(
            lateral_px=lateral,
            theta_deg=theta,
            soc=soc,
            image_width=img_w,
            start_go=start_go,
            valid_lane=lane_ok,
            lane_mode=lane_mode,
            lost_age=_lost_age,
            single_side=single_side,
        )

        # === Step 6: Hard clamp and store outputs ===
        driveTorque = saturate(drive)
        steerAngle = saturate(steer, -0.524, 0.524)  # Limit steer to ~±30 deg

        # === Step 7: Debug overlay save ===
        if SAVE_DEBUG_OVERLAYS and start_go:
            try:
                frame_name = data_manager.get_latest_frame_name(robot_id)
                debug_dir = Path(__file__).parent / "debug"

                # Determine display mode string (shows Pulse state if active)
                debug_use_pulse = _driver.last_debug.get("use_pulse", False)
                debug_pulse_phase = _driver.last_debug.get("pulse_phase", "")
                if debug_use_pulse:
                    display_mode = f"Pulse({debug_pulse_phase})"
                else:
                    display_mode = MODE_LABELS.get((lane_mode or "").lower(), lane_mode)

                if sw is not None and getattr(sw, "canvas_bgr", None) is not None:
                    # Primary path: annotate the sliding-window debug canvas (has lane lines drawn)
                    outp = annotate_and_save_canvas(
                        sw.canvas_bgr,
                        out_dir=str(debug_dir),
                        lateral_px=lateral,
                        theta_deg=theta,
                        drive_torque=driveTorque,
                        steer_angle=steerAngle,
                        mode=display_mode,
                        frame_name=frame_name,
                        src_path=str(rgb_path),
                        jpeg_quality=85,
                    )
                else:
                    # Fallback: annotate raw frame with numeric HUD only
                    outp = overlay_and_save_fallback(
                        pil_img,
                        sw_result=sw,
                        driver_debug=_driver.last_debug,
                        out_dir=str(debug_dir),
                    )

                if outp and (_lost_age % 20 == 0 or _lost_age == 1):
                    print(f"[{robot_id} RuleBased] Saved debug: {outp}")
            except Exception as e:
                if _lost_age % 20 == 0:
                    print(f"[{robot_id} RuleBased] Debug overlay save failed: {e}")

        # === Step 8: Periodic logging ===
        if _lost_age % 20 == 0 or not start_go:
            mode_label = MODE_LABELS.get((lane_mode or "").lower(), str(lane_mode))
            use_pulse = _driver.last_debug.get("use_pulse", False)
            pulse_phase = _driver.last_debug.get("pulse_phase", "")
            if use_pulse:
                mode_label = f"Pulse({pulse_phase})"
            lat_str = "None" if (lateral is None) else f"{lateral:+.1f}"
            tht_str = "None" if (theta is None) else f"{theta:+.1f}"
            soc_str = "None" if (soc is None) else f"{float(soc):.2f}"
            import math
            steer_deg = math.degrees(steerAngle)

            print(
                f"[{robot_id} RuleBased] Drive={driveTorque:+.2f} Steer={steerAngle:+.3f}rad({steer_deg:+.1f}°) | "
                f"GO={start_go} LaneOK={lane_ok}, {mode_label} LostAge={_lost_age} "
                f"Lat={lat_str} Theta={tht_str} SOC={soc_str}"
            )

        return True

    except Exception as e:
        print(f"[{robot_id} RuleBased] Error: {e}")
        return True


def reset():
    """Reset rule-based control state (call between sessions or mode switches)."""
    global _started_latch, _lost_age, driveTorque, steerAngle
    _started_latch = False
    _lost_age = 0
    driveTorque = 0.0
    steerAngle = 0.0
    print(f"[{robot_id} RuleBased] State reset")
