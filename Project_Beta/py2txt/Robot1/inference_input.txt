# inference_input.py (Robot1 version - Updated for CNN Model)
# ==============================================================================
# AI Inference Engine - The Driver's Brain
# ==============================================================================
#
# PURPOSE:
#   Core AI control loop for Robot 1. Reads the camera image from disk, runs the
#   trained DrivingNetwork (CNN + MLP) to produce (drive_torque, steer_angle), and
#   applies strategy hooks from ai_control_strategy.py before sending to Unity.
#
# KEY DESIGN PATTERNS:
#   - importlib loading: All sub-modules (model, ai_control_strategy,
#     perception_Startsignal) are loaded via importlib.util to avoid Python's
#     module cache mixing Robot1 and Robot2 state when both robots run concurrently.
#   - Separation of concerns: This file = framework (do not edit).
#     ai_control_strategy.py = user-editable strategy hooks.
#   - CUDA warmup: preload_model() + warmup_cuda() must be called BEFORE the race
#     starts to eliminate the 10+ second GPU initialization delay on first inference.
#   - Update loop: update() is called at ~20 Hz by main.py via asyncio.sleep(0.05).
#
# IMPORTANT: This file is the FRAMEWORK - don't modify it directly.
# To customize AI behavior, edit: ai_control_strategy.py
#
# ==============================================================================

import os
import sys
from pathlib import Path

import torch
from torchvision import transforms
from PIL import Image

# Module identification (used in log messages to distinguish Robot1 vs Robot2)
MODULE_SOURCE = "Robot1"
print(f"[inference_input] Loaded from {MODULE_SOURCE}/")

# Add Project_Beta (parent of Robot1/) to sys.path so data_manager can be imported
sys.path.insert(0, str(Path(__file__).parent.parent))  # Add Project_Beta to path
import data_manager

# Get the directory of this file (Robot1/)
_this_dir = Path(__file__).parent

# ---- Load model.py via importlib ----
# importlib.util avoids module-name collisions between Robot1.model and Robot2.model.
# Without this, if Robot1 is imported first, Robot2 would reuse Robot1's cached model.
import importlib.util
_model_spec = importlib.util.spec_from_file_location(
    f"{MODULE_SOURCE}.model",
    _this_dir / "model.py"
)
_model_module = importlib.util.module_from_spec(_model_spec)
_model_spec.loader.exec_module(_model_module)
DrivingNetwork = _model_module.DrivingNetwork  # CNN + MLP network class

# ---- Load ai_control_strategy.py via importlib ----
# This file contains user-customizable logic: when to wait, how to adjust AI output.
# Falls back to default implementations if the file is missing or fails to load.
try:
    _strategy_spec = importlib.util.spec_from_file_location(
        f"{MODULE_SOURCE}.ai_control_strategy",
        _this_dir / "ai_control_strategy.py"
    )
    _strategy_module = importlib.util.module_from_spec(_strategy_spec)
    _strategy_spec.loader.exec_module(_strategy_module)
    should_wait_for_start = _strategy_module.should_wait_for_start  # Wait-gate function
    adjust_output = _strategy_module.adjust_output                  # Post-process AI output
    on_race_start = _strategy_module.on_race_start                  # Race start callback
    STRATEGY = _strategy_module.STRATEGY                            # "hybrid" or "pure_e2e"
    print(f"[inference_input] Strategy loaded: {STRATEGY}")
except Exception as e:
    print(f"[inference_input] WARNING: ai_control_strategy.py not found, using defaults: {e}")
    # Default fallback: hybrid mode using rule-based start detection
    STRATEGY = "hybrid"
    def should_wait_for_start(pil_img, race_started):
        # Load perception_Startsignal fresh via importlib (avoids cache clash)
        if not race_started:
            _percept_spec = importlib.util.spec_from_file_location(
                f"{MODULE_SOURCE}.perception_Startsignal",
                _this_dir / "rule_based_algorithms" / "perception_Startsignal.py"
            )
            _percept_module = importlib.util.module_from_spec(_percept_spec)
            _percept_spec.loader.exec_module(_percept_module)
            return not _percept_module.detect_start_signal(pil_img)
        return False
    def adjust_output(drive, steer, pil_img, soc, race_started=False):
        return drive, steer  # No adjustment in fallback mode
    def on_race_start():
        pass

# Global control values (read by get_latest_command(), written by update())
robot_id = "R1"
driveTorque = 0.0
steerAngle = 0.0

# Model state (loaded lazily on first update(), or eagerly via preload_model())
_model = None
_transform = None
_model_loaded = False
_device = None

# Race state (latches True once GO is detected; resets via reset())
_race_started = False


def get_latest_command():
    """Return the latest control command in the expected format."""
    return {
        "type": "control",
        "robot_id": robot_id,
        "driveTorque": round(float(driveTorque), 3),
        "steerAngle": round(float(steerAngle), 3),
    }


def preload_model():
    """
    Preload the AI model before the control loop starts.

    IMPORTANT: Call this BEFORE the race starts (during startup) to avoid model
    loading delays during the start signal sequence. Loading a model for the first
    time typically takes several seconds.

    Pair with warmup_cuda() for full initialization.
    """
    print(f"[{robot_id} Inference] Preloading AI model...")
    _load_model()
    if _model is not None:
        print(f"[{robot_id} Inference] Model preloaded successfully!")
    else:
        print(f"[{robot_id} Inference] WARNING: Model preload failed (will use dummy output)")


def warmup_cuda():
    """
    Warm up CUDA context with a dummy inference run.

    WHY: The very first call to a CUDA model takes 10+ seconds because PyTorch
    initializes GPU memory allocators and JIT-compiles kernels on demand.
    Running one dummy forward pass before the race eliminates this delay.

    CRITICAL: Must be called AFTER preload_model() and BEFORE race start.
    """
    global _model, _transform, _device

    if _model is None:
        print(f"[{robot_id} Inference] CUDA warmup skipped: Model not loaded")
        return

    if _device is None or _device.type != 'cuda':
        print(f"[{robot_id} Inference] CUDA warmup skipped: Using CPU")
        return

    print(f"[{robot_id} Inference] Warming up CUDA context...")

    try:
        # Create dummy input tensors matching real inference shape
        dummy_img = Image.new('RGB', (640, 480), color='black')
        dummy_tensor = _transform(dummy_img).unsqueeze(0).to(_device)  # [1, 3, 224, 224]
        dummy_soc = torch.tensor([[1.0]], dtype=torch.float32).to(_device)  # [1, 1]

        # Run one forward pass to initialize all CUDA kernels
        with torch.no_grad():
            _ = _model(dummy_tensor, dummy_soc)

        print(f"[{robot_id} Inference] CUDA warmup complete! Ready for race.")

    except Exception as e:
        print(f"[{robot_id} Inference] CUDA warmup failed: {e}")
        import traceback
        traceback.print_exc()


def saturate(value, min_val=-1.0, max_val=1.0):
    """Clamp the input value within the specified range."""
    return max(min_val, min(max_val, value))


def _load_model():
    """
    Load the trained model weights from Robot1/models/model.pth.
    Called once lazily on first update(), or eagerly via preload_model().
    Also sets up the image preprocessing transform (must match training pipeline).
    """
    global _model, _transform, _model_loaded, _device

    if _model_loaded:
        return  # Already loaded (idempotent)

    _model_loaded = True  # Mark loaded immediately to prevent re-entry

    # Auto-detect GPU; fall back to CPU
    _device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    print(f"[{robot_id} Inference] Using device: {_device}")

    model_path = Path(__file__).parent / "models" / "model.pth"

    model = DrivingNetwork()

    try:
        model.load_state_dict(torch.load(str(model_path), map_location=_device))
        model.to(_device)
        model.eval()  # Switch to inference mode (disables dropout, etc.)
        print(f"[{robot_id} Inference] Model loaded from {model_path}")
        _model = model
    except FileNotFoundError:
        print(f"[{robot_id} Inference] WARNING: Model not found at {model_path}")
        print(f"[{robot_id} Inference] Using dummy predictions (drive=0.0, steer=0.0)")
        _model = None
    except Exception as e:
        print(f"[{robot_id} Inference] WARNING: Failed to load model: {e}")
        _model = None

    # Image preprocessing transform - MUST match the transforms used during training
    # ImageNet mean/std normalization is standard for models pretrained on ImageNet
    _transform = transforms.Compose([
        transforms.Resize((224, 224)),      # Resize to model input size
        transforms.ToTensor(),              # [0,255] HWC -> [0,1] CHW
        transforms.Normalize(
            mean=[0.485, 0.456, 0.406],     # ImageNet mean (RGB)
            std=[0.229, 0.224, 0.225]       # ImageNet std (RGB)
        )
    ])


def update():
    """
    Single update cycle for AI inference control. Called ~20 Hz by main.py.

    Control flow:
      1. Load model on first call (lazy init)
      2. Read latest camera image and SOC from data_manager
      3. Delegate to should_wait_for_start() (ai_control_strategy.py)
         -> If waiting: output zeros, return
      4. Latch race start (once) and call on_race_start()
      5. Run CNN forward pass -> (raw_drive, raw_steer)
      6. Delegate to adjust_output() for post-processing (smoothing, limits, etc.)
      7. Store results in global driveTorque / steerAngle
      8. Log every 20 frames (~1 second)

    Returns True to continue, True on error (keeps running), False to stop loop.
    """
    global driveTorque, steerAngle, _race_started

    try:
        # Lazy model load (no-op if already loaded)
        _load_model()

        # --- Read inputs from data_manager ---
        soc = data_manager.get_latest_soc(robot_id)  # float 0.0..1.0, can be None
        rgb_path = data_manager.get_latest_rgb_path(robot_id)  # Path to latest JPEG

        if not rgb_path or not rgb_path.exists():
            # Unity hasn't sent the first frame yet; wait silently
            return True

        try:
            pil_img = Image.open(rgb_path).convert("RGB")
        except Exception:
            return True  # Frame may be partially written; skip this cycle

        if soc is None:
            soc = 1.0  # Default SOC if Unity hasn't sent it yet

        # --- Wait gate (delegated to ai_control_strategy.py) ---
        # Hybrid mode: wait until red start lamps go off
        # Pure E2E mode: always False (model decides timing)
        if should_wait_for_start(pil_img, _race_started):
            driveTorque = 0.0
            steerAngle = 0.0

            # Log waiting status every 2 seconds (40 frames)
            if not hasattr(update, '_frame_count'):
                update._frame_count = 0
            update._frame_count += 1
            if update._frame_count % 40 == 0:
                print(f"[{robot_id} Inference] Waiting for start signal... (Strategy: {STRATEGY})")
            return True

        # --- Race start latch ---
        # Once should_wait_for_start() returns False, latch race_started = True
        if not _race_started:
            _race_started = True
            print(f"[{robot_id} Inference] RACE STARTED! (Strategy: {STRATEGY})")
            on_race_start()  # Notify strategy (e.g., start boost countdown)

        # --- AI Inference ---
        if _model is not None:
            # Build input tensors: image [1,3,224,224] + SOC [1,1]
            image_tensor = _transform(pil_img).unsqueeze(0).to(_device)
            soc_tensor = torch.tensor([[soc]], dtype=torch.float32).to(_device)

            with torch.no_grad():
                output = _model(image_tensor, soc_tensor)  # [1, 2]
                raw_drive = output[0, 0].item()  # Index 0: drive_torque
                raw_steer = output[0, 1].item()  # Index 1: steer_angle (rad)

            # Hard clamp to physical limits
            raw_drive = saturate(raw_drive, -1.0, 1.0)
            raw_steer = saturate(raw_steer, -0.524, 0.524)  # ~±30 deg (matches Unity)

            # --- Post-process via strategy (smoothing, boost, corner cap, etc.) ---
            # Pass race_started=True so Start Boost knows the race has begun
            driveTorque, steerAngle = adjust_output(raw_drive, raw_steer, pil_img, soc, race_started=_race_started)
        else:
            # No model loaded: output zeros (safe fallback)
            driveTorque = 0.0
            steerAngle = 0.0

        # --- Periodic logging (~1 second interval) ---
        if not hasattr(update, '_frame_count'):
            update._frame_count = 0
        update._frame_count += 1

        if update._frame_count % 20 == 0:
            import math
            steer_deg = math.degrees(steerAngle)
            soc_str = f"{float(soc):.2f}"
            print(
                f"[{robot_id} Inference] Drive={driveTorque:+.3f}, "
                f"Steer={steerAngle:+.3f}rad({steer_deg:+.1f}deg), "
                f"SOC={soc_str}"
            )

        return True

    except Exception as e:
        print(f"[{robot_id} Inference] Error: {e}")
        import traceback
        traceback.print_exc()
        return True  # Keep running on error


def reset():
    """
    Reset all AI inference state.
    Called when switching modes or restarting a race session.
    Clears model references so the next call to update() triggers a fresh load.
    """
    global driveTorque, steerAngle, _model, _transform, _model_loaded, _device, _race_started
    driveTorque = 0.0
    steerAngle = 0.0
    _model = None
    _transform = None
    _model_loaded = False
    _device = None
    _race_started = False
    if hasattr(update, '_frame_count'):
        update._frame_count = 0
    # Reset start signal detector state (for hybrid mode)
    try:
        from rule_based_algorithms.perception_Startsignal import detect_start_signal
        if hasattr(detect_start_signal, 'ready_to_go'):
            detect_start_signal.ready_to_go = False
    except ImportError:
        pass
    print(f"[{robot_id} Inference] State reset")
