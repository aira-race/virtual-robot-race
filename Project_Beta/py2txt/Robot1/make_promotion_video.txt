# make_promotion_video.py
# Generate promotion video showing perception algorithms in action
# - StartSequence frames: Show perception_Startsignal ROI and detection
# - Lap0+ frames: Show sliding_windows lane detection (simplified for PIL-only)
#
# PURPOSE:
#   Creates an annotated MP4 video from a training_data run directory to visually
#   demonstrate how the rule-based algorithms see and process each frame.
#   Two overlay types are applied depending on the race phase (status column in metadata.csv):
#     - StartSequence: perception_Startsignal ROI boxes + red pixel ratio readouts
#     - Lap0/Lap1/Lap2/Finish: sliding windows lane detection visualization
#
# ALGORITHM OVERVIEW:
#   1. Load metadata.csv to map frame filenames → race status
#   2. For each frame image (frame_*.jpg):
#       a. If status == "StartSequence": call draw_startsignal_overlay()
#       b. If status in Lap0/Lap1/Lap2/Finish: call draw_slidingwindows_overlay()
#   3. Save annotated frames to <run-dir>/promotion/
#   4. Encode frames to MP4 using imageio (libx264, yuv420p)
#
# NOTE ON PIL-ONLY SLIDING WINDOWS:
#   draw_slidingwindows_overlay() uses PIL + numpy only (no OpenCV).
#   The parameters here may differ slightly from sliding_windows.py (which uses OpenCV).
#   This is intentional: the goal is visualization, not exact replication of the algorithm.
#
# Usage:
#   cd Robot1
#   python make_promotion_video.py --run-dir training_data/run_20251207_092050

import numpy as np

# Import is_red() from the actual startsignal module for consistency
from Robot1.rule_based_algorithms.perception_Startsignal import is_red


# ===== Sliding Windows Parameters (matches sliding_windows.py but simplified for PIL) =====
# ROI slightly different from main algorithm (0.40 vs 0.35) for cleaner visualization
ROI_TOP_FRAC = 0.40
ROI_BOT_FRAC = 0.70
NWINDOWS = 9
MARGIN = 60
MINPIX = 50


def draw_startsignal_overlay(img: Image.Image) -> Image.Image:
    """
    Draw perception_Startsignal ROI and detection visualization.
    Shows 3 lamp regions and red pixel detection status.

    Lamp regions (same as perception_Startsignal.py):
      - Lamp 1: 35%~50% of image width
      - Lamp 2: 55%~70% of image width
      - Lamp 3: 75%~90% of image width
    All within top 20% of image height (matches actual detection ROI).

    Color coding:
      - Green outline = lamp is OFF (lamp not lit)
      - Red outline   = lamp is ON (red pixels detected)
    """
    draw = ImageDraw.Draw(img)
    width, height = img.size

    # ROI: top 20% of image (matches perception_Startsignal.py top 30%)
    top = 0
    bottom = int(height * 0.2)

    # Lamp horizontal positions (same as perception_Startsignal.py)
    lamp_positions = [
        (int(width * 0.35), int(width * 0.5)),
        (int(width * 0.55), int(width * 0.7)),
        (int(width * 0.75), int(width * 0.9))
    ]

    lamp_status = []

    for i, (left, right) in enumerate(lamp_positions):
        # Count red pixels in this lamp's region
        red_pixels = 0
        total_pixels = 0
        for y in range(top, bottom):
            for x in range(left, right):
                pixel = img.getpixel((x, y))
                if is_red(pixel):
                    red_pixels += 1
                total_pixels += 1

        ratio = red_pixels / total_pixels if total_pixels > 0 else 0
        is_lamp_on = ratio > 0.03  # Same threshold as perception_Startsignal.py
        lamp_status.append(is_lamp_on)

        # Draw lamp ROI rectangle
        box_color = "red" if is_lamp_on else "lime"
        draw.rectangle([left, top, right, bottom], outline=box_color, width=3)

        # Draw lamp label below the box
        label = f"LAMP{i+1}: {'ON' if is_lamp_on else 'OFF'}"
        label_y = bottom + 5
        draw.text((left, label_y), label, fill=box_color)

    # Draw overall status at bottom of image
    if all(lamp_status):
        status_text = "STATUS: ALL LAMPS ON - READY"
        status_color = "red"
    elif not any(lamp_status):
        status_text = "STATUS: ALL LAMPS OFF - GO!"
        status_color = "lime"
    else:
        status_text = f"STATUS: {sum(lamp_status)}/3 LAMPS ON"
        status_color = "yellow"

    draw.text((10, height - 30), status_text, fill=status_color)

    # Draw algorithm name
    draw.text((10, 10), "perception_Startsignal", fill="cyan")

    return img


def white_binary_pil(img: Image.Image, roi_top: int, roi_bot: int) -> np.ndarray:
    """
    Create white lane binary mask using PIL (HSV-like thresholding without OpenCV).
    White = high brightness, low saturation.

    Thresholds:
      brightness > 190  (average of R, G, B)
      saturation < 60   (max(R,G,B) - min(R,G,B))

    Returns uint8 numpy array where 255 = white pixel, 0 = non-white.
    """
    width, height = img.size
    roi = img.crop((0, roi_top, width, roi_bot))
    roi_np = np.array(roi)

    r, g, b = roi_np[:, :, 0], roi_np[:, :, 1], roi_np[:, :, 2]

    # White: all channels high and similar to each other (low saturation)
    brightness = (r.astype(int) + g.astype(int) + b.astype(int)) / 3
    saturation = np.maximum(np.maximum(r, g), b) - np.minimum(np.minimum(r, g), b)

    mask = (brightness > 190) & (saturation < 60)
    return mask.astype(np.uint8) * 255


def draw_slidingwindows_overlay(img: Image.Image) -> Image.Image:
    """
    Draw sliding windows lane detection visualization using PIL only (no OpenCV).

    Visualization elements:
      - Red rectangle = ROI boundary
      - Green center line = image center (reference for lateral offset)
      - Cyan rectangles = left-side sliding windows
      - Dark cyan rectangles = right-side sliding windows
      - Orange dots = detected lane center points per window
      - Yellow dots = lane centerline (midpoint between left and right)
      - Bottom text = lateral offset in pixels

    NOTE: This is a simplified PIL-only implementation for visualization.
    The actual algorithm (sliding_windows.py) uses OpenCV for efficiency.
    """
    draw = ImageDraw.Draw(img)
    width, height = img.size

    y_top = int(height * ROI_TOP_FRAC)
    y_bot = int(height * ROI_BOT_FRAC)
    cx = width // 2

    # Draw ROI boundary (red) and image center line (green)
    draw.rectangle([0, y_top, width - 1, y_bot], outline="red", width=2)
    draw.line([(cx, y_top), (cx, y_bot)], fill="lime", width=2)

    # Create white binary mask for the ROI
    binary = white_binary_pil(img, y_top, y_bot)
    roi_h = y_bot - y_top

    # Histogram-based initial position: find peak white pixel density in lower half
    half = binary[roi_h // 2:, :]
    hist = np.sum(half, axis=0)
    midx = width // 2

    leftx_base = np.argmax(hist[:midx]) if np.any(hist[:midx]) else midx // 2
    rightx_base = (np.argmax(hist[midx:]) + midx) if np.any(hist[midx:]) else (midx + (width - midx) // 2)

    win_h = roi_h // NWINDOWS
    nonzero = binary.nonzero()
    if len(nonzero[0]) == 0:
        # No white pixels detected - show "no detection" text
        draw.text((10, 10), "sliding_windows (no detection)", fill="red")
        return img

    nonzero_y, nonzero_x = nonzero
    leftx_current, rightx_current = int(leftx_base), int(rightx_base)

    left_pts = []
    right_pts = []

    # Slide windows from bottom to top
    for win in range(NWINDOWS):
        win_y_low = y_bot - (win + 1) * win_h
        win_y_high = y_bot - win * win_h
        ry_low = win_y_low - y_top   # coordinates within ROI (for binary mask indexing)
        ry_high = win_y_high - y_top

        lx_low, lx_high = leftx_current - MARGIN, leftx_current + MARGIN
        rx_low, rx_high = rightx_current - MARGIN, rightx_current + MARGIN

        # Draw window rectangles (cyan=left, dark cyan=right)
        draw.rectangle([lx_low, win_y_low, lx_high, win_y_high], outline="cyan", width=2)
        draw.rectangle([rx_low, win_y_low, rx_high, win_y_high], outline="#00A0A0", width=2)

        # Count white pixels within each window
        good_left = ((nonzero_y >= ry_low) & (nonzero_y < ry_high) &
                     (nonzero_x >= lx_low) & (nonzero_x < lx_high))
        good_right = ((nonzero_y >= ry_low) & (nonzero_y < ry_high) &
                      (nonzero_x >= rx_low) & (nonzero_x < rx_high))

        left_in_win = np.sum(good_left)
        right_in_win = np.sum(good_right)

        # Re-center window on mean x of detected pixels (if enough found)
        if left_in_win > MINPIX:
            leftx_current = int(np.mean(nonzero_x[good_left]))
            left_pts.append((leftx_current, (win_y_low + win_y_high) // 2))
        if right_in_win > MINPIX:
            rightx_current = int(np.mean(nonzero_x[good_right]))
            right_pts.append((rightx_current, (win_y_low + win_y_high) // 2))

    # Draw detected lane center points (orange)
    for pt in left_pts:
        draw.ellipse([pt[0] - 4, pt[1] - 4, pt[0] + 4, pt[1] + 4], fill="orange")
    for pt in right_pts:
        draw.ellipse([pt[0] - 4, pt[1] - 4, pt[0] + 4, pt[1] + 4], fill="orange")

    # Draw centerline and lateral offset if both lanes detected
    if len(left_pts) > 2 and len(right_pts) > 2:
        for lp, rp in zip(left_pts, right_pts):
            cx_lane = (lp[0] + rp[0]) // 2
            cy_lane = (lp[1] + rp[1]) // 2
            draw.ellipse([cx_lane - 3, cy_lane - 3, cx_lane + 3, cy_lane + 3], fill="yellow")

        # Compute lateral offset at midpoint
        mid_idx = len(left_pts) // 2
        if mid_idx < len(left_pts) and mid_idx < len(right_pts):
            lane_center = (left_pts[mid_idx][0] + right_pts[mid_idx][0]) // 2
            lateral = lane_center - cx
            status = f"Lateral: {lateral:+d}px"
            draw.text((10, height - 30), status, fill="yellow")

    # Draw algorithm name
    draw.text((10, 10), "sliding_windows", fill="cyan")

    return img


def load_metadata(metadata_path: str) -> dict:
    """
    Load metadata.csv and return dict mapping filename → race status.
    Used to decide which overlay to apply to each frame.
    """
    frame_status = {}
    with open(metadata_path, 'r', encoding='utf-8') as f:
        reader = csv.DictReader(f)
        for row in reader:
            frame_status[row['filename']] = row['status']
    return frame_status


def process_frames(run_dir: str, output_dir: str):
    """
    Process all frames in a run directory and save annotated versions.

    Frame routing logic:
      - "StartSequence" → draw_startsignal_overlay()
      - "Lap0", "Lap1", "Lap2", "Finish" → draw_slidingwindows_overlay()
      - No metadata entry (early frames) → draw_startsignal_overlay() as fallback
    """
    images_dir = os.path.join(run_dir, "images")
    metadata_path = os.path.join(run_dir, "metadata.csv")

    if not os.path.exists(metadata_path):
        print(f"[ERROR] metadata.csv not found: {metadata_path}")
        return False

    if not os.path.exists(images_dir):
        print(f"[ERROR] images directory not found: {images_dir}")
        return False

    os.makedirs(output_dir, exist_ok=True)

    # Load frame→status mapping
    frame_status = load_metadata(metadata_path)
    print(f"[INFO] Loaded {len(frame_status)} entries from metadata.csv")

    image_files = sorted(glob.glob(os.path.join(images_dir, "frame_*.jpg")))
    print(f"[INFO] Found {len(image_files)} image files")

    processed_count = 0
    for img_path in image_files:
        filename = os.path.basename(img_path)
        status = frame_status.get(filename, None)

        try:
            img = Image.open(img_path).convert("RGB")
        except Exception as e:
            print(f"[WARN] Failed to load {filename}: {e}")
            continue

        # Route to appropriate overlay function based on race status
        if status == "StartSequence":
            processed_img = draw_startsignal_overlay(img.copy())
        elif status in ["Lap0", "Lap1", "Lap2", "Finish"]:
            processed_img = draw_slidingwindows_overlay(img.copy())
        else:
            # No matching status (early frames before start sequence) → show startsignal
            processed_img = draw_startsignal_overlay(img.copy())

        out_path = os.path.join(output_dir, filename)
        processed_img.save(out_path, "JPEG", quality=95)
        processed_count += 1

        if processed_count % 20 == 0:
            print(f"[INFO] Processed {processed_count}/{len(image_files)} frames...")

    print(f"[INFO] Completed! Processed {processed_count} frames to {output_dir}")
    return True


def natural_key(p: str):
    """Natural sort key: sorts 'frame_10' after 'frame_9' (not alphabetically)."""
    base = os.path.basename(p)
    return [int(t) if t.isdigit() else t.lower() for t in re.findall(r'\d+|\D+', base)]


def make_video(images_dir: str, output_path: str, fps: float = 20.0):
    """
    Encode processed frames to an MP4 video using imageio.

    Codec settings:
      - libx264 (H.264) for broad compatibility
      - yuv420p pixel format (required for QuickTime/VLC playback)
      - quality=6 (lower = better quality, larger file)
    """
    import imageio

    frames = sorted(glob.glob(os.path.join(images_dir, "frame_*.jpg")), key=natural_key)
    if not frames:
        print(f"[ERROR] No frames found in {images_dir}")
        return False

    print(f"[INFO] Creating video from {len(frames)} frames at {fps} fps")

    f0 = imageio.imread(frames[0])
    h, w = f0.shape[:2]

    os.makedirs(os.path.dirname(output_path), exist_ok=True)

    writer = imageio.get_writer(
        output_path,
        format="FFMPEG",
        mode="I",
        fps=fps,
        codec="libx264",
        macro_block_size=None,
        pixelformat="yuv420p",
        quality=6
    )

    try:
        for i, p in enumerate(frames, 1):
            frame = imageio.imread(p)
            writer.append_data(frame)
            if i % 50 == 0:
                print(f"[INFO] Video: {i}/{len(frames)} frames...")
    finally:
        writer.close()

    size = os.path.getsize(output_path)
    print(f"[INFO] Video saved: {output_path} ({size/1024:.1f} KB, {w}x{h} @ {fps}fps)")
    return True


def main():
    parser = argparse.ArgumentParser(description="Generate promotion video with algorithm overlays")
    parser.add_argument("--run-dir", type=str, required=True,
                       help="Path to run directory (e.g., training_data/run_20251207_092050)")
    parser.add_argument("--output-dir", type=str, default="",
                       help="Output directory for processed frames (default: <run-dir>/promotion)")
    parser.add_argument("--fps", type=float, default=20.0,
                       help="Video frame rate (default: 20)")
    parser.add_argument("--skip-video", action="store_true",
                       help="Skip video creation, only generate frames")
    args = parser.parse_args()

    run_dir = os.path.abspath(args.run_dir)
    if args.output_dir:
        output_dir = os.path.abspath(args.output_dir)
    else:
        output_dir = os.path.join(run_dir, "promotion")

    print(f"[INFO] Run directory: {run_dir}")
    print(f"[INFO] Output directory: {output_dir}")

    # Step 1: Process all frames and apply overlays
    print("\n=== Step 1: Processing frames with algorithm overlays ===")
    if not process_frames(run_dir, output_dir):
        print("[ERROR] Frame processing failed")
        return 1

    # Step 2: Encode to MP4
    if not args.skip_video:
        print("\n=== Step 2: Creating promotion video ===")
        video_path = os.path.join(run_dir, "promotion_video.mp4")
        if not make_video(output_dir, video_path, fps=args.fps):
            print("[ERROR] Video creation failed")
            return 1
        print(f"\n[SUCCESS] Promotion video created: {video_path}")
    else:
        print("\n[INFO] Skipping video creation (--skip-video)")

    return 0


if __name__ == "__main__":
    sys.exit(main())
