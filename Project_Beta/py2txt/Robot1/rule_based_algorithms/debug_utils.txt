# rule_based_algorithms/debug_utils.py
#
# PURPOSE:
#   Utility functions for saving per-frame debug overlay images during rule-based driving.
#   Two paths for saving:
#
#   1. annotate_and_save_canvas() [PREFERRED]
#      Input: canvas_bgr (the debug image already rendered by sliding_windows.py)
#      Adds a minimal top-left HUD (mode, drive, steer) and a steer vector graphic.
#      Used when sliding_windows returns a valid canvas (return_canvas=True).
#
#   2. overlay_and_save() [FALLBACK]
#      Input: raw PIL image from camera
#      Draws a full HUD (mode, lateral, theta, drive, steer) on the raw frame.
#      Used when no SW canvas is available (SW failed or not yet called).
#
# STEER VECTOR:
#   A small compass-like graphic drawn at the top-right corner.
#   - Circle + crosshair background (light gray)
#   - Arrow: direction = steer_angle, length = abs(drive_torque) * radius
#   - Green = forward, Red = reverse
#
# OUTPUT DIRECTORY:
#   Files are saved in <out_dir>/output/ (subdirectory created automatically).
#   Filename: "debug_<source_filename>.jpg" if frame_name given,
#             otherwise timestamp + "_latest_RGB_a/b.jpg"
#
# USAGE IN rule_based_input.py:
#   if sw.canvas_bgr is not None:
#       annotate_and_save_canvas(sw.canvas_bgr, out_dir=debug_dir, ...)
#   else:
#       overlay_and_save(pil_img, sw, driver_debug, out_dir=debug_dir)

import os
import cv2
import numpy as np
from datetime import datetime
import math
from typing import Optional, Tuple

# ---------------- helpers ----------------
def ensure_dir(path: str) -> None:
    """Create directory if it doesn't exist (no-op if it does)."""
    os.makedirs(path, exist_ok=True)

def pil_to_bgr(pil_img):
    """Convert PIL.Image to OpenCV BGR ndarray."""
    return cv2.cvtColor(np.array(pil_img), cv2.COLOR_RGB2BGR)

def _put_text(img: np.ndarray, text: str, xy: Tuple[int, int], *, scale: float = 0.28, thick: int = 1) -> None:
    """
    Draw readable text with white foreground and black outline.
    Uses FONT_HERSHEY_DUPLEX (unified with sliding_windows.py HUD font).
    """
    font = cv2.FONT_HERSHEY_DUPLEX
    x, y = xy
    cv2.putText(img, text, (x, y), font, scale, (0, 0, 0), thick + 1, cv2.LINE_AA)    # black outline
    cv2.putText(img, text, (x, y), font, scale, (255, 255, 255), thick, cv2.LINE_AA)  # white text

def _resolve_outpath(out_dir: str, *, frame_name: Optional[str] = None, src_path: Optional[str] = None) -> str:
    """
    Decide the output filename for the debug image.

    Priority:
      1. If frame_name given: "debug_<basename(frame_name)>"
      2. Otherwise: timestamp + ("_a" or "_b" inferred from src_path buffer name)

    Output is placed in out_dir/output/ subdirectory.
    """
    actual_out_dir = os.path.join(out_dir, "output")
    ensure_dir(actual_out_dir)
    if frame_name:
        base = os.path.basename(frame_name)  # strip any path prefix
        out_name = f"debug_{base}"
    else:
        # Infer a/b buffer from source path (latest_RGB_a.jpg or latest_RGB_b.jpg)
        tag = "a" if (src_path and "latest_RGB_a" in src_path) else "b"
        ts  = datetime.now().strftime("%Y%m%d_%H%M%S_%f")[:-3]
        out_name = f"{ts}_latest_RGB_{tag}.jpg"
    return os.path.join(actual_out_dir, out_name)

def _draw_steer_vector(img: np.ndarray, drive_tq: float, steer_ang: float, *, radius: int = 34, pad: int = 16) -> None:
    """
    Draw a "steer control vector" arrow in the top-right corner of the image.

    Visual encoding:
      Arrow direction  = steer_ang (radians, + is right turn)
      Arrow length     = abs(drive_tq) × radius pixels (0 drive = dot, full = full radius)
      Arrow color      = green (forward drive) or red (reverse drive)
      Background       = gray circle + crosshair (reference guide)

    This gives a quick visual check of what the controller is outputting.
    """
    h, w = img.shape[:2]
    cx = w - pad - radius
    cy = pad + radius

    # Background: guide circle + crosshair in light gray
    cv2.circle(img, (cx, cy), radius, (210, 210, 210), 1, cv2.LINE_AA)
    cv2.line(img, (cx - radius, cy), (cx + radius, cy), (210, 210, 210), 1, cv2.LINE_AA)
    cv2.line(img, (cx, cy - radius), (cx, cy + radius), (210, 210, 210), 1, cv2.LINE_AA)

    drive = float(drive_tq)
    steer = float(steer_ang)
    length = min(1.0, abs(drive))  # Normalize to [0, 1]

    r_pix = int(radius * length)
    # Convert steer angle to pixel offsets (right = +x, forward = -y in image)
    dx = int(r_pix * math.sin(steer))
    dy = int(-r_pix * math.cos(steer))  # up = negative y in image coords

    # Green for forward, red for reverse
    color = (0, 255, 0) if drive >= 0 else (0, 0, 255)  # BGR

    # Draw arrow with black outline for visibility
    cv2.line(img, (cx, cy), (cx + dx, cy + dy), (0, 0, 0), 3, cv2.LINE_AA)  # black outline
    cv2.line(img, (cx, cy), (cx + dx, cy + dy), color, 2, cv2.LINE_AA)      # colored arrow

# ------------- 1) Minimal HUD on SW canvas + steer vector (preferred) -------------
def annotate_and_save_canvas(
    canvas_bgr: Optional[np.ndarray],
    *,
    out_dir: str = "debug",
    lateral_px: Optional[float] = None,    # accepted for API consistency (not drawn here, SW HUD shows it)
    theta_deg: Optional[float] = None,     # accepted but not drawn (SW HUD shows it)
    drive_torque: float = 0.0,
    steer_angle: float = 0.0,
    mode: str = "normal",
    frame_name: Optional[str] = None,
    src_path: Optional[str] = None,
    jpeg_quality: int = 85,
    origin: Tuple[int, int] = (6, 12),     # top-left HUD start position
    line: int = 12,                        # line height [px]
) -> Optional[str]:
    """
    Overlay a minimal HUD on the already-rendered SW canvas and save to debug/.
    The SW canvas already shows lateral/theta values; this adds mode + drive/steer + vector.

    Returns the output file path, or None if canvas_bgr is None.
    """
    if canvas_bgr is None:
        return None

    img = canvas_bgr.copy()

    steer_deg = math.degrees(steer_angle)
    x, y = origin

    # Top-left mini-HUD: frame name, mode, drive, steer
    if frame_name:
        display_name = os.path.basename(frame_name)
        _put_text(img, display_name, (x, y)); y += line

    _put_text(img, f"Mode: {mode}",                           (x, y)); y += line
    _put_text(img, f"Drive : {drive_torque:+.3f}",            (x, y)); y += line
    _put_text(img, f"Steer : {steer_deg:+.1f} deg", (x, y))

    # Top-right steer vector
    _draw_steer_vector(img, drive_torque, steer_angle, radius=24, pad=10)

    out_path = _resolve_outpath(out_dir, frame_name=frame_name, src_path=src_path)
    cv2.imwrite(out_path, img, [int(cv2.IMWRITE_JPEG_QUALITY), int(jpeg_quality)])
    print(f"[DebugUtils] Saved {out_path}")
    return out_path

# ----------- 2) Fallback: full HUD on raw frame + steer vector -----------
def overlay_and_save(
    pil_img,
    sw_result,
    driver_debug: Optional[dict],
    out_dir: str = "debug",
    frame_name: Optional[str] = None,
    src_path: Optional[str] = None,
    jpeg_quality: int = 85,
    origin: Tuple[int, int] = (6, 12),
    line: int = 12,
) -> str:
    """
    Draw a full HUD on the raw camera frame (no SW canvas available).
    Includes: Mode, Lateral, Theta, Drive, Steer, and the steer vector graphic.
    Used as fallback when sliding_windows() returns ok=False with no canvas.

    Returns the output file path.
    """
    ensure_dir(out_dir)
    bgr = pil_to_bgr(pil_img).copy()

    # Extract values from sw_result and driver_debug (safely handles None)
    lateral = getattr(sw_result, "lateral_px", None) if sw_result is not None else None
    theta   = getattr(sw_result, "theta_deg",  None) if sw_result is not None else None

    drive = (driver_debug or {}).get("drive_torque")
    steer = (driver_debug or {}).get("steer_angle")
    mode = (driver_debug or {}).get("lane_mode", "unknown")

    x, y = origin

    if frame_name:
        display_name = os.path.basename(frame_name)
        _put_text(bgr, display_name, (x, y)); y += line

    # Draw all available values
    _put_text(bgr, f"Mode: {mode}", (x, y)); y += line
    _put_text(bgr, f"Lateral : {'None' if lateral is None else f'{lateral:+.1f} px'}", (x, y)); y += line
    _put_text(bgr, f"Theta   : {'None' if theta   is None else f'{theta:+.1f} deg'}", (x, y)); y += line
    _put_text(bgr, f"Drive   : {'None' if drive is None else f'{drive:+.3f}'}",  (x, y)); y += line
    if steer is not None:
        steer_deg = math.degrees(steer)
        _put_text(bgr, f"Steer   : {steer_deg:+.1f} deg", (x, y))
    else:
        _put_text(bgr, f"Steer   : None", (x, y))

    # Draw steer vector if both drive and steer are available
    if drive is not None and steer is not None:
        _draw_steer_vector(bgr, float(drive), float(steer), radius=24, pad=10)

    out_path = _resolve_outpath(out_dir, frame_name=frame_name, src_path=src_path)
    cv2.imwrite(out_path, bgr, [int(cv2.IMWRITE_JPEG_QUALITY), int(jpeg_quality)])
    print(f"[DebugUtils] Saved {out_path}")
    return out_path
