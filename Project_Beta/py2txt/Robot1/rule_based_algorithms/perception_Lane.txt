# perception_Lane.py
# Red-lane perception: compute lateral deviation, heading angle, confidence.
# Debug-friendly: run on a specified JPG once and print the result.
#
# PURPOSE:
#   Detect a RED lane marking in the camera image and compute:
#     - lateral  : How far the robot is from lane center [-1 to +1, right is +]
#     - heading_deg : Lane direction relative to forward [degrees, right is +]
#     - confidence : Ratio of red pixels in ROI (0 to 1)
#
# NOTE: This module detects RED lanes. For WHITE lane detection (used in the
# main rule-based pipeline), see sliding_windows.py.
# perception_Lane.py is kept for track position estimation or alternative algorithms.
#
# DETECTION ALGORITHM:
#   1. Crop ROI to rows [ROI_TOP_FRAC, ROI_BOT_FRAC] of image (45%~90%)
#   2. Convert ROI to HSV color space
#   3. Create red mask using dual HSV range (hue 0-10 and 170-180, handles hue wrap-around)
#   4. Apply morphological open/close to reduce noise
#   5. Compute image moments to find centroid (cx) of the red region
#   6. Lateral = (cx - image_center_x) / (image_center_x), normalized to [-1, +1]
#   7. Heading angle: arctan2 from robot center (bottom-center of ROI) to centroid
#
# DUAL HSV RANGES FOR RED:
#   Red hue wraps around in HSV (0 ≡ 360 degrees).
#   Range 1: Hue 0-10  (orange-red)
#   Range 2: Hue 170-180 (magenta-red)
#   Both require Saturation > 80 and Value > 60.
#
# NOTE ON DUPLICATE detect_from_pil():
#   This file defines detect_from_pil() twice. The second definition (using fitLine)
#   replaces the first (using arctan2). Python uses the last definition. Both are
#   kept in source to document the evolution of the heading angle computation.

from dataclasses import dataclass
from PIL import Image
import cv2, numpy as np, os, argparse
import csv, glob

@dataclass
class LaneObs:
    """
    Observation from red lane detection.

    Fields:
      ok          : True if lane was detected (area >= 1000 pixels)
      lateral     : Normalized horizontal offset from image center [-1 to +1, right is +]
      heading_deg : Lane heading angle in degrees (from robot center to centroid)
      confidence  : Ratio of red pixels relative to reference area (capped at 1.0)
    """
    ok: bool
    lateral: float      # -1..1 (left negative)
    heading_deg: float  # lane direction in degrees (left negative-ish depending on camera)
    confidence: float   # 0..1 (mask area ratio)

# === HSV ranges for RED (dual range due to hue wrap-around at 0/180) ===
# Adjust these if lighting conditions change (e.g., dark environments may need lower V_MIN)
RED1_LO, RED1_HI = (0, 80, 60),   (10, 255, 255)   # Orange-red hue
RED2_LO, RED2_HI = (170, 80, 60), (180, 255, 255)  # Magenta-red hue

# === ROI vertical range (fraction of image height) ===
ROI_TOP_FRAC  = 0.45   # Ignore sky and far distance
ROI_BOT_FRAC  = 0.90   # Stop before camera hood/bumper

# === Debug output directory ===
project_root = os.path.dirname(os.path.dirname(__file__))
DEBUG_DIR = os.path.join(project_root, "debug")
os.makedirs(DEBUG_DIR, exist_ok=True)


def _make_red_mask(hsv_roi: np.ndarray) -> np.ndarray:
    """
    Create binary mask for red pixels in an HSV image.
    Combines two ranges (hue 0-10 and 170-180) to handle the circular hue wrap.
    Applies morphological operations to remove noise and fill gaps.
    """
    m1 = cv2.inRange(hsv_roi, np.array(RED1_LO), np.array(RED1_HI))
    m2 = cv2.inRange(hsv_roi, np.array(RED2_LO), np.array(RED2_HI))
    mask = cv2.bitwise_or(m1, m2)  # Combine both red hue ranges
    mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN,  np.ones((5,5), np.uint8))  # Remove noise
    mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, np.ones((5,5), np.uint8))  # Fill gaps
    return mask

def detect_from_pil(pil_img: Image.Image, save_debug: bool=False) -> LaneObs:
    """
    Compute lane observation from a PIL image.
    VERSION 1: Uses centroid-to-center arctan2 for heading angle.
    (Overridden by VERSION 2 below; kept for documentation of evolution.)
    """
    bgr = cv2.cvtColor(np.array(pil_img), cv2.COLOR_RGB2BGR)
    h, w, _ = bgr.shape
    roi_top = int(h * ROI_TOP_FRAC)
    roi_bot = int(h * ROI_BOT_FRAC)

    roi = bgr[roi_top:roi_bot, :]
    hsv = cv2.cvtColor(roi, cv2.COLOR_BGR2HSV)

    mask = _make_red_mask(hsv)
    M = cv2.moments(mask)
    area = M["m00"]

    # Confidence = ratio of red area to reference (15% of ROI area), capped at 1.0
    conf = 0.0 if area <= 0 else float(min(1.0, area / (mask.size * 0.15)))

    if area < 1000:  # Too few pixels → detection failed
        if save_debug:
            _save_debug(bgr, w, roi_top, roi_bot, None, None, mask)
        return LaneObs(False, 0.0, 0.0, conf)

    # Centroid x position → lateral offset normalized to [-1, +1]
    cx = M["m10"] / area
    lateral = (cx - (w/2)) / (w/2)

    # Centroid y position (in full image coordinates)
    pts = cv2.findNonZero(mask)
    cy_roi = float(pts[:, :, 1].mean()) if pts is not None else (roi_bot - roi_top) * 0.5
    cy = roi_top + cy_roi

    # Heading angle: from robot center (bottom-center of ROI) to centroid
    # Convention: 0° = straight forward, + = right, - = left
    x0, y0 = (w / 2.0), float(roi_bot)  # Robot center at bottom-center of ROI
    dx = cx - x0
    dy_up = (y0 - cy)  # Distance in forward (upward) direction
    heading_deg = float(np.degrees(np.arctan2(dx, dy_up + 1e-6)))

    if save_debug:
        _save_debug(
            bgr, w, roi_top, roi_bot,
            (cx, cy),
            None,
            mask,
            heading_deg
        )

    return LaneObs(True, float(lateral), heading_deg, conf)


def _save_debug(bgr, w, roi_top, roi_bot, cpt, fit, mask, angle_deg=None):
    """
    Save a debug visualization image with:
      - Red ROI box
      - Green image center line
      - Centroid marker and arrow from robot center to centroid
      - Small red mask preview (top-left corner)
      - Angle value text overlay
    Debug images are saved with incremental counter (debug_red_000001.jpg, etc.)
    """
    dbg = bgr.copy()
    # ROI box and image center reference
    cv2.rectangle(dbg, (0, roi_top), (w, roi_bot), (0, 0, 255), 2)
    cv2.line(dbg, (w//2, roi_top), (w//2, roi_bot), (0, 255, 0), 2)

    # Draw robot center → centroid arrow
    if cpt is not None:
        cx, cy = int(cpt[0]), int(cpt[1])
        x0, y0 = int(w//2), int(roi_bot)
        cv2.drawMarker(dbg, (cx, cy), (0, 0, 255), cv2.MARKER_TILTED_CROSS, 24, 2)
        cv2.arrowedLine(dbg, (x0, y0), (cx, cy), (0, 200, 0), 3, tipLength=0.15)

    # Small mask preview embedded in top-left corner
    if mask is not None:
        mcolor = cv2.cvtColor(mask, cv2.COLOR_GRAY2BGR)
        mh, mw = mask.shape
        scale = 0.35
        mrsz = cv2.resize(mcolor, (int(mw*scale), int(mh*scale)))
        h_img, w_img, _ = dbg.shape
        y0 = 10; x0 = 10
        dbg[y0:y0+mrsz.shape[0], x0:x0+mrsz.shape[1]] = mrsz

    # Angle text overlay
    if angle_deg is not None:
        txt = f"Angle = {angle_deg:+.1f} deg"
        cv2.putText(dbg, txt, (20, roi_top - 10 if roi_top > 30 else 30),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.8, (0, 0, 0), 3, cv2.LINE_AA)
        cv2.putText(dbg, txt, (20, roi_top - 10 if roi_top > 30 else 30),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.8, (255, 255, 255), 2, cv2.LINE_AA)

    # Incremental counter for unique filenames
    counter_path = os.path.join(DEBUG_DIR, "counter.txt")
    try:
        cnt = int(open(counter_path, "r").read().strip())
    except:
        cnt = 0
    cnt += 1
    with open(counter_path, "w") as f:
        f.write(str(cnt))

    out_path = os.path.join(DEBUG_DIR, f"debug_red_{cnt:06d}.jpg")
    cv2.imwrite(out_path, dbg)
    print(f"[Perception] Saved debug: {out_path}")


def _fit_line_heading_deg(pts):
    """
    Fit a line through the red lane pixel positions using OpenCV fitLine.
    Returns the line parameters (vx, vy, x0, y0) and heading angle in degrees.
    NOTE: This function is used by VERSION 2 of detect_from_pil() below.
    """
    # cv2.fitLine returns small arrays; convert to pure Python floats to avoid FutureWarnings
    vx, vy, x0, y0 = cv2.fitLine(pts, cv2.DIST_L2, 0, 0.01, 0.01)
    vx = float(vx.ravel()[0]); vy = float(vy.ravel()[0])
    x0 = float(x0.ravel()[0]); y0 = float(y0.ravel()[0])
    heading_deg = float(np.degrees(np.arctan2(vy, vx)))
    return (vx, vy, x0, y0), heading_deg

def detect_from_pil(pil_img: Image.Image, save_debug: bool=False) -> LaneObs:
    """
    Compute lane observation from a PIL image.
    VERSION 2: Uses cv2.fitLine for heading angle (overrides VERSION 1 above).
    fitLine fits a line through all red pixel positions and returns the line direction.
    This is more robust to centroid position when the lane is curved.
    """
    bgr = cv2.cvtColor(np.array(pil_img), cv2.COLOR_RGB2BGR)
    h, w, _ = bgr.shape
    roi_top = int(h * ROI_TOP_FRAC)
    roi_bot = int(h * ROI_BOT_FRAC)

    roi = bgr[roi_top:roi_bot, :]
    hsv = cv2.cvtColor(roi, cv2.COLOR_BGR2HSV)

    mask = _make_red_mask(hsv)
    M = cv2.moments(mask)
    area = M["m00"]

    conf = 0.0 if area <= 0 else float(min(1.0, area / (mask.size * 0.15)))

    if area < 1000:
        if save_debug:
            _save_debug(bgr, w, roi_top, roi_bot, None, None, mask)
        return LaneObs(False, 0.0, 0.0, conf)

    cx = M["m10"] / area
    lateral = (cx - (w/2)) / (w/2)

    # Use fitLine for heading angle (line direction through all red pixels)
    pts = cv2.findNonZero(mask)
    fit, heading_deg = _fit_line_heading_deg(pts)

    if save_debug:
        cy_roi = float(pts[:,:,1].mean()) if pts is not None else (roi_bot - roi_top) * 0.5
        _save_debug(bgr, w, roi_top, roi_bot, (cx, roi_top + cy_roi), fit, mask)

    return LaneObs(True, float(lateral), heading_deg, conf)

def _iter_images(folder):
    """Collect all image files from a folder (common extensions)."""
    pats = ["*.jpg", "*.jpeg", "*.png", "*.JPG", "*.JPEG", "*.PNG"]
    paths = []
    for p in pats:
        paths.extend(glob.glob(os.path.join(folder, p)))
    return sorted(paths)

def run_batch(folder, save_debug=True, csv_out=os.path.join(DEBUG_DIR, "perception_batch_results.csv")):
    """Run red lane detection on all images in a folder and write results to CSV."""
    paths = _iter_images(folder)
    if not paths:
        print(f"[Batch] No images found in: {folder}")
        return
    print(f"[Batch] Found {len(paths)} images in {folder}")
    os.makedirs(os.path.dirname(csv_out), exist_ok=True)
    with open(csv_out, "w", newline="", encoding="utf-8") as f:
        w = csv.writer(f)
        w.writerow(["filename", "ok", "lateral", "heading_deg", "confidence"])
        for p in paths:
            try:
                pil = Image.open(p).convert("RGB")
                obs = detect_from_pil(pil, save_debug=save_debug)
                w.writerow([os.path.basename(p), int(obs.ok), f"{obs.lateral:.6f}", f"{obs.heading_deg:.3f}", f"{obs.confidence:.3f}"])
            except Exception as e:
                print(f"[Batch] Skip {p}: {e}")
    print(f"[Batch] Results saved to: {csv_out}")

# --- CLI (single image OR folder) ---
def main():
    parser = argparse.ArgumentParser()
    g = parser.add_mutually_exclusive_group(required=True)
    g.add_argument("--image", help="Path to a single test image (JPG/PNG)")
    g.add_argument("--folder", help="Folder containing images to process (batch)")
    parser.add_argument("--save_debug", action="store_true", help="Save debug overlay(s) to project/debug")
    args = parser.parse_args()

    if args.folder:
        run_batch(args.folder, save_debug=args.save_debug)
        return

    pil = Image.open(args.image).convert("RGB")
    obs = detect_from_pil(pil, save_debug=args.save_debug)
    print(f"ok={obs.ok}  lateral={obs.lateral:.3f}  heading_deg={obs.heading_deg:.1f}  conf={obs.confidence:.2f}")

if __name__ == "__main__":
    main()
