# perception_trackposition.py
# Detect lane position (lane_name + lane_offset) based on a fixed color order.
# AAGP track order (from outside to inside):
# Black | White(outside) | Red | White(center) | Green | Blue | White(inside) | Black
#
# PURPOSE:
#   Identifies which lane/color band the robot is currently driving on.
#   Used for track position awareness and potential future strategy decisions.
#
# DETECTION ALGORITHM:
#   1. Crop ROI to rows [60%, 80%] of image height (mid-lower portion)
#   2. Convert to HSV and classify each pixel as one of: Black/White/Red/Green/Blue
#   3. At the middle row of the ROI, check the pixel directly below image center
#   4. lane_name = the color at the center pixel
#   5. lane_offset = how far left/right the robot is within that color band [-1 to +1]
#
# WHITE DISAMBIGUATION:
#   The track has 3 white bands (outside/center/inside). To distinguish them,
#   check 30 pixels left and right of center and look at neighbor colors:
#     Neighbor = Red   → "CenterWhite"   (white band between red lanes)
#     Neighbor = Black → "OutsideWhite"  (outermost white boundary)
#     Neighbor = Blue  → "InsideWhite"   (innermost white boundary)
#     Otherwise        → "White"         (generic/unknown white)
#
# TRACK COLOR LAYOUT (lateral cross-section):
#   |<------ Outside ------>|<------- Inside ------->|
#   Black OutWhite  Red  CentWhite  Green  Blue  InWhite  Black
#
# OUTPUT (LaneResult):
#   lane_name   : String label (Black/OutsideWhite/Red/CenterWhite/Green/Blue/InsideWhite/White/Unknown)
#   lane_offset : Position within the detected band, -1 = at left edge, +1 = at right edge
#   confidence  : Fraction of the middle row that is the detected color (coverage)
#   ok          : Always True when called with valid image (no failure path currently)

from dataclasses import dataclass
from PIL import Image
import cv2, numpy as np, os, argparse, csv, glob
from typing import Tuple, Optional

@dataclass
class LaneResult:
    """
    Result of track position detection.

    Fields:
      ok          : True if detection ran without error
      lane_name   : Detected lane color/position label (see module docstring)
      lane_offset : Normalized offset within the detected lane band [-1..+1, left negative]
      confidence  : Coverage ratio of detected color in center row [0..1]
    """
    ok: bool
    lane_name: str
    lane_offset: float   # normalized offset within the detected lane band: -1..+1 (left negative)
    confidence: float    # simple coverage ratio: 0..1


# === HSV color thresholds ===
# Tuned for Unity's track colors under default lighting
COLOR_RANGES = {
    "Red1":  ((0,   100, 80),  (10,  255, 255)),   # Orange-red hue range
    "Red2":  ((170, 100, 80),  (180, 255, 255)),   # Magenta-red hue range (hue wrap)
    "Green": ((35,  80,  80),  (85,  255, 255)),   # Green hue
    "Blue":  ((95,  80,  80),  (140, 255, 255)),   # Blue hue
    "White": ((0,   0,  200),  (180, 40,  255)),   # High brightness, low saturation
    "Black": ((0,   0,  0),    (180, 255, 50)),    # Very low brightness
}

# === ROI: center-bottom of image (robot's near-field view) ===
ROI_TOP_FRAC  = 0.60   # Top boundary of detection region
ROI_BOT_FRAC  = 0.80   # Bottom boundary (above camera hood)

# === Debug directory ===
project_root = os.path.dirname(os.path.dirname(__file__))
DEBUG_DIR = os.path.join(project_root, "debug")
os.makedirs(DEBUG_DIR, exist_ok=True)


def classify_colors(hsv_roi: np.ndarray) -> np.ndarray:
    """
    Build per-color binary masks and assign each pixel to the class with the highest mask value.

    The Red1+Red2 masks are merged before argmax to handle the hue wrap-around.
    Order of classification priority: Black, White, Red, Green, Blue
    (argmax returns the first tied index, so ordering matters when masks overlap).

    Returns:
        2D numpy array of string labels (same shape as hsv_roi's first 2 dims)
    """
    masks = {}
    for name, (lo, hi) in COLOR_RANGES.items():
        masks[name] = cv2.inRange(hsv_roi, np.array(lo, np.uint8), np.array(hi, np.uint8))

    # Merge the two red hue ranges into a single "Red" mask
    masks["Red"] = cv2.bitwise_or(masks["Red1"], masks["Red2"])
    del masks["Red1"], masks["Red2"]

    # Stack all masks and assign each pixel to the class with the highest value
    order = ["Black", "White", "Red", "Green", "Blue"]
    stacked = np.stack([masks[c] for c in order], axis=-1)
    idx = np.argmax(stacked, axis=-1)  # Index of strongest class at each pixel

    # Map integer indices back to color name strings
    label_map = np.vectorize(lambda i: order[i])(idx)
    return label_map


def _put_label_top(img: np.ndarray, text: str, top_frac: float = 0.06) -> None:
    """
    Draw text near the top of the image with auto-scaling and a semi-transparent panel.
    Text uses white foreground + black outline over a 45% opacity black background.
    """
    h, w = img.shape[:2]
    font = cv2.FONT_HERSHEY_SIMPLEX
    base_scale = 1.0
    thick = 2

    # Auto-scale to fit image width
    (tw, th), _ = cv2.getTextSize(text, font, base_scale, thick)
    scale = min(1.2, max(0.6, (w - 40) / max(1, tw)))
    (tw, th), _ = cv2.getTextSize(text, font, scale, thick)

    baseline_y = max(th + 16, int(h * top_frac))

    # Semi-transparent background panel
    x0, y0 = 10, baseline_y - th - 10
    x1, y1 = x0 + tw + 20, baseline_y + 10
    x0 = max(0, x0); y0 = max(0, y0)
    x1 = min(w - 1, x1); y1 = min(h - 1, y1)
    roi = img[y0:y1, x0:x1].copy()
    panel = roi.copy()
    cv2.rectangle(panel, (0, 0), (x1 - x0, y1 - y0), (0, 0, 0), -1)
    cv2.addWeighted(panel, 0.45, roi, 0.55, 0, roi)  # 45% black panel
    img[y0:y1, x0:x1] = roi

    # Draw text
    org = (x0 + 10, baseline_y)
    cv2.putText(img, text, org, font, scale, (0, 0, 0), thick + 2, cv2.LINE_AA)  # black outline
    cv2.putText(img, text, org, font, scale, (255, 255, 255), thick, cv2.LINE_AA)


def _nearest_color_run(row_labels: np.ndarray, target_color: str, cx: int) -> Optional[Tuple[int, int]]:
    """
    Find the run (contiguous sequence) of target_color pixels in a row that is
    closest to the center x position cx.

    Returns:
        (start_x, end_x) of the nearest run, or None if no target_color pixels in row.
    """
    xs = np.where(row_labels == target_color)[0]
    if xs.size == 0:
        return None

    # Compress consecutive pixels into runs
    runs = []
    start = xs[0]
    prev = xs[0]
    for x in xs[1:]:
        if x == prev + 1:
            prev = x
        else:
            runs.append((start, prev))
            start = x
            prev = x
    runs.append((start, prev))

    # Return run whose center is closest to cx
    best = min(runs, key=lambda ab: abs((ab[0] + ab[1]) * 0.5 - cx))
    return best


def _draw_plus(img: np.ndarray, x: int, y: int, size: int = 6,
               color: Tuple[int, int, int] = (255, 0, 255), thickness: int = 2) -> None:
    """Draw a tilted cross (+) marker at (x, y). Default color: purple (BGR)."""
    cv2.drawMarker(img, (int(x), int(y)), color,
                   markerType=cv2.MARKER_TILTED_CROSS,
                   markerSize=size, thickness=thickness)


def detect_lane(pil_img: Image.Image, save_debug: bool = False) -> LaneResult:
    """
    Detect the lane color under the image center and estimate lateral offset.

    Algorithm:
      1. Crop ROI to [60%, 80%] of image height
      2. Classify each pixel by color using HSV thresholds
      3. Read the color at the center pixel of the middle row
      4. For white: check ±30px neighbors to identify which white band
      5. Compute lane_offset = position within the color band at center
      6. Compute confidence = fraction of center row in detected color

    Args:
        pil_img: PIL Image (RGB)
        save_debug: If True, save visualization to debug/ directory

    Returns:
        LaneResult with lane_name, lane_offset, confidence
    """
    bgr = cv2.cvtColor(np.array(pil_img), cv2.COLOR_RGB2BGR)
    h, w, _ = bgr.shape
    roi_top = int(h * ROI_TOP_FRAC)
    roi_bot = int(h * ROI_BOT_FRAC)
    roi = bgr[roi_top:roi_bot, :]
    hsv = cv2.cvtColor(roi, cv2.COLOR_BGR2HSV)

    # Classify all pixels to color labels
    label_map = classify_colors(hsv)

    # Inspect center pixel of the middle row
    mid_y = label_map.shape[0] // 2
    row = label_map[mid_y, :]

    cx = w // 2
    lane_color = row[cx]   # Color at image center

    # === Classify lane name ===
    lane_name = "Unknown"
    if lane_color == "Red":
        lane_name = "Red"
    elif lane_color == "Green":
        lane_name = "Green"
    elif lane_color == "Blue":
        lane_name = "Blue"
    elif lane_color == "White":
        # White disambiguation: check neighbor colors ±30px
        left  = row[max(cx - 30, 0)]
        right = row[min(cx + 30, w - 1)]
        if "Red" in (left, right):
            lane_name = "CenterWhite"   # Between red lanes
        elif "Black" in (left, right):
            lane_name = "OutsideWhite"  # Outermost boundary
        elif "Blue" in (left, right):
            lane_name = "InsideWhite"   # Innermost boundary
        else:
            lane_name = "White"         # Unknown which white band
    elif lane_color == "Black":
        lane_name = "Black"

    # === Compute lateral offset within the detected color band ===
    mask_lane = (row == lane_color)
    xs = np.where(mask_lane)[0]
    if xs.size > 5:
        left_x, right_x = xs[0], xs[-1]
        lane_cx = 0.5 * (left_x + right_x)   # Center of the color band
        # Normalize: 0 = centered in band, -1 = at left edge, +1 = at right edge
        lane_offset = (cx - lane_cx) / ((right_x - left_x) / 2.0 + 1e-6)
        confidence = float(xs.size) / float(w)
    else:
        lane_offset, confidence = 0.0, 0.0

    # === Debug output ===
    if save_debug:
        dbg = bgr.copy()
        cv2.rectangle(dbg, (0, roi_top), (w, roi_bot), (0, 0, 255), 2)   # ROI box
        cv2.line(dbg, (cx, roi_top), (cx, roi_bot), (0, 255, 0), 2)      # Center line
        txt = f"{lane_name}, offset={lane_offset:+.2f}"
        _put_label_top(dbg, txt, top_frac=0.06)

        # Mark left/right edges of the detected color at each row (every 5px)
        STEP_Y = 5
        target_color = lane_color
        for yy in range(0, label_map.shape[0], STEP_Y):
            row_labels = label_map[yy, :]
            run = _nearest_color_run(row_labels, target_color, cx)
            if run is None:
                continue
            lx, rx = run
            y_img = roi_top + yy
            _draw_plus(dbg, int(lx), y_img, size=8)  # left edge marker
            _draw_plus(dbg, int(rx), y_img, size=8)  # right edge marker

        # Save with incremental counter
        counter_path = os.path.join(DEBUG_DIR, "counter.txt")
        try:
            with open(counter_path, "r") as f:
                cnt = int(f.read().strip())
        except Exception:
            cnt = 0
        cnt += 1
        with open(counter_path, "w") as f:
            f.write(str(cnt))
        out_path = os.path.join(DEBUG_DIR, f"debug_lane_{cnt:06d}.jpg")
        cv2.imwrite(out_path, dbg)
        print(f"[TrackPosition] Saved {out_path}")

    return LaneResult(True, lane_name, float(lane_offset), float(confidence))


def _iter_images(folder: str):
    """Collect all image files from a folder."""
    patterns = ["*.jpg", "*.jpeg", "*.png", "*.JPG", "*.JPEG", "*.PNG"]
    paths = []
    for p in patterns:
        paths.extend(glob.glob(os.path.join(folder, p)))
    return sorted(paths)


def run_batch(folder: str, save_debug: bool = True,
              csv_out: str = os.path.join(DEBUG_DIR, "trackposition_results.csv")) -> None:
    """Run track position detection on all images in folder and write CSV."""
    paths = _iter_images(folder)
    if not paths:
        print(f"[Batch] No images found in {folder}")
        return
    with open(csv_out, "w", newline="", encoding="utf-8") as f:
        w = csv.writer(f)
        w.writerow(["filename", "ok", "lane_name", "lane_offset", "confidence"])
        for p in paths:
            try:
                pil = Image.open(p).convert("RGB")
            except Exception as e:
                print(f"[Batch] Skip {p}: load error: {e}")
                continue
            res = detect_lane(pil, save_debug=save_debug)
            w.writerow([
                os.path.basename(p), int(res.ok), res.lane_name,
                f"{res.lane_offset:.3f}", f"{res.confidence:.3f}"
            ])
    print(f"[Batch] Results saved: {csv_out}")


# --- CLI ---
def main():
    parser = argparse.ArgumentParser()
    g = parser.add_mutually_exclusive_group(required=True)
    g.add_argument("--image", help="Path to a single image")
    g.add_argument("--folder", help="Folder of images (batch)")
    parser.add_argument("--save_debug", action="store_true", help="Save debug overlays to /debug")
    args = parser.parse_args()

    if args.folder:
        run_batch(args.folder, save_debug=args.save_debug)
    else:
        pil = Image.open(args.image).convert("RGB")
        res = detect_lane(pil, save_debug=args.save_debug)
        print(res)


if __name__ == "__main__":
    main()
