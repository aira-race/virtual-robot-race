# status_Robot.py
# Maintains global state of the robot (e.g., waiting for start, running, etc.)
#
# PURPOSE:
#   A minimal shared state machine for the rule-based control pipeline.
#   The state is used to distinguish driving phases (waiting / straight / cornering).
#   Currently only WAITING_START and RUN_STRAIGHT are used by rule_based_input.py.
#   RUN_CORNER is reserved for future expansion (e.g., different driver parameters
#   on corners vs straights).
#
# USAGE:
#   from rule_based_algorithms import status_Robot
#   status_Robot.set_state(status_Robot.RUN_STRAIGHT)
#   if status_Robot.get_state() == status_Robot.WAITING_START:
#       ...
#
# DESIGN NOTE:
#   Using a module-level global (not a class) keeps the code simple.
#   Since rule_based_input.py loads this via 'from rule_based_algorithms import status_Robot',
#   the global robot_state is shared across all call sites within the same import context.

# State constants
WAITING_START = 0     # Waiting for the race start signal (before GO)
RUN_STRAIGHT = 1      # Currently running (general driving state after GO)
RUN_CORNER = 2        # Placeholder for future cornering-specific logic

# Global state variable (starts in WAITING_START at module load)
robot_state = WAITING_START

def get_state():
    """Returns the current global robot state (one of WAITING_START, RUN_STRAIGHT, RUN_CORNER)."""
    return robot_state

def set_state(state):
    """Sets the current global robot state. Use the constants above as values."""
    global robot_state
    robot_state = state
