# table_input.py (Robot1 version)
# ============================================================
# Table (CSV Playback) Input Module - Robot 1
# ============================================================
# PURPOSE: Reads pre-recorded drive/steer commands from a CSV file and
#          replays them to Unity, one row per update cycle (~20Hz).
#          This mode is used to replay a previous keyboard-controlled run
#          or to test specific command sequences.
#
# CSV FORMAT:
#   The input CSV must have at least these columns (case-insensitive):
#     drive_torque  - float, -1.0 to 1.0
#     steer_angle   - float, -0.524 to 0.524 rad
#   These match the columns in metadata.csv generated during a data-save run.
#
# USAGE:
#   1. Set MODE_NUM=2 in Robot1/robot_config.txt
#   2. Place table_input.csv in the Robot1/ directory
#   3. Run main.py - commands replay at 20Hz
#
# NOTE: robot_id = "R1" identifies this as the Robot 1 table controller.
# ============================================================

import pandas as pd
import os
import sys

# Module identification (used in log messages)
MODULE_SOURCE = "Robot1"
print(f"[table_input] Loaded from {MODULE_SOURCE}/")

# CSV file path: looks for table_input.csv in the Robot1/ directory
INPUT_CSV_FILE = os.path.join(os.path.dirname(os.path.abspath(__file__)), "table_input.csv")

# Module-level state (persists across get_latest_command() calls)
df = None          # Pandas DataFrame holding the CSV data
csv_loaded = False # True after CSV has been successfully loaded
current_index = 0  # Current row position in the CSV
robot_id = "R1"   # Robot identifier for WebSocket messages

# Latest command values (updated by advance_command())
driveTorque: float = 0.0
steerAngle: float = 0.0


def load_csv():
    """
    Load the CSV file into a Pandas DataFrame.
    Normalizes column names to lowercase.
    Returns True if successful, False if file not found or invalid format.
    Called lazily on first get_latest_command() call.
    """
    global df, csv_loaded

    if csv_loaded:
        return True  # Already loaded

    if not os.path.exists(INPUT_CSV_FILE):
        print(f"[TableInput] ERROR: CSV file not found: {INPUT_CSV_FILE}")
        return False

    try:
        df = pd.read_csv(INPUT_CSV_FILE)

        # Normalize column names to lowercase for flexible matching
        # (metadata.csv uses lowercase, but users might have other casings)
        df.columns = df.columns.str.lower()

        # Validate that the required columns are present
        required_columns = ['drive_torque', 'steer_angle']
        missing_columns = [col for col in required_columns if col not in df.columns]
        if missing_columns:
            print(f"[TableInput] ERROR: Missing required columns: {missing_columns}")
            print(f"[TableInput] Available columns: {list(df.columns)}")
            print(f"[TableInput] Expected columns: {required_columns}")
            print(f"[TableInput] Hint: Copy data from metadata.csv including 'drive_torque' and 'steer_angle' columns")
            return False

        csv_loaded = True
        print(f"[TableInput] Loaded {len(df)} command rows from CSV")
        print(f"[TableInput] Columns: {list(df.columns)}")
        return True
    except Exception as e:
        print(f"[TableInput] ERROR loading CSV: {e}")
        return False


def advance_command():
    """
    Read the next row from the CSV and update driveTorque/steerAngle.
    Increments current_index after each successful read.
    Returns True if a row was read, False if end of data was reached.
    """
    global current_index, driveTorque, steerAngle

    if not csv_loaded:
        if not load_csv():
            return False

    if current_index >= len(df):
        print(f"[TableInput] End of CSV data reached (index={current_index})")
        return False

    try:
        row = df.iloc[current_index]
        # Use lowercase column names (normalized in load_csv)
        driveTorque = float(row.get("drive_torque", 0.0))
        steerAngle = float(row.get("steer_angle", 0.0))
        current_index += 1
        return True
    except KeyError as e:
        print(f"[TableInput] ERROR: Column not found in row {current_index}: {e}")
        print(f"[TableInput] Available columns: {list(df.columns)}")
        return False
    except ValueError as e:
        print(f"[TableInput] ERROR: Invalid value in row {current_index}: {e}")
        print(f"[TableInput] Row data: {row.to_dict()}")
        return False
    except Exception as e:
        print(f"[TableInput] ERROR reading row {current_index}: {type(e).__name__}: {e}")
        return False


def get_latest_command():
    """
    Return the next control command from the CSV as a dict for WebSocket sending.
    Automatically advances to the next row on each call.
    When the CSV is exhausted, returns zero commands (drive=0, steer=0).
    Raises RuntimeError if the CSV cannot be loaded (file missing or invalid).
    """
    # Ensure CSV is loaded before accessing data
    if not csv_loaded and not load_csv():
        raise RuntimeError("[TableInput] Failed to load CSV. Check the error messages above.")

    # Advance to the next row; return zeros if end of data
    if not advance_command():
        return {
            "type": "control",
            "robot_id": robot_id,
            "driveTorque": 0.0,
            "steerAngle": 0.0,
        }

    return {
        "type": "control",
        "robot_id": robot_id,
        "driveTorque": round(float(driveTorque), 3),
        "steerAngle": round(float(steerAngle), 3),
    }


def reset():
    """Reset the CSV playback to the beginning (row 0)."""
    global current_index
    current_index = 0
    print("[TableInput] Reset to beginning of CSV")


def get_progress():
    """
    Return current playback progress as (current_index, total_rows).
    Used by main.py for progress logging every 100 commands.
    """
    total = len(df) if df is not None else 0
    return (current_index, total)
